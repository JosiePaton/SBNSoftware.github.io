677 bool FRFillNumuCC(const simb::MCTruth &mctruth,
678  const std::vector<art::Ptr<sim::MCTrack>> &mctracks,
679  const std::vector<geo::BoxBoundedGeo> &volumes,
680  TRandom &rand,
681  caf::SRFakeReco &fakereco) {
682  // Configuration -- TODO: make configurable?
683 
684  // Fiducial Volume
685  float xmin = 10.;
686  float xmax = 10.;
687  float ymin = 10.;
688  float ymax = 10.;
689  float zmin = 10.;
690  float zmax = 100.;
691 
692  // energy smearing
693  float hadron_smearing = 0.05;
694  float lepton_contained_smearing = 0.02;
695  std::function<float (float)> lepton_exiting_smearing = [](float length) {
696  float A = 0.102;
697  float B = 0.000612;
698  return -A * TMath::Log(B * length);
699  };
700 
701  // visible energy threshold
702  float hadronic_energy_threshold = 0.021; // GeV
703 
704  // length cut
705  float contained_length_cut = 50.; // cm
706  float exiting_length_cut = 100.; // cm
707 
708  fakereco.filled = false;
709 
710  // first check if neutrino exists
711  if (!mctruth.NeutrinoSet()) return false;
712 
713  // then check if fiducial
714  int cryo_index = -1;
715  for (unsigned i = 0; i < volumes.size(); i++) {
716  const geo::BoxBoundedGeo &vol = volumes[i];
717  geo::BoxBoundedGeo FV(vol.MinX() + xmin, vol.MaxX() - xmax, vol.MinY() + ymin, vol.MaxY() - ymax, vol.MinZ() + zmin, vol.MaxZ() - zmax);
718  if (FV.ContainsPosition(mctruth.GetNeutrino().Nu().Position().Vect())) {
719  cryo_index = i;
720  break;
721  }
722  }
723 
724  if (cryo_index == -1) return false;
725 
726  std::vector<geoalgo::AABox> aa_volumes;
727  const geo::BoxBoundedGeo &v = volumes.at(cryo_index);
728  aa_volumes.emplace_back(v.MinX(), v.MinY(), v.MinZ(), v.MaxX(), v.MaxY(), v.MaxZ());
729 
730  // look for CC lepton or a \pi^+/- which can "fake" a numu CC interaction
731 
732  int lepton_ind = -1;
733  // CC lepton
734  if (abs(mctruth.GetNeutrino().Nu().PdgCode()) == 14 && mctruth.GetNeutrino().CCNC() == 0) {
735  for (int i = 0; i < (int)mctracks.size(); i++) {
736  if (isFromNuVertex(mctruth, *mctracks[i]) && abs(mctracks[i]->PdgCode()) == 13 && mctracks[i]->Process() == "primary") {
737  if (lepton_ind == -1 || mctracks[lepton_ind]->Start().E() < mctracks[i]->Start().E()) {
738  lepton_ind = i;
739  }
740  }
741  }
742  }
743  // NC pion
744  else if (mctruth.GetNeutrino().CCNC() == 1) {
745  for (int i = 0; i < (int)mctracks.size(); i++) {
746  if (isFromNuVertex(mctruth, *mctracks[i]) && abs(mctracks[i]->PdgCode()) == 211 && mctracks[i]->Process() == "primary") {
747  if (lepton_ind == -1 || mctracks[lepton_ind]->Start().E() < mctracks[i]->Start().E()) {
748  lepton_ind = i;
749  }
750  }
751  }
752  }
753 
754  // no matching track -- no fake reco
755  if (lepton_ind == -1) return false;
756 
757  // Now set the "lepton"
758  caf::SRFakeRecoParticle fake_lepton;
759 
760  fake_lepton.pid = 13;
761  fake_lepton.contained = false;
762  for (const geo::BoxBoundedGeo &vol: volumes) {
763  if (vol.ContainsPosition(mctracks[lepton_ind]->Start().Position().Vect()) && vol.ContainsPosition(mctracks[lepton_ind]->Start().Position().Vect())) {
764  fake_lepton.contained = true;
765  }
766  }
767  fake_lepton.len = ContainedLength(mctracks[lepton_ind]->Start().Position().Vect(), mctracks[lepton_ind]->End().Position().Vect(), aa_volumes);
768  fake_lepton.costh = mctracks[lepton_ind]->Start().Position().Vect().CosTheta();
769 
770  // apply length cut
771  if (fake_lepton.contained && fake_lepton.len < contained_length_cut) return false;
772  if (!fake_lepton.contained && fake_lepton.len < exiting_length_cut) return false;
773 
774 
775  // smear the lepton energy
776  float smearing = fake_lepton.contained ? lepton_contained_smearing : lepton_exiting_smearing(fake_lepton.len);
777  float ke = (mctracks[lepton_ind]->Start().E() - PDGMass(mctracks[lepton_ind]->PdgCode())) / 1000. /* MeV -> GeV*/;
778  fake_lepton.ke = rand.Gaus(ke, smearing * ke);
779  fake_lepton.ke = std::max(fake_lepton.ke, 0.f);
780 
781  // get the hadronic state
782  std::vector<caf::SRFakeRecoParticle> hadrons;
783 
784  for (int i = 0; i < (int)mctracks.size(); i++) {
785  if (isFromNuVertex(mctruth, *mctracks[i]) // from this interaction
786  && (abs(mctracks[i]->PdgCode()) == 211 || abs(mctracks[i]->PdgCode()) == 321 || abs(mctracks[i]->PdgCode()) == 2212) // hadronic
787  && mctracks[i]->Process() == "primary" // primary
788  && i != lepton_ind // not the fake lepton
789  ) {
790  caf::SRFakeRecoParticle hadron;
791  hadron.pid = abs(mctracks[i]->PdgCode());
792  hadron.contained = false;
793  for (const geo::BoxBoundedGeo &vol: volumes) {
794  if (vol.ContainsPosition(mctracks[i]->Start().Position().Vect()) && vol.ContainsPosition(mctracks[i]->Start().Position().Vect())) {
795  hadron.contained = true;
796  }
797  }
798  hadron.len = ContainedLength(mctracks[i]->Start().Position().Vect(), mctracks[i]->Start().Position().Vect(), aa_volumes);
799  hadron.costh = mctracks[i]->Start().Position().Vect().CosTheta();
800 
801  float ke = (mctracks[i]->Start().E() - PDGMass(mctracks[i]->PdgCode())) / 1000. /* MeV -> GeV*/;
802  if (ke < hadronic_energy_threshold) continue;
803 
804  hadron.ke = rand.Gaus(ke, hadron_smearing * ke);
805  hadron.ke = std::max(hadron.ke, 0.f);
806 
807  hadrons.push_back(hadron);
808  }
809  }
810 
811  // total up the energy to get the reco neutrino energy
812  fakereco.nuE = fake_lepton.ke;
813  for (const caf::SRFakeRecoParticle &had: hadrons) fakereco.nuE += had.ke;
814  fakereco.nuE += PDGMass(13) / 1000.; // MeV -.> GeV
815 
816  // save the particles
817  fakereco.lepton = fake_lepton;
818  fakereco.hadrons = hadrons;
819 
820  // other info
821  TVector3 vertex = mctruth.GetNeutrino().Nu().Position().Vect();
822  fakereco.vtx.x = vertex.X();
823  fakereco.vtx.y = vertex.Y();
824  fakereco.vtx.z = vertex.Z();
825 
826  // signal
827  if (abs(mctruth.GetNeutrino().Nu().PdgCode()) == 14 && mctruth.GetNeutrino().CCNC() == 0) {
828  fakereco.wgt = 0.8;
829  }
830  // bkg
831  else {
832  fakereco.wgt = 1.;
833  }
834  fakereco.nhad = fakereco.hadrons.size();
835  fakereco.filled = true;
836 
837  return true;
838 }

959 float ContainedLength(const TVector3 &v0, const TVector3 &v1,
960  const std::vector<geoalgo::AABox> &boxes) {
961  static const geoalgo::GeoAlgo algo;
962 
963  // if points are the same, return 0
964  if ((v0 - v1).Mag() < 1e-6) return 0;
965 
966  // construct individual points
967  geoalgo::Point_t p0(v0);
968  geoalgo::Point_t p1(v1);
969 
970  // construct line segment
971  geoalgo::LineSegment line(p0, p1);
972 
973  double length = 0;
974 
975  // total contained length is sum of lengths in all boxes
976  // assuming they are non-overlapping
977  for (auto const &box: boxes) {
978  int n_contained = box.Contain(p0) + box.Contain(p1);
979  // both points contained -- length is total length (also can break out of loop)
980  if (n_contained == 2) {
981  length = (v1 - v0).Mag();
982  break;
983  }
984  // one contained -- have to find intersection point (which must exist)
985  if (n_contained == 1) {
986  auto intersections = algo.Intersection(line, box);
987  // Because of floating point errors, it can sometimes happen
988  // that there is 1 contained point but no "Intersections"
989  // if one of the points is right on the edge
990  if (intersections.size() == 0) {
991  // determine which point is on the edge
992  double tol = 1e-5;
993  bool p0_edge = algo.SqDist(p0, box) < tol;
994  bool p1_edge = algo.SqDist(p1, box) < tol;
995  assert(p0_edge || p1_edge);
996  // contained one is on edge -- can treat both as not contained
997  //
998  // In this case, no length
999  if ((p0_edge && box.Contain(p0)) || (box.Contain(p1) && p1_edge))
1000  continue;
1001  // un-contaned one is on edge -- treat both as contained
1002  else if ((p0_edge && box.Contain(p1)) || (box.Contain(p0) && p1_edge)) {
1003  length = (v1 - v0).Mag();
1004  break;
1005  }
1006  else {
1007  assert(false); // bad
1008  }
1009  }
1010  // floating point errors can also falsely cause 2 intersection points
1011  //
1012  // in this case, one of the intersections must be very close to the
1013  // "contained" point, so the total contained length will be about
1014  // the same as the distance between the two intersection points
1015  else if (intersections.size() == 2) {
1016  length += (intersections.at(0).ToTLorentzVector().Vect() - intersections.at(1).ToTLorentzVector().Vect()).Mag();
1017  continue;
1018  }
1019  // "Correct"/ideal case -- 1 intersection point
1020  else if (intersections.size() == 1) {
1021  // get TVector at intersection point
1022  TVector3 int_tv(intersections.at(0).ToTLorentzVector().Vect());
1023  length += ( box.Contain(p0) ? (v0 - int_tv).Mag() : (v1 - int_tv).Mag() );
1024  }
1025  else assert(false); // bad
1026  }
1027  // none contained -- either must have zero or two intersections
1028  if (n_contained == 0) {
1029  auto intersections = algo.Intersection(line, box);
1030  if (!(intersections.size() == 0 || intersections.size() == 2)) {
1031  // more floating point error fixes...
1032  //
1033  // figure out which points are near the edge
1034  double tol = 1e-5;
1035  bool p0_edge = algo.SqDist(p0, box) < tol;
1036  bool p1_edge = algo.SqDist(p1, box) < tol;
1037  // and which points are near the intersection
1038  TVector3 vint = intersections.at(0).ToTLorentzVector().Vect();
1039 
1040  bool p0_int = (v0 - vint).Mag() < tol;
1041  bool p1_int = (v1 - vint).Mag() < tol;
1042  // exactly one of them should produce the intersection
1043  assert((p0_int && p0_edge) != (p1_int && p1_edge));
1044  // void variables when assert-ions are turned off
1045  (void) p0_int; (void) p1_int;
1046
