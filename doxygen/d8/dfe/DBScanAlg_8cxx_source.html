71 //
72 // Roughly quivalent to what FindNeighbors was doing before, except
73 // that it doesn't handle dead wires
74 struct AcceptEllipse {
75  const BoundingBox& m_bound;
76  double r[2];
77  double c[2]; ///< center of the bounding box
78  double d[2]; ///< half
79  explicit AcceptEllipse(const BoundingBox& b, double r1, double r2) : m_bound(b), r(), c()
80  {
81  r[0] = r1;
82  r[1] = r2;
83  c[0] = (m_bound.edges[0].second + m_bound.edges[0].first) / 2.0;
84  c[1] = (m_bound.edges[1].second + m_bound.edges[1].first) / 2.0;
85  d[0] = (m_bound.edges[0].second - m_bound.edges[0].first) / 2.0;
86  d[1] = (m_bound.edges[1].second - m_bound.edges[1].first) / 2.0;
87  }
88  bool
89  operator()(const RTree::Node* const node) const
90  {
91  // At the node level we use a rectangualr overlap condition
92  return m_bound.overlaps(node->bound);
93  }
94  bool
95  operator()(const RTree::Leaf* const leaf) const
96  {
97  // At the leaf level we have to more careful
98  double C[2], D[2];
99  C[0] = (leaf->bound.edges[0].second + leaf->bound.edges[0].first) / 2.0;
100  C[1] = (leaf->bound.edges[1].second + leaf->bound.edges[1].first) / 2.0;
101  D[0] = (leaf->bound.edges[0].second - leaf->bound.edges[0].first) / 2.0;
102  D[1] = (leaf->bound.edges[1].second - leaf->bound.edges[1].first) / 2.0;
103  double t = 0;
104  for (int i = 0; i < 2; ++i) {
105  // This is only approximate, it will accept a few classes of
106  // non-overlapping ellipses
107  t += ((c[i] - C[i]) * (c[i] - C[i])) / ((d[i] + D[i]) * (d[i] + D[i]));
108  }
109  return (t < 1);
110  }
111 
112 private:
113  static const BoundingBox EmptyBoundingBox; // for uninitialized bounds
114 
115  AcceptEllipse() : m_bound(EmptyBoundingBox), r(), c()
116  {
117  r[0] = r[1] = 1.0;
118  c[0] = (m_bound.edges[0].second - m_bound.edges[0].first) / 2.0;
119  c[1] = (m_bound.edges[1].second - m_bound.edges[1].first) / 2.0;
120  }

127 //
128 // Exactly equivalent to what FindNeighbors was doing before (assuming
129 // that points are preresented with no width in wire-space and with
130 // width in time-space
131 //
132 // We're going to make this work with nodal bounding boxes by
133 // accepting on center-inside the box or comparing to the nearest
134 // point on the edge using the point-to-point comparison function and
135 // a the maximum time-width.
136 struct AcceptFindNeighbors {
137  const BoundingBox& fBound;
138  double fEps[2];
139  double fMaxWidth;
140  double fWireDist;
141  std::vector<unsigned int>& fBadWireSum;
142  AcceptFindNeighbors(const BoundingBox& b,
143  double eps,
144  double eps2,
145  double maxWidth,
146  double wireDist,
147  std::vector<unsigned int>& badWireSum)
148  : fBound(b), fEps(), fMaxWidth(maxWidth), fWireDist(wireDist), fBadWireSum(badWireSum)
149  {
150  fEps[0] = eps;
151  fEps[1] = eps2;
152  }
153  // return a point-like box at the center of b
154  BoundingBox
155  center(const BoundingBox& b) const
156  {
157  BoundingBox c;
158  c.edges[0].first = c.edges[0].second = (b.edges[0].first + b.edges[0].second) / 2.0;
159  c.edges[1].first = c.edges[1].second = (b.edges[1].first + b.edges[1].second) / 2.0;
160  return c;
161  }
162  BoundingBox
163  center() const

165  return center(fBound);
166  }
167  // Here we implement the findNeighbor algorithm from point to point
168  bool
169  isNear(const BoundingBox& b) const
170  {
171  // Precomupation of a few things...box centers, wire bridging
172  // quantities, etc...
173  double bCenter0 = center(b).edges[0].first;
174  double bCenter1 = center(b).edges[1].first;
175  double tCenter0 = center().edges[0].first; // "t" is for test-point
176  double tCenter1 = center().edges[1].first;
177  // widths in the time direction
178  double bWidth = std::abs(b.edges[1].second - b.edges[1].first);
179  double tWidth = std::abs(fBound.edges[1].second - fBound.edges[1].first);
180  // bad channel counting
181  unsigned int wire1 = (unsigned int)(tCenter0 / fWireDist + 0.5);
182  unsigned int wire2 = (unsigned int)(bCenter0 / fWireDist + 0.5);
183  // Clamp the wize number to something resonably.
184  ///\todo activating these should throw a warning or something
185  if (wire1 < fBadWireSum.size()) wire1 = fBadWireSum.size();
186  if (wire2 < fBadWireSum.size()) wire2 = fBadWireSum.size();
187  // The getSimilarity[2] wirestobridge calculation is asymmetric,
188  // but is plugged into the cache symmetrically.I am assuming that
189  // this is OK because the wires that are hit cannot be bad.
190  unsigned int wirestobridge = util::absDiff(fBadWireSum[wire1], fBadWireSum[wire2]);
191  double cmtobridge = wirestobridge * fWireDist;
192 
193  double sim = std::abs(tCenter0 - bCenter0) - cmtobridge;
194  sim *= sim; // square it
195 
196  if (std::abs(tCenter0 - bCenter0) > 1e-10) {
197  cmtobridge *= std::abs((tCenter1 - bCenter1) / (tCenter0 - bCenter0));
198  }
199  double sim2 = std::abs(tCenter1 - bCenter1) - cmtobridge;
200  sim2 *= sim2; // square it
201 
202  double k = 0.1;
203  double WFactor = (exp(4.6 * ((tWidth * tWidth) + (bWidth * bWidth)))) * k; // ??
204  // We clamp WFactor on [ 1.0, 6.25 ]
205  if (WFactor < 1.0) WFactor = 1.0;
206  if (WFactor > 6.25) WFactor = 6.25;
207 
208  // Now we implement the test...see FindNeighbors
209  return (((sim) / (fEps[0] * fEps[0])) + ((sim2) / (fEps[1] * fEps[1] * (WFactor))) <= 1);
210  }
211  BoundingBox
212  nearestPoint(const BoundingBox& b) const
213  {
214  BoundingBox n;
215  BoundingBox c = center();
216  for (int i = 0; i < 2; ++i) {
217  // The work for finding the nearest point is the same in both
218  // dimensions
219  if (b.edges[i].first > c.edges[i].second) {
220  // Our point is lower than the low edge of the box
221  n.edges[i].first = n.edges[i].second = b.edges[i].first;
222  }
223  else if (b.edges[0].second < c.edges[0].first) {
224  // Our point is higher than the high edge of the box
225  n.edges[i].first = n.edges[i].second = b.edges[i].second;
226  }
227  else {
228  // In this dimension our point lies within the boxes bounds
229  n.edges[i].first = n.edges[i].second = c.edges[i].first;
230  }
231  }
232  // Now give the time dimension a width
233  n.edges[1].first -= fMaxWidth / 2.0;
234  n.edges[1].second += fMaxWidth / 2.0;
235  return n;
236  }
237  bool
238  operator()(const RTree::Node* const node) const
239  {
240  // if the our point overlaps the bounding box, accept immediately
241  if (fBound.overlaps(node->bound)) return true;
242  // No overlap, so compare to the nearest point on the bounding box
243  // under the assumption that the maximum width applies for that
244  // point
245  return isNear(nearestPoint(node->bound));
246  }
247  bool
248  operator()(const RTree::Leaf* const leaf) const
249  {
250  return isNear(leaf->bound);
251  }

273 cluster::DBScanAlg::InitScan(const detinfo::DetectorClocksData& clockData,
274  const detinfo::DetectorPropertiesData& detProp,

276  std::set<uint32_t> badChannels,
277  const std::vector<geo::WireID>& wireids)

279  if (wireids.size() && wireids.size() != allhits.size()) {

281  << " wireids size = " << wireids.size() << " do not match\n";
282  }
283  // clear all the data member vectors for the new set of hits
284  fps.clear();
285  fpointId_to_clusterId.clear();
286  fnoise.clear();
287  fvisited.clear();
288  fsim.clear();
289  fsim2.clear();
290  fsim3.clear();
291  fclusters.clear();
292  fWirePitch.clear();
293 
294  fBadChannels = badChannels;
295  fBadWireSum.clear();
296 
297  // Clear the RTree
298  fRTree.Remove(RTree::AcceptAny(), RTree::RemoveLeaf());
299  // and the bounds list
300  fRect.clear();
301 
302  //------------------------------------------------------------------
303  // Determine spacing between wires (different for each detector)
304  ///get 2 first wires and find their spacing (wire_dist)
305 
306  art::ServiceHandle<geo::Geometry const> geom;
307 
308  for (size_t p = 0; p < geom->Nplanes(); ++p)
309  fWirePitch.push_back(geom->WirePitch(p));
310 
311  // Collect the bad wire list into a useful form
312  if (fClusterMethod) { // Using the R*-tree
313  fBadWireSum.resize(geom->Nchannels());
314  unsigned int count = 0;
315  for (unsigned int i = 0; i < fBadWireSum.size(); ++i) {
316  count += fBadChannels.count(i);
317  fBadWireSum[i] = count;
318  }
319  }
320 
321  // Collect the hits in a useful form,
322  // and take note of the maximum time width
323  fMaxWidth = 0.0;
324  for (unsigned int j = 0; j < allhits.size(); ++j) {
325  int dims = 3; //our point is defined by 3 elements:wire#,center of the hit, and the hit width
326  std::vector<double> p(dims);
327 
328  double tickToDist = detProp.DriftVelocity(detProp.Efield(), detProp.Temperature());
329  tickToDist *= 1.e-3 * sampling_rate(clockData); // 1e-3 is conversion of 1/us to 1/ns
330  if (!wireids.size())
331  p[0] = (allhits[j]->WireID().Wire) * fWirePitch[allhits[j]->WireID().Plane];
332  else
333  p[0] = (wireids[j].Wire) * fWirePitch[allhits[j]->WireID().Plane];
334  p[1] = allhits[j]->PeakTime() * tickToDist;
335  p[2] = 2. * allhits[j]->RMS() * tickToDist; //width of a hit in cm
336 
337  // check on the maximum width condition
338  if (p[2] > fMaxWidth) fMaxWidth = p[2];
339 
340  fps.push_back(p);
341 
342  if (fClusterMethod) { // Using the R*-tree
343  // Convert these same values into dbsPoints to feed into the R*-tree
344  dbsPoint pp(p[0], p[1], 0.0, p[2] / 2.0); // note dividing by two
345  fRTree.Insert(j, pp.bounds());
346  // Keep a parallel list already made up. We could use fps instead, but...
347  fRect.push_back(pp);
348  }
349  }
350 
351  fpointId_to_clusterId.resize(fps.size(), kNO_CLUSTER); // Not zero as before!
352  fnoise.resize(fps.size(), false);
353  fvisited.resize(fps.size(), false);
354 
355  if (fClusterMethod) { // Using the R*-tree
356  Visitor visitor = fRTree.Query(RTree::AcceptAny(), Visitor());
357  mf::LogInfo("DBscan") << "InitScan: hits RTree loaded with " << visitor.count << " items.";
358  }
359  mf::LogInfo("DBscan") << "InitScan: hits vector size is " << fps.size();
360 
361  return;
362 }

546 // This is the algorithm that finds clusters:
547 //
548 // DWM's implementation of DBScanAlg as much like the paper as possible
549 void
550 cluster::DBScanAlg::run_dbscan_cluster()
551 {
552  unsigned int cid = 0;
553  // foreach pid
554  for (size_t pid = 0; pid < fps.size(); pid++) {
555  // not already visited
556  if (fpointId_to_clusterId[pid] == kNO_CLUSTER) {
557  if (ExpandCluster(pid, cid)) { cid++; }
558  } // if (!visited
559  } // for
560  // END DBSCAN
561 
562  // Construct clusters, count noise, etc..
563  int noise = 0;
564  fclusters.resize(cid);
565  for (size_t y = 0; y < fpointId_to_clusterId.size(); ++y) {
566  if (fpointId_to_clusterId[y] == kNO_CLUSTER) {
567  // This shouldn't happen...all points should be clasified by now!
568  mf::LogWarning("DBscan") << "Unclassified point!";
569  }
570  else if (fpointId_to_clusterId[y] == kNOISE_CLUSTER) {
571  ++noise;
572  }
573  else {
574  unsigned int c = fpointId_to_clusterId[y];
575  if (c >= cid) {
576  mf::LogWarning("DBscan") << "Point in cluster " << c << " when only " << cid
577  << " clusters wer found [0-" << cid - 1 << "]";
578  }
579  fclusters[c].push_back(y);
580  }

631 bool
632 cluster::DBScanAlg::ExpandCluster(unsigned int point, unsigned int clusterID)
633 {
634  /* GetSetOfPoints for point*/

636 
637  // not enough support -> mark as noise
638  if (seeds.size() < fMinPts) {
639  fpointId_to_clusterId[point] = kNOISE_CLUSTER;
640  return false;
641  }
642  else {
643  // Add to the currecnt cluster
644  fpointId_to_clusterId[point] = clusterID;
645  for (std::set<unsigned int>::iterator itr = seeds.begin(); itr != seeds.end(); itr++) {
646  fpointId_to_clusterId[*itr] = clusterID;
647  }
648  seeds.erase(point);
649  while (!seeds.empty()) {
650  unsigned int currentP = *(seeds.begin());
651  std::set<unsigned int> result = RegionQuery(currentP);
652 
653  if (result.size() >= fMinPts) {
654  for (std::set<unsigned int>::iterator itr = result.begin(); itr != result.end(); itr++) {
655  unsigned int resultP = *itr;
656  // not already assigned to a cluster
657  if (fpointId_to_clusterId[resultP] == kNO_CLUSTER ||
658  fpointId_to_clusterId[resultP] == kNOISE_CLUSTER) {
659  if (fpointId_to_clusterId[resultP] == kNO_CLUSTER) { seeds.insert(resultP); }
660  fpointId_to_clusterId[resultP] = clusterID;
661  } // unclassified or noise
662  } // for
663  } // enough support
664  seeds.erase(currentP);
665  } // while
666  return true;
667  }
