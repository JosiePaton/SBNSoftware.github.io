415 void BasicWireAnalysis::findHitCandidates(Waveform::const_iterator startItr,
416  Waveform::const_iterator stopItr,
417  size_t roiStartTick,
418  size_t planeIdx,
419  HitCandidateVec& hitCandidateVec) const
420 {
421  // Search for candidate hits...
422  // The idea will be to find the largest deviation in the input derivative waveform as the starting point. Depending
423  // on if a maximum or minimum, we search forward or backward to find the minimum or maximum that our extremum
424  // corresponds to.
425  std::pair<Waveform::const_iterator, Waveform::const_iterator> minMaxPair = std::minmax_element(startItr, stopItr);
426 
427  Waveform::const_iterator maxItr = minMaxPair.second;
428  Waveform::const_iterator minItr = minMaxPair.first;
429 
430  // Use the larger of the two as the starting point and recover the nearest max or min
431  if (std::fabs(*maxItr) > std::fabs(*minItr)) minItr = findNearestMin(maxItr, stopItr);
432  else maxItr = findNearestMax(minItr,startItr);
433 
434  int deltaTicks = std::distance(maxItr,minItr);
435  float range = *maxItr - *minItr;
436 
437  fDeltaTicksHist.at(planeIdx)->Fill(deltaTicks, 1.);
438  fRangeHist.at(planeIdx)->Fill(range, 1.);
439 
440  // std::cout << "** max at tick: " << std::distance(startItr,maxItr) << ", val: " << *maxItr << ", min at tick: " << std::distance(startItr,minItr) << ", val: " << *minItr << ", delta: " << deltaTicks << ", range: " << range << std::endl;
441 
442  // At some point small rolling oscillations on the waveform need to be ignored...
443  if (deltaTicks >= fMinDeltaTicks.at(planeIdx) && range > fMinDeltaPeaks.at(planeIdx))
444  {
445  // Need to back up to find zero crossing, this will be the starting point of our
446  // candidate hit but also the endpoint of the pre sub-waveform we'll search next
447  Waveform::const_iterator newEndItr = findStartTick(maxItr, startItr);
448 
449  int startTick = std::distance(startItr,newEndItr);
450 
451  // Now need to go forward to again get close to zero, this will then be the end point
452  // of our candidate hit and the starting point for the post sub-waveform to search
453  Waveform::const_iterator newStartItr = findStopTick(minItr, stopItr);
454 
455  int stopTick = std::distance(startItr,newStartItr);
456 
457  // Find hits in the section of the waveform leading up to this candidate hit
458  if (startTick > 2) findHitCandidates(startItr,newEndItr,roiStartTick,planeIdx,hitCandidateVec);
459 
460  // Create a new hit candidate and store away
461  HitCandidate_t hitCandidate;
462 
463  Waveform::const_iterator peakItr = std::min_element(maxItr,minItr,[](const auto& left, const auto& right){return std::fabs(left) < std::fabs(right);});
464 
465  // Check balance
466  if (2 * std::distance(peakItr,minItr) < std::distance(maxItr,peakItr)) peakItr--;
467  else if (2 * std::distance(maxItr,peakItr) < std::distance(peakItr,minItr)) peakItr++;
468 
469  hitCandidate.startTick = roiStartTick + startTick;
470  hitCandidate.stopTick = roiStartTick + stopTick;
471  hitCandidate.maxTick = roiStartTick + std::distance(startItr,maxItr);
472  hitCandidate.minTick = roiStartTick + std::distance(startItr,minItr);
473  hitCandidate.maxDerivative = maxItr != stopItr ? *maxItr : 0.;
474  hitCandidate.minDerivative = minItr != stopItr ? *minItr : 0.;
475  hitCandidate.hitCenter = roiStartTick + std::distance(startItr,peakItr) + 0.5;
476  hitCandidate.hitSigma = 0.5 * float(hitCandidate.minTick - hitCandidate.maxTick);
477  hitCandidate.hitHeight = hitCandidate.hitSigma * (hitCandidate.maxDerivative - hitCandidate.minDerivative) / 1.2130;
478 
479  hitCandidateVec.push_back(hitCandidate);
480 
481  // Finally, search the section of the waveform following this candidate for more hits
482  if (std::distance(newStartItr,stopItr) > 2) findHitCandidates(newStartItr,stopItr,roiStartTick + stopTick,planeIdx,hitCandidateVec);
483  }
484 
485  return;
486 }

488 void BasicWireAnalysis::findHitCandidates(Waveform::const_iterator eroStartItr,
489  Waveform::const_iterator eroStopItr,
490  Waveform::const_iterator diffStartItr,
491  Waveform::const_iterator diffStopItr,
492  size_t roiStartTick,
493  size_t planeIdx,
494  HitCandidateVec& hitCandidateVec) const
495 {
496  // Search for candidate hits...
497  // First task is to recover the maximum and minimum difference and reject waveform if no chance for an actual hit
498  std::pair<Waveform::const_iterator,Waveform::const_iterator> diffMinMaxPair = std::minmax_element(diffStartItr,diffStopItr);
499 
500  float deltaDiff = *diffMinMaxPair.second - *diffMinMaxPair.first;
501 
502  if (deltaDiff < 7) return;
503 
504  // The idea will be to find the largest deviation in the input derivative waveform as the starting point. Depending
505  // on if a maximum or minimum, we search forward or backward to find the minimum or maximum that our extremum
506  // corresponds to.
507  Waveform::const_iterator peakLeftItr = std::max_element(eroStartItr,eroStopItr);
508  Waveform::const_iterator peakRightItr = peakLeftItr;
509  Waveform::const_iterator diffLeftItr = diffStartItr;
510 
511  std::advance(diffLeftItr, std::distance(eroStartItr,peakLeftItr));
512 
513  Waveform::const_iterator diffRightItr = diffLeftItr;
514 
515  // Look for the case of a large waveform, handle differently
516  if (*diffLeftItr < *peakLeftItr)
517  {
518  // Find the position where the erosion vector is clearly less than the difference
519  // should this be a ratio test?
520  while(std::distance(eroStartItr,peakLeftItr) > 0 && *diffLeftItr - 1. < *peakLeftItr) {diffLeftItr--; peakLeftItr--;}
521  while(std::distance(peakRightItr,eroStopItr) > 0 && *diffRightItr - 1. < *peakRightItr) {diffRightItr++; peakRightItr++;}
522  }
523 
524  // Find maximum in difference to left of peak
525  Waveform::const_iterator leftMaxItr = diffLeftItr;
526  float leftMaxVal = *leftMaxItr;
527 
528  while(std::distance(diffStartItr,leftMaxItr) > 0 && *(leftMaxItr - 1) >= leftMaxVal) leftMaxVal = *(--leftMaxItr);
529 
530  // Find the rise point in the difference distribution
531  Waveform::const_iterator leftRiseItr = leftMaxItr;
532  float leftRiseVal = *leftRiseItr;
533 
534  while(std::distance(diffStartItr,leftRiseItr) >= 0 && *(leftRiseItr - 1) < leftRiseVal) leftRiseVal = *(--leftRiseItr);
535 
536  // Switch gears and look to the right of the peak
537  Waveform::const_iterator rightMaxItr = diffRightItr;
538  float rightMaxVal = *rightMaxItr;
539 
540  while(std::distance(rightMaxItr,diffStopItr) > 0 && *(rightMaxItr + 1) >= rightMaxVal) rightMaxVal = *(++rightMaxItr);
541 
542  // Find the rise point in the difference distribution
543  Waveform::const_iterator rightRiseItr = rightMaxItr;
544  float rightRiseVal = *rightRiseItr;
545 
546  while(std::distance(rightRiseItr,diffStopItr) > 0 && *(rightRiseItr + 1) < rightRiseVal) rightRiseVal = *(++rightRiseItr);
547 
548  // Find hits in the section of the waveform leading up to this candidate hit
549  if (std::distance(diffStartItr,leftRiseItr) > 4)
550  {
551  Waveform::const_iterator newEroStopItr = eroStartItr;
552 
553  std::advance(newEroStopItr,std::distance(diffStartItr,leftRiseItr));
554 
555  findHitCandidates(eroStartItr,newEroStopItr,diffStartItr,leftRiseItr,roiStartTick,planeIdx,hitCandidateVec);
556  }
557 
558  // Fill the data structure
559  HitCandidate_t hitCandidate;
560 
561  hitCandidate.startTick = roiStartTick + std::distance(diffStartItr,leftRiseItr);
562  hitCandidate.stopTick = roiStartTick + std::distance(diffStartItr,rightRiseItr);
563  hitCandidate.maxTick = roiStartTick + std::distance(diffStartItr,leftMaxItr);
564  hitCandidate.minTick = roiStartTick + std::distance(diffStartItr,rightMaxItr);
565  hitCandidate.maxDerivative = *leftMaxItr;
566  hitCandidate.minDerivative = *rightMaxItr;
567  hitCandidate.hitCenter = roiStartTick + (std::distance(eroStartItr,peakLeftItr) + std::distance(eroStartItr,peakRightItr) + 0.5)/2;
568  hitCandidate.hitSigma = 0.5 * float(hitCandidate.minTick - hitCandidate.maxTick);
569  hitCandidate.hitHeight = hitCandidate.hitSigma * (hitCandidate.maxDerivative - hitCandidate.minDerivative) / 1.2130;
570 
571  hitCandidateVec.push_back(hitCandidate);
572 
573  // Finally, search the section of the waveform following this candidate for more hits
574  if (std::distance(rightRiseItr,diffStopItr) > 4)
575  {
576  Waveform::const_iterator newEroStartItr = eroStartItr;
577  int newStartTick = std::distance(diffStartItr,rightRiseItr);
578 
579  std::advance(newEroStartItr, newStartTick);
580 
581  findHitCandidates(newEroStartItr,eroStopItr,rightRiseItr,diffStopItr,roiStartTick + newStartTick,planeIdx,hitCandidateVec);
582  }
583 
584  return;
585 }

633 BasicWireAnalysis::Waveform::const_iterator BasicWireAnalysis::findStartTick(Waveform::const_iterator maxItr, Waveform::const_iterator startItr) const
634 {
635  Waveform::const_iterator lastItr = maxItr;
636 
637  // If we can't back up then there is nothing to do
638  if (std::distance(startItr,lastItr) > 0)
639  {
640  // In theory, we are starting at a maximum and want to find the "start" of the candidate peak
641  // Ideally we would look to search backward to the point where the (derivative) waveform crosses zero again.
642  // However, the complication is that we need to watch for the case where two peaks are merged together and
643  // we might run through another peak before crossing zero...
644  // So... loop until we hit the startItr...
645  Waveform::const_iterator loopItr = lastItr - 1;
646 
647  while(loopItr != startItr)
648  {
649  // Ideal world case, we cross zero... but we might encounter a minimum... or an inflection point
650  if (*loopItr < 0. || !(*loopItr < *lastItr)) break;
651 
652  lastItr = loopItr--;
653  }
654  }
655  else lastItr = startItr;
656 
657  return lastItr;
658 }
