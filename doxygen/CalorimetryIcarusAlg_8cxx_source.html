161  // ----------------------------------------------------------------------------------//
162  double calo::CalorimetryIcarusAlg::LifetimeCorrection(detinfo::DetectorClocksData const& clockData,
163  detinfo::DetectorPropertiesData const& detProp,
164  double time, double T0) const
165  {
166  float t = time;
167 
168  double timetick = sampling_rate(clockData)*1.e-3; //time sample in microsec
169  double presamplings = trigger_offset(clockData);
170 
171  t -= presamplings;
172  time = t * timetick - T0*1e-3; // (in microsec)
173 
174  if (fLifeTimeForm==0){
175  //Exponential form
176  double tau = detProp.ElectronLifetime();
177 
178  double correction = exp(time/tau);
179  return correction;
180  }
181  else if (fLifeTimeForm==1){
182  //Exponential+constant form
183  const lariov::ElectronLifetimeProvider& elifetime_provider = art::ServiceHandle<lariov::ElectronLifetimeService const>()->GetProvider();
184  double correction = elifetime_provider.Lifetime(time);
185  //std::cout<<correction<<std::endl;
186  return correction;
187  }
188  else{
189  std::cout << "Unknow CaloLifeTimeForm "<<fLifeTimeForm<<std::endl;
190 return 0;
191  }
192  }
