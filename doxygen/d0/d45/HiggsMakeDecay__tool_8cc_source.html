96 double higgs_momentum(double kaon_mass, double pion_mass, double higs_mass) {
97  return sqrt(kaon_mass * kaon_mass * kaon_mass * kaon_mass
98  -2 * kaon_mass * kaon_mass * pion_mass * pion_mass

106 // Get the partial width for lepton decays
107 double LeptonPartialWidth(double lep_mass, double higs_mass, double mixing) {
108  if (lep_mass * 2 >= higs_mass) return 0.;
109 
110  double higgs_vev = Constants::Instance().higgs_vev;
111 
112  double width = (mixing * mixing * lep_mass * lep_mass * higs_mass / (8 * M_PI * higgs_vev * higgs_vev)) * pow(1 - 4 * lep_mass * lep_mass / (higs_mass * higs_mass), 3. / 2.);
113  return width;
114 }

124 double PionPartialWidth(double pion_mass, double higs_mass, double mixing) {
125  if (pion_mass * 2 >= higs_mass) return 0.;
126 
127  double higgs_vev = Constants::Instance().higgs_vev;
128 
129  double form_factor = (2. / 9.) * higs_mass * higs_mass + (11. / 9.) * pion_mass * pion_mass;
130 
131  double width = (mixing * mixing * 3 * form_factor * form_factor / (32 * M_PI * higgs_vev * higgs_vev * higs_mass)) * pow(1- 4. * pion_mass * pion_mass / (higs_mass * higs_mass), 1. / 2.);
132 
133  return width;
134 }

157 void HiggsMakeDecay::configure(fhicl::ParameterSet const &pset)
158 {
159  fReferenceRayLength = pset.get<double>("ReferenceRayLength", -1);
160  fReferenceRayDistance = pset.get<double>("ReferenceRayDistance", 0.);
161  fReferenceHiggsMass = pset.get<double>("ReferenceHiggsMass", -1);
162  fReferenceHiggsMixing = pset.get<double>("ReferenceHiggsMixing", -1);
163  fReferenceHiggsEnergy = pset.get<double>("ReferenceHiggsEnergy", -1);
164  fReferenceHiggsKaonEnergy = pset.get<double>("ReferenceHiggsEnergyFromKaonEnergy", -1.);
165 
166  if (fReferenceHiggsEnergy < 0. && fReferenceHiggsKaonEnergy > 0.) {
167  fReferenceHiggsEnergy = std::min(forwardPrtlEnergy(Constants::Instance().kplus_mass, Constants::Instance().piplus_mass, fReferenceHiggsMass, fReferenceHiggsKaonEnergy),
168  forwardPrtlEnergy(Constants::Instance().klong_mass, Constants::Instance().pizero_mass, fReferenceHiggsMass, fReferenceHiggsKaonEnergy));
169  }
170 
171  // if configured to, divide out some of the decay weight
172  if (fReferenceRayLength > 0. && fReferenceHiggsMass > 0. && fReferenceHiggsMixing > 0. && fReferenceHiggsEnergy > 0.) {
173  // Get each partial width
174  double width_elec = ElectronPartialWidth(fReferenceHiggsMass, fReferenceHiggsMixing);
175  double width_muon = MuonPartialWidth(fReferenceHiggsMass, fReferenceHiggsMixing);
176  double width_piplus = PiPlusPartialWidth(fReferenceHiggsMass, fReferenceHiggsMixing);
177  double width_pizero = PiZeroPartialWidth(fReferenceHiggsMass, fReferenceHiggsMixing);
178 
179  // total lifetime
180  double lifetime_ns = Constants::Instance().hbar / (width_elec + width_muon + width_piplus + width_pizero);
181 
182  // multiply by gamma*v to get the length
183  double gamma_v = sqrt(fReferenceHiggsEnergy * fReferenceHiggsEnergy - fReferenceHiggsMass * fReferenceHiggsMass) * Constants::Instance().c_cm_per_ns / fReferenceHiggsMass;
184  double mean_dist = lifetime_ns * gamma_v;
185 
186  // compute the decay weight
187  fMaxWeight = forcedecay_weight(mean_dist, fReferenceRayDistance, fReferenceRayDistance + fReferenceRayLength);
188  }
189  else {
190  fMaxWeight = -1.;
191  }
192 }

194 int HiggsMakeDecay::RandDaughter(double elec_width, double muon_width, double piplus_width, double pizero_width) {
195  double total_width = elec_width + muon_width + piplus_width + pizero_width;
196 
197  double flat_rand = CLHEP::RandFlat::shoot(fEngine, 0, 1.);
198 
199  if (flat_rand < elec_width / total_width) {
200  return 11;
201  }
202  else if (flat_rand < (elec_width + muon_width) / total_width) {
203  return 13;
204  }
205  else if (flat_rand < (elec_width + muon_width + piplus_width) / total_width) {
206  return 211;
207  }
208  else {
209  return 111;
210  }
211 
212  // unreachable
213  return -1;
214 }

216 bool HiggsMakeDecay::Decay(const MeVPrtlFlux &flux, const TVector3 &in, const TVector3 &out, MeVPrtlDecay &decay, double &weight) {
217  // Handle bad mass value
218  if (flux.mass < 2*Constants::Instance().elec_mass) {
219  throw cet::exception("HiggsMakeDecay Tool: BAD MASS. Configured mass (" + std::to_string(flux.mass) +
220  ") is smaller than lowest mass available decay e+e- (" + std::to_string(2*Constants::Instance().elec_mass) +")");
221  }
222 
223  double mixing = flux.C1; // constant-1 saves the mixing
224 
225  // Get each partial width
226  double width_elec = ElectronPartialWidth(flux.mass, mixing);
227  double width_muon = MuonPartialWidth(flux.mass, mixing);
228  double width_piplus = PiPlusPartialWidth(flux.mass, mixing);
229  double width_pizero = PiZeroPartialWidth(flux.mass, mixing);
230 
231  // total lifetime
232  double lifetime_ns = Constants::Instance().hbar / (width_elec + width_muon + width_piplus + width_pizero);
233 
234  // multiply by gamma*v to get the length
235  double mean_dist = lifetime_ns * flux.mom.Gamma() * flux.mom.Beta() * Constants::Instance().c_cm_per_ns;
236 
237  // distance inside detector
238  double in_dist = (flux.pos.Vect() - in).Mag();
239  double out_dist = (flux.pos.Vect() - out).Mag();
240 
241  // compute the decay weight
242  weight = forcedecay_weight(mean_dist, in_dist, out_dist);
243 
244  // if the weight is literally zero, then there is no way in hell
245  // that this higgs is going to decay in the detector -- reject it
246  if (weight == 0) {
247  return false;
248  }
249 
250  // get the decay location
251  double flat_rand = CLHEP::RandFlat::shoot(fEngine, 0, 1.);
252  double decay_rand = flat_to_exp_rand(flat_rand, mean_dist, in_dist, out_dist);
253  TVector3 decay_pos3 = flux.pos.Vect() + decay_rand * (in - flux.pos.Vect()).Unit();
254 
255  // decay time
256  double decay_time = TimeOfFlight(flux, decay_pos3);
257  TLorentzVector decay_pos(decay_pos3, decay_time);
258 
259  // get the decay type
260  int daughter_pdg = RandDaughter(width_elec, width_muon, width_piplus, width_pizero);
261 
262  double daughter_mass = TDatabasePDG::Instance()->GetParticle(daughter_pdg)->Mass();
263 
264  // daughter mom+energy in the parent rest-frame
265  double daughterE_HRF = flux.mass / 2.;
266  double daughterP_HRF = sqrt(daughterE_HRF * daughterE_HRF - daughter_mass * daughter_mass);
267 
268  // make the two daughters in the higgs rest-frame with a random direction
269  TVector3 pA_HRF = RandomUnitVector() * daughterP_HRF;
270 
271  TVector3 pB_HRF = -pA_HRF;
272 
273  TLorentzVector p4A = TLorentzVector(pA_HRF, daughterE_HRF);
274  TLorentzVector p4B = TLorentzVector(pB_HRF, daughterE_HRF);
275 
276  // Boost
277  p4A.Boost(flux.mom.BoostVector());
278  p4B.Boost(flux.mom.BoostVector());
279 
280  // save the decay info
281  decay.decay_width = width_elec + width_muon + width_piplus + width_pizero;
282  decay.mean_lifetime = lifetime_ns;
283  decay.mean_distance = mean_dist;
284 
285  decay.pos = decay_pos;
286 
287  decay.daughter_mom.push_back(p4A.Vect());
288  decay.daughter_e.push_back(p4A.E());
289  decay.daughter_pdg.push_back(daughter_pdg);
290 
291  decay.daughter_mom.push_back(p4B.Vect());
292  decay.daughter_e.push_back(p4B.E());
293  // daughter B is anti-particle
294  if (daughter_pdg == 111) { // pi0 is its own anti-particle
295  decay.daughter_pdg.push_back(daughter_pdg);
296  }
297  else {
298  decay.daughter_pdg.push_back(-daughter_pdg);
299  }
300 
301  return true;
302 }
