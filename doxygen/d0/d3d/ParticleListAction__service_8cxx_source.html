2 /// \file ParticleListAction.cxx
3 /// \brief Use Geant4's user "hooks" to maintain a list of particles generated by Geant4.
4 ///
5 /// \author seligman@nevis.columbia.edu
6 ///
7 /// Design considerations
8 /// ---------------------
9 ///
10 /// This class relies on the MCTruth index from
11 /// g4b::PrimaryParticleInformation to operate correctly. This index
12 /// is an integer value that corresponds to an MCTruth object, as
13 /// accessed through art::Handle<std::vector<simb::MCTruth>> objects.
14 /// However, the order in which MCTruth objects are processed must be
15 /// consistent between this service and the MCTruthEventAction
16 /// service, which creates the PrimaryParticleInformation object,
17 /// otherwise the Assns objects created here will be incorrect.
18 ///
19 /// Through art 3.09, one can rely on the order returned by a given
20 /// Event::getMany call to be predictable and consistent within the
21 /// same program. However, this behavior should not necessarily be
22 /// relied upon, and a different implementation of this class would
23 /// insulate users from such details, making the implementation
24 /// simpler. One should determine whether storing an art::ProductID
25 /// object along with an MCTruthIndex might be more helpful.
26 ///

239  // Create our initial simb::MCParticle object and add it to the sim::ParticleList.
240  void
241  ParticleListActionService::preUserTrackingAction(const G4Track* track)
242  {
243  // Particle type.
244  G4ParticleDefinition* particleDefinition = track->GetDefinition();
245  G4int pdgCode = particleDefinition->GetPDGEncoding();
246 
247  // Get Geant4's ID number for this track. This will be the same
248  // ID number that we'll use in the ParticleList.
249  // It is offset by the number of tracks accumulated from the previous Geant4
250  // runs (if any)
251  int const trackID = track->GetTrackID() + fTrackIDOffset;
252  fCurrentTrackID = trackID;
253 
254  // And the particle's parent (same offset as above):
255  int parentID = track->GetParentID() + fTrackIDOffset;
256 
257  std::string process_name = "unknown";
258  std::string mct_primary_process = "unknown";
259  bool isFromMCTProcessPrimary = false;
260 
261  // Is there an MCTruth object associated with this G4Track? We
262  // have to go up a "chain" of information to find out:
263  const G4DynamicParticle* dynamicParticle = track->GetDynamicParticle();
264  const G4PrimaryParticle* primaryParticle = dynamicParticle->GetPrimaryParticle();
265  simb::GeneratedParticleIndex_t primaryIndex = simb::NoGeneratedParticleIndex;
266  size_t primarymctIndex = 0;
267  if (primaryParticle != nullptr) {
268  const G4VUserPrimaryParticleInformation* gppi = primaryParticle->GetUserInformation();
269  const g4b::PrimaryParticleInformation* ppi =
270  dynamic_cast<const g4b::PrimaryParticleInformation*>(gppi);
271  if (ppi != nullptr) {
272  primaryIndex = ppi->MCParticleIndex();
273  primarymctIndex = ppi->MCTruthIndex();
274 
275  mct_primary_process = ppi->GetMCParticle()->Process();
276 
277  // If we've made it this far, a PrimaryParticleInformation
278  // object exists and we are using a primary particle, set the
279  // process name accordingly
280 
281  // -- D.R. : if process == "primary" exactly (this will most likely be the case), mark it as
282  // a primary with keepable trajectory for itself and its descendants.
283  // -- elsif it simply starts with "primary", accept it but mark it as non-keep
284  // -- else force it to be primary because we have determined that it is a primary particle
285  // This was the original behavior of the code i.e. process was _set_ to "primary"
286  // regardless of what the process name was in the generator MCTruth object.
287  //
288  // -- NOTE: This enforces a convention for process names assigned in the gen stage.
289  if (mct_primary_process.compare("primary") == 0) {
290  process_name = "primary";
291  isFromMCTProcessPrimary = true;
292  }
293  else if (mct_primary_process.find("primary") == 0) {
294  process_name = mct_primary_process;
295  isFromMCTProcessPrimary = false;
296  mf::LogDebug("PrimaryParticle") << "MCTruth primary process name contains \"primary\" "
297  << " but is not solely \"primary\" : " << process_name
298  << ".\nWill not store full set of trajectory points.";
299  }
300  else { // -- override it
301  process_name = "primary";
302  isFromMCTProcessPrimary = true;
303  mf::LogWarning("PrimaryParticle")
304  << "MCTruth primary process does not beging with string"
305  << " literal \"primary\" : " << process_name << "\nOVERRIDING it to \"primary\"";
306  }
307  // -- The process_name check is simply to allow an additional way to reduce memory usage,
308  // namely, by creating MCTruth input particles with multiple process labels (e.g. "primary"
309  // and "primaryBackground") that can be used to veto storage of trajectory points.
310 
311  // primary particles should have parentID = 0, even if there
312  // are multiple MCTruths for this event
313  parentID = 0;
314  } // end else no primary particle information
315  } // Is there a G4PrimaryParticle?
316  // If this is not a primary particle...
317  else {
318  // check if this particle was made in an undesirable process. For example:
319  // if one is not interested in EM shower particles, don't put it in the particle
320  // list as one wouldn't care about secondaries, tertiaries, etc. For these showers
321  // figure out what process is making this track - skip it if it is
322  // one of pair production, compton scattering, photoelectric effect
323  // bremstrahlung, annihilation, or ionization
324  process_name = track->GetCreatorProcess()->GetProcessName();
325  if (!fKeepEMShowerDaughters) {
326  bool notstore = false;
327  for (auto const& p : fNotStoredPhysics) {
328  if (process_name.find(p) != std::string::npos) {
329  notstore = true;
330  mf::LogDebug("NotStoredPhysics") << "Found process : " << process_name;
331 
332  int old = 0;
333  auto search = fNotStoredCounterUMap.find(p);
334  if (search != fNotStoredCounterUMap.end()) { old = search->second; }
335  fNotStoredCounterUMap.insert_or_assign(p, (old + 1));
336  break;
337  }
338  }
339 
340  if (notstore) {
341 
342  // figure out the ultimate parentage of this particle
343  // first add this track id and its parent to the fParentIDMap
344  fParentIDMap[trackID] = parentID;
345  fCurrentTrackID = this->GetParentage(trackID);
346  // check that fCurrentTrackID is in the particle list - it is possible
347  // that this particle's parent is a particle that did not get tracked.
348  // An example is a parent that was made due to muMinusCaptureAtRest
349  // and the daughter was made by the phot process. The parent likely
350  // isn't saved in the particle list because it is below the energy cut
351  // which will put a bogus track id value into the sim::IDE object for
352  // the sim::SimChannel if we don't check it.
353  if (!fParticleList.KnownParticle(fCurrentTrackID)) fCurrentTrackID = sim::NoParticleId;
354  // clear current particle as we are not stepping this particle and
355  // adding trajectory points to it
356  fdroppedTracksSet.insert(trackID);
357  fCurrentParticle.clear();
358  return;
359  } // end if process matches an undesired process
360  } // end if keeping EM shower daughters
361 
362  // Check the energy of the particle. If it falls below the energy
363  // cut, don't add it to our list.
364  G4double energy = track->GetKineticEnergy();
365  if (energy < fenergyCut) {
366  fdroppedTracksSet.insert(trackID);
367  fCurrentParticle.clear();
368  // do add the particle to the parent id map though
369  // and set the current track id to be it's ultimate parent
370 
371  fCurrentTrackID = this->GetParentage(trackID);
372 
373  return;
374  }
375 
376  // check to see if the parent particle has been stored in the particle navigator
377  // if not, then see if it is possible to walk up the fParentIDMap to find the
378  // ultimate parent of this particle. Use that ID as the parent ID for this
379  // particle
380  if (!fParticleList.KnownParticle(parentID)) {
381  // do add the particle to the parent id map
382  // just in case it makes a daughter that we have to track as well
383  fParentIDMap[trackID] = parentID;
384  int pid = this->GetParentage(parentID);
385 
386  // if we still can't find the parent in the particle navigator,
387  // we have to give up
388  if (!fParticleList.KnownParticle(pid)) {
389  MF_LOG_WARNING("ParticleListActionService")
390  << "can't find parent id: " << parentID << " in the particle list, or fParentIDMap."
391  << " Make " << parentID << " the mother ID for"
392  << " track ID " << fCurrentTrackID << " in the hope that it will aid debugging.";
393  }
394  else
395  parentID = pid;
396  }
397 
398  // Once the parentID is secured, inherit the MCTruth Index
399  // which should have been set already
400  if (auto it = fMCTIndexMap.find(parentID); it != cend(fMCTIndexMap)) {
401  primarymctIndex = it->second;
402  }
403  else {
404  throw art::Exception(art::errors::LogicError)
405  << "Could not locate MCT Index for parent trackID of " << parentID;
406  }
407 
408  // Inherit whether the parent is from a primary with MCTruth process_name == "primary"
409  isFromMCTProcessPrimary = fMCTPrimProcessKeepMap[parentID];
410  } // end if not a primary particle
411 
412  // This is probably the PDG mass, but just in case:
413  double mass = dynamicParticle->GetMass() / CLHEP::GeV;
414 
415  // Create the sim::Particle object.
416  fCurrentParticle.clear();
417  fCurrentParticle.particle =
418  new simb::MCParticle{trackID, pdgCode, process_name, parentID, mass};
419  fCurrentParticle.truthIndex = primaryIndex;
420 
421  fMCTIndexMap[trackID] = primarymctIndex;
422 
423  fMCTPrimProcessKeepMap[trackID] = isFromMCTProcessPrimary;
424 
425  // -- determine whether full set of trajectorie points should be stored or only the start and end points
426  fCurrentParticle.keepFullTrajectory =
427  (!fstoreTrajectories) ?
428  false : /*don't want trajectory points at all, bail*/
429  (!(fMCTIndexToGeneratorMap[primarymctIndex].second)) ?
430  false : /*particle is not from a storable generator*/
431  (!fkeepOnlyPrimaryFullTraj) ?
432  true : /*want all primaries tracked for a storable generator*/
433  (isFromMCTProcessPrimary) ?
434  true : /*only descendants from primaries with MCTruth process == "primary"*/
435  false; /*not from MCTruth process "primary"*/
436 
437  // Polarization.
438  const G4ThreeVector& polarization = track->GetPolarization();
439  fCurrentParticle.particle->SetPolarization(
440  TVector3{polarization.x(), polarization.y(), polarization.z()});
441 
442  // Save the particle in the ParticleList.
443  fParticleList.Add(fCurrentParticle.particle);
444  //hjw std::cout <<" pre Tracking fParticleList.size: " << fParticleList.size()<<std::endl;
445  }

448  void
449  ParticleListActionService::postUserTrackingAction(const G4Track* aTrack)
450  {
451  if (!fCurrentParticle.hasParticle()) return;
452 
453  if (aTrack) {
454  fCurrentParticle.particle->SetWeight(aTrack->GetWeight());
455 
456  // Get the post-step information from the G4Step.
457  const G4StepPoint* postStepPoint = aTrack->GetStep()->GetPostStepPoint();
458  if (!postStepPoint->GetProcessDefinedStep()) {
459  // Now we get to do some awkward cleanup because the
460  // fparticleList was augmented during the
461  // preUserTrackingAction. We cannot call 'Archive' because
462  // that only sets the mapped type of the entry to
463  // nullptr...which is really bad whenever we iterate through
464  // entries in the endOfEventAction and dereference the mapped
465  // type. We have to entirely erase the entry.
466  auto key_to_erase = fParticleList.key(fCurrentParticle.particle);
467  fParticleList.erase(key_to_erase);
468  // after the particle is archived, it is deleted
469  fCurrentParticle.clear();
470  return;
471  }
472 
473  G4String process = postStepPoint->GetProcessDefinedStep()->GetProcessName();
474  fCurrentParticle.particle->SetEndProcess(process);
475 
476  // -- D.R. Store the final point only for particles that have not had intermediate trajectory
477  // points saved. This avoids double counting the final trajectory point for particles from
478  // generators with storable trajectory points.
479 
480  if (!fCurrentParticle.keepFullTrajectory) {
481  const G4ThreeVector position = postStepPoint->GetPosition();
482  G4double time = postStepPoint->GetGlobalTime();
483 
484  // Remember that LArSoft uses cm, ns, GeV.
485  TLorentzVector fourPos(position.x() / CLHEP::cm,
486  position.y() / CLHEP::cm,
487  position.z() / CLHEP::cm,
488  time / CLHEP::ns);
489 
490  const G4ThreeVector momentum = postStepPoint->GetMomentum();
491  const G4double energy = postStepPoint->GetTotalEnergy();
492  TLorentzVector fourMom(momentum.x() / CLHEP::GeV,
493  momentum.y() / CLHEP::GeV,
494  momentum.z() / CLHEP::GeV,
495  energy / CLHEP::GeV);
496 
497  // Add another point in the trajectory.
498  AddPointToCurrentParticle(fourPos, fourMom, std::string(process));
499  }
500  // -- particle has a full trajectory, apply SparsifyTrajectory method if enabled
501  else if (fSparsifyTrajectories) {
502  fCurrentParticle.particle->SparsifyTrajectory(fSparsifyMargin, fKeepSecondToLast);
503  }
504  }
505 
506  // store truth record pointer, only if it is available
507  if (fCurrentParticle.isPrimary()) {
508  fPrimaryTruthMap[fCurrentParticle.particle->TrackId()] = fCurrentParticle.truthInfoIndex();
509  }
510 
511  return;
512  }

515  // With every step, add to the particle's trajectory.
516  void
517  ParticleListActionService::userSteppingAction(const G4Step* step)
518  {
519  // N.B. G4 guarantees that following are non-null:
520  // - step
521  // - step->GetPostStepPoint()
522  if (!fCurrentParticle.hasParticle() || !step->GetPostStepPoint()->GetProcessDefinedStep()) {
523  return;
524  }
525  // Temporary fix for problem where DeltaTime on the first step
526  // of optical photon propagation is calculated incorrectly. -wforeman
527  double const globalTime = step->GetTrack()->GetGlobalTime();
528  double const velocity_G4 = step->GetTrack()->GetVelocity();
529  double const velocity_step = step->GetStepLength() / step->GetDeltaTime();
530  if ((step->GetTrack()->GetDefinition()->GetPDGEncoding() == 0) &&
531  fabs(velocity_G4 - velocity_step) > 0.0001) {
532  // Subtract the faulty step time from the global time,
533  // and add the correct step time based on G4 velocity.
534  step->GetPostStepPoint()->SetGlobalTime(globalTime - step->GetDeltaTime() +
535  step->GetStepLength() / velocity_G4);
536  }
537 
538  // For the most part, we just want to add the post-step
539  // information to the particle's trajectory. There's one
540  // exception: In PreTrackingAction, the correct time information
541  // is not available. So add the correct vertex information here.
542 
543  if (fCurrentParticle.particle->NumberTrajectoryPoints() == 0) {
544 
545  // Get the pre/along-step information from the G4Step.
546  const G4StepPoint* preStepPoint = step->GetPreStepPoint();
547 
548  const G4ThreeVector position = preStepPoint->GetPosition();
549  G4double time = preStepPoint->GetGlobalTime();
550 
551  // Remember that LArSoft uses cm, ns, GeV.
552  TLorentzVector fourPos(position.x() / CLHEP::cm,
553  position.y() / CLHEP::cm,
554  position.z() / CLHEP::cm,

615  operator()(sim::ParticleList::value_type& particleListEntry) const
616  {
617  // We're looking at this Particle in the list.
618  int particleID = particleListEntry.first;
619

621  // we ask the particle list since the particle itself might have been lost
622  // ("archived"), but the particle list still holds the information we need
623  int parentID = particleList->GetMotherOf(particleID);
624 
625  // If the parentID <= 0, this is a primary particle.
626  if (parentID <= 0) return;
627 
628  // If we get here, this particle is somebody's daughter. Add
629  // it to the list of daughter particles for that parent.
630 
631  // Get the parent particle from the list.
632  sim::ParticleList::iterator parentEntry = particleList->find(parentID);
633 
634  if (parentEntry == particleList->end()) {
635  // We have an "orphan": a particle whose parent isn't
636  // recorded in the particle list. This is not signficant;
637  // it's possible for a particle not to be saved in the list
638  // because it failed an energy cut, but for it to have a
639  // daughter that passed the cut (e.g., a nuclear decay).
640  return;
641  }
642  if (!parentEntry->second) return; // particle archived, nothing to update
643 
644  // Add the current particle to the daughter list of the parent.
645  simb::MCParticle* parent = parentEntry->second;
646  //std::cout<<"AddDaughter: "<< particleID <<std::endl;
647  parent->AddDaughter(particleID);
648  }
