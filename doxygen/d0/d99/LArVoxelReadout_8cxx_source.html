207  // Called for each step.
208  G4bool
209  LArVoxelReadout::ProcessHits(G4Step* step, G4TouchableHistory* pHistory)
210  {
211  // All work done for the "parallel world" "box of voxels" in
212  // LArVoxelReadoutGeometry makes this a fairly simple routine.
213  // First, the usual check for non-zero energy:
214 
215  // Only process the hit if the step is inside the active volume and
216  // it has deposited energy. The hit being inside the active volume
217  // is virtually sure to happen because the LArVoxelReadoutGeometry
218  // that this class makes use of only has voxels for inside the TPC.
219 
220  // The step can be no bigger than the size of the voxel,
221  // because of the geometry set up in LArVoxelGeometry and the
222  // transportation set up in PhysicsList. Find the mid-point
223  // of the step.
224 
225  if (step->GetTotalEnergyDeposit() > 0) {
226 
227  // Make sure we have the IonizationAndScintillation singleton
228  // reset to this step
229  larg4::IonizationAndScintillation::Instance()->Reset(step);
230  fNSteps++;
231  if (!fDontDriftThem) {
232 
233  G4ThreeVector midPoint =
234  0.5 * (step->GetPreStepPoint()->GetPosition() + step->GetPostStepPoint()->GetPosition());
235  double g4time = step->GetPreStepPoint()->GetGlobalTime();
236 
237  // Find the Geant4 track ID for the particle responsible for depositing the
238  // energy. if we are only storing primary EM shower particles, and this energy
239  // is from a secondary etc EM shower particle, the ID returned is the primary
240  const int trackID = ParticleListAction::GetCurrentTrackID();
241 
242  // Find out which TPC we are in.
243  // If this readout object covers just one, we already know it.
244  // Otherwise, we have to ask Geant where we are.
245  unsigned short int cryostat = 0, tpc = 0;
246  if (bSingleTPC) {
247  cryostat = fCstat;
248  tpc = fTPC;
249  }
250  else {
251  // detect the TPC we are in
252  const G4VTouchable* pTouchable = step->GetPreStepPoint()->GetTouchable();
253  if (!pTouchable) {
254  throw cet::exception("LArG4") << "Untouchable step in LArVoxelReadout::ProcessHits()";
255  }
256 
257  // one of the ancestors of the touched volume is supposed to be
258  // actually a G4PVPlacementInTPC that knows which TPC it covers;
259  // currently, it's the 4th in the ladder:
260  // [0] voxel [1] voxel tower [2] voxel plane [3] the full box;
261  G4int depth = 0;
262  while (depth < pTouchable->GetHistoryDepth()) {
263  const G4PVPlacementInTPC* pPVinTPC =
264  dynamic_cast<const G4PVPlacementInTPC*>(pTouchable->GetVolume(depth++));
265  if (!pPVinTPC) continue;
266  cryostat = pPVinTPC->ID.Cryostat;
267  tpc = pPVinTPC->ID.TPC;
268  if (Has(fSkipWireSignalInTPCs, tpc)) { return true; }
269  break;
270  } // while
271  if (depth < pTouchable->GetHistoryDepth()) {
272  // this is a fundamental error where the step does not happen in
273  // any TPC; this should not happen in the readout geometry!
274  throw cet::exception("LArG4") << "No TPC ID found in LArVoxelReadout::ProcessHits()";
275  } // if
276  MF_LOG_DEBUG("LArVoxelReadoutHit") << " hit in C=" << cryostat << " T=" << tpc;
277  } // if more than one TPC
278 
279  // Note that if there is no particle ID for this energy deposit, the
280  // trackID will be sim::NoParticleId.
281 
282  DriftIonizationElectrons(*fClockData, midPoint, g4time, trackID, cryostat, tpc);
283  } // end we are drifting
284  } // end there is non-zero energy deposition
285 
286  return true;
287  }

298  geo::Point_t
299  LArVoxelReadout::RecoverOffPlaneDeposit(geo::Point_t const& pos, geo::PlaneGeo const& plane) const
300  {
301  //
302  // translate the landing position on the two frame coordinates
303  // ("width" and "depth")
304  //
305  auto const landingPos = plane.PointWidthDepthProjection(pos);
306 
307  //
308  // compute the distance of the landing position on the two frame
309  // coordinates ("width" and "depth");
310  // keep the point within 10 micrometers (0.001 cm) from the border
311  //
312  auto const offPlane = plane.DeltaFromActivePlane(landingPos, 0.001);
313 
314  //
315  // if both the distances are below the margin, move the point to
316  // the border
317  //
318 
319  // nothing to recover: landing is inside
320  if ((offPlane.X() == 0.0) && (offPlane.Y() == 0.0)) return pos;
321 
322  // won't recover: too far
323  if ((std::abs(offPlane.X()) > fOffPlaneMargin) || (std::abs(offPlane.Y()) > fOffPlaneMargin))
324  return pos;
325 
326  // we didn't fully decompose because it might be unnecessary;
327  // now we need the full thing
328  auto const distance = plane.DistanceFromPlane(pos);
329 
330  return plane.ComposePoint<geo::Point_t>(distance, landingPos + offPlane);
331  } // LArVoxelReadout::RecoverOffPlaneDeposit()

333  //----------------------------------------------------------------------------
334  // energy is passed in with units of MeV, dx has units of cm
335  void
336  LArVoxelReadout::DriftIonizationElectrons(detinfo::DetectorClocksData const& clockData,
337  G4ThreeVector stepMidPoint,
338  const double simTime,
339  int trackID,
340  unsigned short int cryostat,
341  unsigned short int tpc)
342  {
343  auto const tpcClock = clockData.TPCClock();
344 
345  // this must be always true, unless caller has been sloppy
346  assert(fPropGen); // No propagation random generator provided?!
347 
348  CLHEP::RandGauss PropRand(*fPropGen);
349 
350  // This routine gets called frequently, once per every particle
351  // traveling through every voxel. Use whatever tricks we can to
352  // increase its execution speed.
353 
354  static double LifetimeCorr_const = -1000. * fElectronLifetime;
355  static double LDiff_const = std::sqrt(2. * fLongitudinalDiffusion);
356  static double TDiff_const = std::sqrt(2. * fTransverseDiffusion);
357  static double RecipDriftVel[3] = {
358  1. / fDriftVelocity[0], 1. / fDriftVelocity[1], 1. / fDriftVelocity[2]};
359 
360  struct Deposit_t {
361  double energy = 0.;
362  double electrons = 0.;
363 
364  void
365  add(double more_energy, double more_electrons)
366  {
367  energy += more_energy;
368  electrons += more_electrons;
369  }
370  }; // Deposit_t
371 
372  // Map of electrons to store - catalogued by map[channel][tdc]
373  std::map<raw::ChannelID_t, std::map<unsigned int, Deposit_t>> DepositsToStore;
374 
375  double xyz1[3] = {0.};
376 
377  double const xyz[3] = {
378  stepMidPoint.x() / CLHEP::cm, stepMidPoint.y() / CLHEP::cm, stepMidPoint.z() / CLHEP::cm};
379 
380  // Already know which TPC we're in because we have been told
381 
382  try {
383  const geo::TPCGeo& tpcg = fGeoHandle->TPC(tpc, cryostat);
384 
385  // X drift distance - the drift direction can be either in
386  // the positive or negative direction, so use std::abs
387 
388  /// \todo think about effects of drift between planes
389  double XDrift = std::abs(stepMidPoint.x() / CLHEP::cm - tpcg.PlaneLocation(0)[0]);
390  //std::cout<<tpcg.DriftDirection()<<std::endl;
391  if (tpcg.DriftDirection() == geo::kNegX)
392  XDrift = stepMidPoint.x() / CLHEP::cm - tpcg.PlaneLocation(0)[0];
393  else if (tpcg.DriftDirection() == geo::kPosX)
394  XDrift = tpcg.PlaneLocation(0)[0] - stepMidPoint.x() / CLHEP::cm;
395 
396  if (XDrift < 0.) return;
397 
398  // Get SCE {x,y,z} offsets for particular location in TPC
399  geo::Vector_t posOffsets;
400  auto const* SCE = lar::providerFrom<spacecharge::SpaceChargeService>();
401  if (SCE->EnableSimSpatialSCE() == true) {
402  posOffsets = SCE->GetPosOffsets({xyz[0], xyz[1], xyz[2]});
403  if (larsim::Utils::SCE::out_of_bounds(posOffsets)) {
404  return;
405  }
406  }
407  posOffsets.SetX(-posOffsets.X());
408 
409  // Drift time (nano-sec)
410  double TDrift;
411  XDrift += posOffsets.X();
412 
413  // Space charge distortion could push the energy deposit beyond the wire
414  // plane (see issue #15131). Given that we don't have any subtlety in the
415  // simulation of this region, bringing the deposit exactly on the plane
416  // should be enough for the time being.
417  if (XDrift < 0.) XDrift = 0.;
418 
419  TDrift = XDrift * RecipDriftVel[0];
420  if (tpcg.Nplanes() == 2) { // special case for ArgoNeuT (plane 0 is the second wire plane)
421  TDrift = ((XDrift - tpcg.PlanePitch(0, 1)) * RecipDriftVel[0] +
422  tpcg.PlanePitch(0, 1) * RecipDriftVel[1]);
423  }
424 
425  const double lifetimecorrection = TMath::Exp(TDrift / LifetimeCorr_const);
426  const int nIonizedElectrons =
427  larg4::IonizationAndScintillation::Instance()->NumberIonizationElectrons();
428  const double energy = larg4::IonizationAndScintillation::Instance()->EnergyDeposit();
429 
430  // if we have no electrons (too small energy or too large recombination)
431  // we are done already here
432  if (nIonizedElectrons <= 0) {
433  MF_LOG_DEBUG("LArVoxelReadout")
434  << "No electrons drifted to readout, " << energy << " MeV lost.";
435  return;
436  }
437  // includes the effect of lifetime
438  const double nElectrons = nIonizedElectrons * lifetimecorrection;
439 
440  // Longitudinal & transverse diffusion sigma (cm)
441  double SqrtT = std::sqrt(TDrift);
442  double LDiffSig = SqrtT * LDiff_const;
443  double TDiffSig = SqrtT * TDiff_const;
444  double electronclsize = fElectronClusterSize;
445 
446  int nClus = (int)std::ceil(nElectrons / electronclsize);
447  if (nClus < fMinNumberOfElCluster) {
448  electronclsize = nElectrons / fMinNumberOfElCluster;
449  if (electronclsize < 1.0) { electronclsize = 1.0; }
450  nClus = (int)std::ceil(nElectrons / electronclsize);
451  }
452 
453  // Compute arrays of values as quickly as possible.
454  std::vector<double> XDiff(nClus);
455  std::vector<double> YDiff(nClus);
456  std::vector<double> ZDiff(nClus);
457  std::vector<double> nElDiff(nClus, electronclsize);
458  std::vector<double> nEnDiff(nClus);
459 
460  // fix the number of electrons in the last cluster, that has smaller size
461  nElDiff.back() = nElectrons - (nClus - 1) * electronclsize;
462 
463  for (size_t xx = 0; xx < nElDiff.size(); ++xx) {
464  if (nElectrons > 0)
465  nEnDiff[xx] = energy / nElectrons * nElDiff[xx];
466  else
467  nEnDiff[xx] = 0.;
468  }
469 
470  double const avegageYtransversePos = (stepMidPoint.y() / CLHEP::cm) + posOffsets.Y();
471  double const avegageZtransversePos = (stepMidPoint.z() / CLHEP::cm) + posOffsets.Z();
472 
473  // Smear drift times by x position and drift time
474  if (LDiffSig > 0.0)
475  PropRand.fireArray(nClus, &XDiff[0], 0., LDiffSig);
476  else
477  XDiff.assign(nClus, 0.0);
478 
479  if (TDiffSig > 0.0) {
480  // Smear the Y,Z position by the transverse diffusion
481  PropRand.fireArray(nClus, &YDiff[0], avegageYtransversePos, TDiffSig);
482  PropRand.fireArray(nClus, &ZDiff[0], avegageZtransversePos, TDiffSig);
483  }
484  else {
485  YDiff.assign(nClus, avegageYtransversePos);
486  ZDiff.assign(nClus, avegageZtransversePos);
487  }
488 
489  // make a collection of electrons for each plane
490  for (size_t p = 0; p < tpcg.Nplanes(); ++p) {
491 
492  geo::PlaneGeo const& plane = tpcg.Plane(p);
493 
494  double Plane0Pitch = tpcg.Plane0Pitch(p);
495 
496  // "-" sign is because Plane0Pitch output is positive. Andrzej
497  xyz1[0] = tpcg.PlaneLocation(0)[0] - Plane0Pitch;
498 
499  // Drift nClus electron clusters to the induction plane
500  for (int k = 0; k < nClus; ++k) {
501  // Correct drift time for longitudinal diffusion and plane
502  double TDiff = TDrift + XDiff[k] * RecipDriftVel[0];
503  // Take into account different Efields between planes
504  // Also take into account special case for ArgoNeuT where Nplanes = 2.
505  for (size_t ip = 0; ip < p; ++ip) {
506  TDiff +=
507  tpcg.PlanePitch(ip, ip + 1) * RecipDriftVel[tpcg.Nplanes() == 3 ? ip + 1 : ip + 2];
508  }
509  xyz1[1] = YDiff[k];
510  xyz1[2] = ZDiff[k];
511 
512  /// \todo think about effects of drift between planes
513 
514  // grab the nearest channel to the xyz1 position
515  try {
516  if (fOffPlaneMargin != 0) {
517  // get the effective position where to consider the charge landed;
518  //
519  // Some optimisations are possible; in particular, this method
520  // could be extended to inform us if the point was too far.
521  // Currently, if that is the case the code will proceed, find the
522  // point is off plane, emit a warning and skip the deposition.
523  //
524  auto const landingPos = RecoverOffPlaneDeposit({xyz1[0], xyz1[1], xyz1[2]}, plane);
525  xyz1[0] = landingPos.X();
526  xyz1[1] = landingPos.Y();
527  xyz1[2] = landingPos.Z();
528 
529  } // if charge lands off plane
530  uint32_t channel = fGeoHandle->NearestChannel(xyz1, p, tpc, cryostat);
531 
532  /// \todo check on what happens if we allow the tdc value to be
533  /// \todo beyond the end of the expected number of ticks
534  // Add potential decay/capture/etc delay effect, simTime.
535  unsigned int tdc = tpcClock.Ticks(clockData.G4ToElecTime(TDiff + simTime));
536 
537  // Add electrons produced by each cluster to the map
538  DepositsToStore[channel][tdc].add(nEnDiff[k], nElDiff[k]);
539  }
540  catch (cet::exception& e) {
541  MF_LOG_DEBUG("LArVoxelReadout")
542  << "unable to drift electrons from point (" << xyz[0] << "," << xyz[1] << ","
543  << xyz[2] << ") with exception " << e;
544  }
545  } // end loop over clusters
546  } // end loop over planes
547 
548  // Now store them in SimChannels
549  ChannelMap_t& ChannelDataMap = fChannelMaps[cryostat][tpc];
550 
551  // browse deposited data on each channel: (channel; deposit data in time)
552  for (auto const& deposit_per_channel : DepositsToStore) {
553 
554  raw::ChannelID_t channel = deposit_per_channel.first;
555 
556  // find whether we already have this channel
557  auto iChannelData = ChannelDataMap.find(channel);
558 
559  // channelData is the SimChannel these deposits are going to be added to
560  // If there is such a channel already, use it (first beanch).
561  // If it's a new channel, the inner assignment creates a new SimChannel
562  // in the map, and we save its reference in channelData
563  sim::SimChannel& channelData = (iChannelData == ChannelDataMap.end()) ?
564  (ChannelDataMap[channel] = sim::SimChannel(channel)) :
565  iChannelData->second;
566 
567  // go through all deposits, one for each TDC: (TDC, deposit data)
