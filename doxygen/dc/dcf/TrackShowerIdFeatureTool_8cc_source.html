362 float ConeChargeFeatureTool::CalculateConicalness(const CaloHitList &caloHitList, const CartesianVector &pfoStart, const CartesianVector &pfoDir, const float pfoLength)
363 {
364  float totalChargeStart(0.f), totalChargeEnd(0.f);
365  float chargeConStart(0.f), chargeConEnd(0.f);
366  unsigned int nHitsConStart(0), nHitsConEnd(0);
367 
368  for (const CaloHit *const pCaloHit : caloHitList)
369  {
370  const float distFromTrackFit(((pCaloHit->GetPositionVector() - pfoStart).GetCrossProduct(pfoDir)).GetMagnitude());
371  const float hitLength(std::fabs((pCaloHit->GetPositionVector() - pfoStart).GetDotProduct(pfoDir)));
372 
373  if (hitLength / pfoLength < m_conFracRange)
374  {
375  chargeConStart += distFromTrackFit * distFromTrackFit * pCaloHit->GetInputEnergy();
376  ++nHitsConStart;
377  totalChargeStart += pCaloHit->GetInputEnergy();
378  }
379  else if (1.f - hitLength / pfoLength < m_conFracRange)
380  {
381  chargeConEnd += distFromTrackFit * distFromTrackFit * pCaloHit->GetInputEnergy();
382  ++nHitsConEnd;
383  totalChargeEnd += pCaloHit->GetInputEnergy();
384  }
385  }
386 
387  float conicalness(1.f);
388 
389  if (nHitsConStart >= m_conMinHits && nHitsConEnd >= m_conMinHits && totalChargeEnd > m_minCharge &&
390  std::sqrt(chargeConStart) > m_minCharge && totalChargeStart > m_minCharge)
391  conicalness = (std::sqrt(chargeConEnd / chargeConStart)) / (totalChargeEnd / totalChargeStart);
392 
393  return conicalness;
394 }

727 float ThreeDOpeningAngleFeatureTool::OpeningAngle(const CartesianVector &principal, const CartesianVector &secondary, const CartesianVector &eigenValues) const
728 {
729  const float principalMagnitude(principal.GetMagnitude());
730  const float secondaryMagnitude(secondary.GetMagnitude());
731 
732  if (std::fabs(principalMagnitude) < std::numeric_limits<float>::epsilon())
733  {
734  std::cout << "ThreeDOpeningAngleFeatureTool::OpeningAngle - The principal eigenvector is 0." << std::endl;
735  throw StatusCodeException(STATUS_CODE_INVALID_PARAMETER);
736  }
737  else if (std::fabs(secondaryMagnitude) < std::numeric_limits<float>::epsilon())
738  {
739  return 0.f;
740  }
741 
742  const float cosTheta(principal.GetDotProduct(secondary) / (principalMagnitude * secondaryMagnitude));
743 
744  if (cosTheta > 1.f)
745  {
746  std::cout << "PcaShowerParticleBuildingAlgorithm::OpeningAngle - cos(theta) reportedly greater than 1." << std::endl;
747  throw StatusCodeException(STATUS_CODE_INVALID_PARAMETER);
748  }
749 
750  const float sinTheta(std::sqrt(1.f - cosTheta * cosTheta));
751 
752  if (eigenValues.GetX() < std::numeric_limits<float>::epsilon())
753  {
754  std::cout << "PcaShowerParticleBuildingAlgorithm::OpeningAngle - principal eigenvalue less than or equal to 0." << std::endl;
755  throw StatusCodeException(STATUS_CODE_INVALID_PARAMETER);
756  }
757  else if (eigenValues.GetY() < std::numeric_limits<float>::epsilon())
758  {
759  return 0.f;
760  }
761 
762  return std::atan(std::sqrt(eigenValues.GetY()) * sinTheta / std::sqrt(eigenValues.GetX()));
763 }

872 void ThreeDChargeFeatureTool::CalculateChargeVariables(const Algorithm *const pAlgorithm, const pandora::Cluster *const pCluster,
873  float &totalCharge, float &chargeSigma, float &chargeMean, float &endCharge)
874 {

876  chargeSigma = 0.f;
877  chargeMean = 0.f;
878  endCharge = 0.f;
879 
880  CaloHitList orderedCaloHitList;
881  this->OrderCaloHitsByDistanceToVertex(pAlgorithm, pCluster, orderedCaloHitList);
882 
883  FloatVector chargeVector;
884  unsigned int hitCounter(0);
885  const unsigned int nTotalHits(orderedCaloHitList.size());
886 
887  for (const CaloHit *const pCaloHit : orderedCaloHitList)
888  {
889  ++hitCounter;
890  const float pCaloHitCharge(pCaloHit->GetInputEnergy());
891 
892  if (pCaloHitCharge >= 0.f)
893  {
894  totalCharge += pCaloHitCharge;
895  chargeVector.push_back(pCaloHitCharge);
896 
897  if (hitCounter >= std::floor(static_cast<float>(nTotalHits) * (1.f - m_endChargeFraction)))
898  endCharge += pCaloHitCharge;
899  }
900  }
901 
902  if (!chargeVector.empty())
903  {
904  chargeMean = totalCharge / static_cast<float>(chargeVector.size());
905 
906  for (const float charge : chargeVector)
907  chargeSigma += (charge - chargeMean) * (charge - chargeMean);
908 
909  chargeSigma = std::sqrt(chargeSigma / static_cast<float>(chargeVector.size()));
910  }
911 }
