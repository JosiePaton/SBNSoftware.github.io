419  }

427  }

429  ++nHitInPl2[hitInPl2];

432  // number of cluster hits

434  short imtru = -1;
435  for (unsigned int ipl = 0; ipl < nHitInPl2.size(); ++ipl) {
436  if (nHitInPl2[ipl] > nhit) {
437  nhit = nHitInPl2[ipl];
438  imtru = ipl;
439  }
440  } // ipl

460  }
