112 //------------------------------------------------------------------------------
113 auto icarus::trigger::PMTverticalSlicingAlg::clusterPMTby
114  (PMTlist_t const& PMTs, geo::Vector_t const& dir) -> std::vector<PMTlist_t>
115 {
116  // use the projection of the PMT center on the specified direction as
117  // clustering key; we need a reference point to project... we pick `origin()`.
118  auto const PMTcenterProjection = [dir](geo::OpDetGeo const* opDet)
119  { return (opDet->GetCenter() - geo::origin()).Dot(dir); };
120 
121  // PMTs with a coordinate within 5 cm (kind of) will be clustered together
122  constexpr double tol = 5.0; // cm
123  auto const closeEnough
124  = [](double a, double b){ return std::abs(a - b) < tol; };
125 
126  return util::clusterBy(PMTs, PMTcenterProjection, closeEnough, std::less<>());
127 
128 } // icarus::trigger::PMTverticalSlicingAlg::clusterPMTby()
