17 /**
18  * @file larreco/TrackFinder/TrackProducerFromTrackTrajectory_module.cc
19  * @class TrackProducerFromTrackTrajectory
20  *
21  * @brief Produce a reco::Track collection, as a result of the fit of an existing recob::TrackTrajectory collection.
22  *
23  * This producer takes an input an existing recob::TrackTrajectory collection (and the associated hits) and fits it.
24  * The mandatory output are: the resulting recob::Track collection, the associated hits, and the association
25  * between the input TrackTrajectory and the output Track.
26  * Optional outputs are recob::TrackFitHitInfo and recob::SpacePoint collections, plus the Assns of SpacePoints to Hits.
27  * An option is provided to create SpacePoints from the TrajectoryPoints in the Track.
28  * The fit is performed by an user-defined tool, which must inherit from larreco/TrackFinder/TrackMaker.
29  *
30  * Parameters: trackMaker (fhicl::ParameterSet for the trkmkr::TrackMaker tool used to do the fit), inputCollection (art::InputTag of the input recob::TrackTrajectory collection),
31  * doTrackFitHitInfo (bool to decide whether to produce recob::TrackFitHitInfo's), doSpacePoints (bool to decide whether to produce recob::SpacePoint's), and
32  * spacePointsFromTrajP (bool to decide whether the produced recob::SpacePoint's are taken from the recob::tracking::TrajectoryPoint_t's of the fitted recob::Track).
33  *
34  * @author G. Cerati (FNAL, MicroBooNE)
35  * @date 2017
36  * @version 1.0
37  */

149  const art::Ptr<recob::SpacePoint> apsp = spacePointPtrMaker(outputSpacePoints->size() - 1);

151  }

156  for (auto it = osp.begin(); it != osp.end(); ++it) {
157  outputSpacePoints->emplace_back(std::move(it->first));
158  const art::Ptr<recob::SpacePoint> apsp = spacePointPtrMaker(outputSpacePoints->size() - 1);
159  outputHitSpacePointAssn->addSingle(it->second, apsp);
160  }
161  }
162  if (doTrackFitHitInfo_) { outputHitInfo->emplace_back(optionals.trackFitHitInfos()); }
163  }
164 
165  // Put collections in the event
166  e.put(std::move(outputTracks));
167  e.put(std::move(outputHits));
168  e.put(std::move(outputTTjTAssn));
169  if (doTrackFitHitInfo_) { e.put(std::move(outputHitInfo)); }
170  if (doSpacePoints_) {
171  e.put(std::move(outputSpacePoints));
172  e.put(std::move(outputHitSpacePointAssn));
173  }
