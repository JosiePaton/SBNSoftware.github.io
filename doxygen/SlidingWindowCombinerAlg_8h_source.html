148 template <typename GateObject>
149 void icarus::trigger::SlidingWindowCombinerAlg::checkInput
150  (icarus::trigger::TriggerGateIndex<GateObject> const& gateIndex) const
151 {
152  /*
153  * Checks:
154  *
155  * * no gate is only partially contained in a window
156  * * no channel in the gates is skipped (optional)
157  *
158  */
159 
160  auto const isChannelInWindow
161  = [](raw::Channel_t channel, std::vector<raw::Channel_t> const& window)
162  { return std::binary_search(window.begin(), window.end(), channel); };
163 
164  std::set<GateObject const*> usedGates;
165  for (std::vector<raw::Channel_t> const& window: fWindowChannels) {
166 
167  for (raw::Channel_t const channel: window) {
168 
169  if (isMissingChannel(channel)) continue;
170 
171  //
172  // check that we have all the channels we need
173  //
174  GateObject const* gate = gateIndex.find(channel);
175  if (!gate) {
176  cet::exception e("SlidingWindowCombinerAlg");
177  e << "SlidingWindowCombinerAlg::checkInput(): No gate has channel "
178  << channel << ", required in the window including channels: ";
179  for (raw::Channel_t const channel: window) e << " " << channel;
180  e << ".\n";
181  throw e;
182  }
183 
184  //
185  // check that all channels of this gate are in this window
186  //
187  for (raw::Channel_t const gateChannel: gate->channels()) {
188  if (isChannelInWindow(gateChannel, window)) continue;
189 
190  cet::exception e("SlidingWindowCombinerAlg");
191  e << "SlidingWindowCombinerAlg::checkInput(): gate with channels {";
192  for (raw::Channel_t const channel: gate->channels())
193  e << " " << channel;
194  e << " } is not fully included in window including channels: ";
195  for (raw::Channel_t const channel: window) e << " " << channel;
196  e << " (" << gateChannel << " is missing).\n";
197  throw e;
198  } // for gate channels
199 
200  usedGates.insert(gate);
201  } // for channel in window
202 
203  } // for windows
204 
205  //
206  // check for channels that should be missing but are here
207  //
208  std::vector<raw::Channel_t> spuriousChannels;
209  for (raw::Channel_t const channel: fMissingChannels) {
210  if (gateIndex.find(channel)) spuriousChannels.push_back(channel);
211  } // for
212  if (!spuriousChannels.empty()) {
213  cet::exception e("SlidingWindowCombinerAlg");
214  e << "SlidingWindowCombinerAlg::checkInput(): "
215  << spuriousChannels.size() << " of the " << fMissingChannels.size()
216  << " supposedly missing channels are actually included:";
217  for (raw::Channel_t const channel: spuriousChannels) e << " " << channel;
218  e << "\n";
219  throw e;
220  }
221 
222  //
223  // check that all channels are here (except known missing ones)
224  //
225  if (fRequireFullCoverage) {
226  std::set<GateObject const*> allGates;
227  for (auto const channel: util::counter<raw::Channel_t >(gateIndex.nChannels())) {
228  if (isMissingChannel(channel)) continue;
229  allGates.insert(&(gateIndex[channel]));
230  }
231 
232  std::vector<GateObject const*> unusedGates;
233  std::set_difference(
234  allGates.cbegin(), allGates.cend(), usedGates.cbegin(), usedGates.cend(),
235  std::back_inserter(unusedGates)
236  );
237  if (!unusedGates.empty()) {
238  // we don't have much information to identify "which" gates...
239  cet::exception e("SlidingWindowCombinerAlg");
240  e << "SlidingWindowCombinerAlg::checkInput(): "
241  << "used only " << usedGates.size() << "/" << allGates.size()
242  << " input trigger gates, " << unusedGates.size() << " were not used:";
243  for (GateObject const* gate: unusedGates) {
244  e << "{";
245  for (raw::Channel_t const channel: gate->channels())
246  e << " ch:" << std::to_string(channel);
247  e << " }";
248  } // for
249  e << "\n";
250  throw e;
251  } // if unused
252  } // if full coverage
253 
254 } // icarus::trigger::SlidingWindowCombinerAlg::checkInput()

258 template <typename GateObject>
259 icarus::trigger::MultiChannelOpticalTriggerGate
260 icarus::trigger::SlidingWindowCombinerAlg::combineChannels(
261  icarus::trigger::TriggerGateIndex<GateObject> const& gates,
262  WindowChannels_t const& channels
263  ) const
264 {
265  using TriggerGate_t = GateObject;
266 
267  assert(!channels.empty());
268 
269  auto iChannel = firstChannelPresent(channels);
270  auto const cend = channels.end();
271  if (iChannel == cend) return {}; // empty gate, no channels inside
272 
273  TriggerGate_t const& firstGate = gates[*iChannel];
274 
275  // add the first gate
276  icarus::trigger::MultiChannelOpticalTriggerGate gate
277  { static_cast<icarus::trigger::OpticalTriggerGate const&>(firstGate) };
278 
279  mf::LogTrace(fLogCategory) << "Input: " << firstGate;
280  while (++iChannel != cend) {
281  if (isMissingChannel(*iChannel)) continue;
282 
283  TriggerGate_t const& inputGate = gates[*iChannel];
284 
285  mf::LogTrace(fLogCategory) << "Input: " << inputGate;
286  mergeGateInto(gate, inputGate);
287 
288  } // while
289  mf::LogTrace(fLogCategory) << "Output: " << gate;
290 
291  return gate;
292 
293 } // icarus::trigger::SlidingWindowCombinerAlg::combineChannel()

297 template <typename GateObject>
298 bool icarus::trigger::SlidingWindowCombinerAlg::mergeGateInto(
299  icarus::trigger::MultiChannelOpticalTriggerGate& dest,
300  GateObject const& input
301  )
302 {
303  // loose check: assumes channel sets of all possible inputs do not overlap
304  // and checks whether the destination has already any one of input channels
305  auto const& channels = dest.channels();
306  if (inList(channels, *(input.channels().begin()))) return false;
307 
308  dest.Sum(input);
309 
310  return true;
311 
312 } // icarus::trigger::SlidingWindowCombinerAlg::mergeGateInto()
