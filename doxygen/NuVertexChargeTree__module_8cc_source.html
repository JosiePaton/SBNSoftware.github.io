411 const simb::MCParticle *Genie2G4MCParticle(

414  const std::vector<art::Ptr<simb::MCParticle>> &g4_mcparticles,

416 
417  const simb::MCParticle *ret = nullptr;
418  for (int iparticle = 0; iparticle < (int)g4_mcparticles.size(); iparticle++) {
419  if (infos[iparticle]->hasGeneratedParticleIndex() &&
420  (int)infos[iparticle]->generatedParticleIndex() < mctruth.NParticles() && // TODO: why is this number sometimes bigger than the number of particles?
421  mctruth.GetParticle(infos[iparticle]->generatedParticleIndex()).TrackId() == genie_part.TrackId() &&
422  g4_mcparticles[iparticle]->Process() == "primary" /* TODO: will have to remove this restriction to include secondary particles*/) {
423 
424  // if a genie particle re-scatters in g4 and makes more particles, then multiple g4 particles can match to a
425  // genie particle. Thus, we also check that the start location of the associated genie particle matches the g4
426  // and that the pdgid matches (to be on the safe side)
427  //
428  // Note that this should be accounted for by requiring the process to be primary. This is a bit of redundancy.
429  const simb::MCParticle& matched_genie_particle = mctruth.GetParticle(infos[iparticle]->generatedParticleIndex());
430  if ((matched_genie_particle.Position().Vect() - g4_mcparticles[iparticle]->Position().Vect()).Mag() < 1e-4 &&
431  matched_genie_particle.PdgCode() == g4_mcparticles[iparticle]->PdgCode()) {
432 
433  // this should only be true for one particle
434  assert(ret == nullptr);
435  ret = g4_mcparticles[iparticle].get();
436  }
437  }
438  }
