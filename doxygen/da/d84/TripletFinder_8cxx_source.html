179

181  while (xvit_begin != xvs.end() && LessThanXHit(*xvit_begin, xu))
182  ++xvit_begin;
183

186  const HitOrChan& v = xvit->b;

210  ++nx;
211  }
212  if (v.hit) {

214  ++nx;
