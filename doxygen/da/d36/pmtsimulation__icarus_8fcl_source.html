76 #
77 # Single photoelectron response with exponential rise and a fall shapes.
78 #
79 # Parameters are reworked from an ICARUS trigger note (not yet released as
80 # of April 2020), which come from data analysis from the BNL group.
81 #
82 # In particular, raise and fall times are reported to be 1.4 and 3.2 ns for
83 # the PMT, and 1.8 and 4.2 ns including the cable distortion at CERN (10 m),
84 # which is not the setup of the actual ICARUS detector (37 m).
85 # The average gain is considered to be 7 x 10^6 (somehow lower than the nominal
86 # 10^7 which we had targetted).
87 # Attenuation from the cable is considered negligible.
88 #
89 # The peak charge is computed starting from the charge of a photoelectron
90 # (-1.6 x 10^-7 pC), multiplied by the gain (7 x 10^6). At signal peak time,
91 # the current from that charge is -0.14 mA, i.e. a voltage drop on 50 ohm
92 # of -7.07 mV. In this parametrisation, we inject the gain and let the code
93 # compute the amplitude.
94 # The digitizer converts a range up to 2 volt with 14 bit, which translates into
95 # about 8.192 ADC/mV and 409.6 ADC/mA.
96 # In short, the peak signal is expected to be -7.07 mV, -0.14 mA, -57.93 ADC.
97 #
98 # The integral of the shape is 191.407 ADC (time 0 to infinity, 2 ns sampling).
99 #
100 icarus_photoelectronresponse_exponentials: {
101 
102  # run `lar --print-description AsymGaussPulseFunctionTool` for configuration
103  tool_type: AsymExpPulseFunctionTool
104 
105  TransitTime: "55.1 ns"
106  RaiseTimeConstant: "1.8 ns"
107  FallTimeConstant: "4.2 ns"
108  Gain: @local::icarus_settings_opdet.nominalPMTgain # Gain = 7.0e6 => PeakCurrent: -0.14144 mA
109  ADC: 409.6 # ADC / mA
110 
111 } # icarus_photoelectronresponse_exponentials
