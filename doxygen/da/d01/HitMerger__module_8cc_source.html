129 void HitMerger::produce(art::Event & evt)
130 {
131  // container for our new hit collection
132  std::unique_ptr<std::vector<recob::Hit>> outputHitPtrVec(new std::vector<recob::Hit>);
133 
134  /// Associations with wires.
135  std::unique_ptr<art::Assns<recob::Wire, recob::Hit>> wireAssns(new art::Assns<recob::Wire, recob::Hit>);
136 
137  /// Associations with raw digits.
138  std::unique_ptr<art::Assns<raw::RawDigit, recob::Hit>> rawDigitAssns(new art::Assns<raw::RawDigit, recob::Hit>);
139 
140  RecobHitToPtrMap recobHitToPtrMap;
141 
142  // Use this handy art utility to make art::Ptr objects to the new recob::Hits for use in the output phase
143  art::PtrMaker<recob::Hit> ptrMaker(evt);
144 
145  // Outside loop over the input hit producers
146  for(const auto& inputTag : HitMergerfHitProducerLabelVec)
147  {
148  // Start by looking up the original hits
149  art::Handle< std::vector<recob::Hit> > hitHandle;
150  evt.getByLabel(inputTag, hitHandle);
151 
152  for(size_t hitIdx = 0; hitIdx < hitHandle->size(); hitIdx++)
153  {
154  art::Ptr<recob::Hit> hitPtr(hitHandle,hitIdx);
155 
156  const recob::Hit* hit2D = hitPtr.get();
157 
158  // Create and save the new recob::Hit with the correct WireID
159  outputHitPtrVec->emplace_back(recob::HitCreator(*hit2D, hit2D->WireID()).copy());
160 
161  // Recover a pointer to it...
162  recob::Hit* newHit = &outputHitPtrVec->back();
163 
164  // Create a mapping from this hit to an art Ptr representing it
165  recobHitToPtrMap[newHit] = ptrMaker(outputHitPtrVec->size()-1);
166  }
167  }
168 
169  // Set up to make the associations (if desired)
170  makeWireAssns(evt, *wireAssns, recobHitToPtrMap);
171 
172  makeRawDigitAssns(evt, *rawDigitAssns, recobHitToPtrMap);
173 
174  // Move everything into the event
175  evt.put(std::move(outputHitPtrVec));
176  evt.put(std::move(wireAssns));
177  evt.put(std::move(rawDigitAssns));
178 
179  return;
180 }

182 void HitMerger::makeWireAssns(const art::Event& evt, art::Assns<recob::Wire, recob::Hit>& wireAssns, RecobHitToPtrMap& recobHitPtrMap) const
183 {
184  // Let's make sure the input associations container is empty
185  wireAssns = art::Assns<recob::Wire, recob::Hit>();
186 
187  // First task is to recover all of the previous wire <--> hit associations and map them by channel number
188  // Create the temporary container
189  std::unordered_map<raw::ChannelID_t, art::Ptr<recob::Wire>> channelToWireMap;
190 
191  // Go through the list of input sources and fill out the map
192  for(const auto& inputTag : HitMergerfHitProducerLabelVec)
193  {
194  art::ValidHandle<std::vector<recob::Hit>> hitHandle = evt.getValidHandle<std::vector<recob::Hit>>(inputTag);
195 
196  art::FindOneP<recob::Wire> hitToWireAssns(hitHandle, evt, inputTag);
197 
198  if (hitToWireAssns.isValid())
199  {
200  for(size_t wireIdx = 0; wireIdx < hitToWireAssns.size(); wireIdx++)
201  {
202  art::Ptr<recob::Wire> wire = hitToWireAssns.at(wireIdx);
203 
204  channelToWireMap[wire->Channel()] = wire;
205  }
206  }
207  }
208 
209  // Now fill the container
210  for(const auto& hitPtrPair : recobHitPtrMap)
211  {
212  raw::ChannelID_t channel = hitPtrPair.first->Channel();
213 
214  std::unordered_map<raw::ChannelID_t, art::Ptr<recob::Wire>>::iterator chanWireItr = channelToWireMap.find(channel);
215 
216  if (!(chanWireItr != channelToWireMap.end())) continue;
217 
218  wireAssns.addSingle(chanWireItr->second, hitPtrPair.second);
219  }
220 
221  return;
222 }
