515 
516  const int nbinsx = fBins[0].NBins();
517  const int nbinsy = fBins[1].NBins();
518 
519  // Normalize 2D histogram to X-axis spectrum

521  double norm = 0.0;

709  if(fPOT == 0 && fLivetime == 0){
710  if(rhs.fHist) fHist->Add(rhs.fHist, sign);
711  if(rhs.fHistSparse) fHistSparse->Add(rhs.fHistSparse, sign);
712  fPOT = rhs.fPOT;
713  fLivetime = rhs.fLivetime;
714  return *this;
715  }
716 
717  if(!fLivetime && !rhs.fPOT){

719  << fPOT << ") but no livetime and Spectrum with livetime ("
720  << rhs.fLivetime << " sec) but no POT." << std::endl;
721  abort();
722  }
723 
724  if(!fPOT && !rhs.fLivetime){
725  std::cout << "Error: can't sum Spectrum with livetime ("
726  << fLivetime << " sec) but no POT and Spectrum with POT ("
727  << rhs.fPOT << ") but no livetime." << std::endl;
728  abort();
729  }
730 
731  // And now there are still a bunch of good cases to consider
732 
733  if(fPOT && rhs.fPOT){
734  // Scale by POT when possible
735  if(rhs.fHist) fHist->Add(rhs.fHist, sign*fPOT/rhs.fPOT);
736  if(rhs.fHistSparse) fHistSparse->Add(rhs.fHistSparse, sign*fPOT/rhs.fPOT);
737 
738  if(fLivetime && rhs.fLivetime){
739  // If POT/livetime ratios match, keep regular lifetime, otherwise zero
740  // it out.
741  if(!AlmostEqual(fLivetime*rhs.fPOT, rhs.fLivetime*fPOT)){
742  std::cout << "Attempting to combine two spectra with incompatible POT / livetime ratio " << fPOT << " / " << fLivetime << " vs " << rhs.fPOT << " / " << rhs.fLivetime << std::endl;
743  abort();
744  }
745  }
746  if(!fLivetime && rhs.fLivetime){
747  // If the RHS has a livetime and we don't, copy it in (suitably scaled)
748  fLivetime = rhs.fLivetime * fPOT/rhs.fPOT;
749  }
750  // Otherwise, keep our own livetime (if any)
751 
752  return *this;
753  }
754 
755  if(fLivetime && rhs.fLivetime){
756  // Scale by livetime, the only thing in common
757  if(rhs.fHist) fHist->Add(rhs.fHist, sign*fLivetime/rhs.fLivetime);
758  if(rhs.fHistSparse) fHistSparse->Add(rhs.fHistSparse, sign*fLivetime/rhs.fLivetime);
759 
760  if(!fPOT && rhs.fPOT){
761  // If the RHS has a POT and we don't, copy it in (suitably scaled)
762  fPOT = rhs.fPOT * fLivetime/rhs.fLivetime;
763  }
764  // Otherwise, keep our own POT (if any)
765 
766  return *this;
767  }
768 
769  // That should have been all the cases. I definitely want to know what
770  // happened if it wasn't.
771  std::cout << "Spectrum::operator+=(). How did we get here? "
772  << fPOT << " " << fLivetime << " "
773  << rhs.fPOT << " " << rhs.fLivetime << std::endl;
774  abort();
775  }
