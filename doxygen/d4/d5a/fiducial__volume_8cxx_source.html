65  double min_z = std::min( fabs(vec[2] - paras.s_tpc_active_ZMin) , fabs(vec[2] - paras.s_tpc_active_ZMax));
66 
67  return ( (min_x<min_y) ? std::min(min_x,min_z) : std::min(min_y,min_z) );
