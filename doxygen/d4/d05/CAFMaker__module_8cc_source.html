294 void CAFMaker::InitVolumes() {
295  const geo::GeometryCore *geometry = lar::providerFrom<geo::Geometry>();
296 
297  // first the TPC volumes
298  for (auto const &cryo: geometry->IterateCryostats()) {
299  geo::GeometryCore::TPC_iterator iTPC = geometry->begin_TPC(cryo.ID()),
300  tend = geometry->end_TPC(cryo.ID());
301  std::vector<geo::BoxBoundedGeo> this_tpc_volumes;
302  while (iTPC != tend) {
303  geo::TPCGeo const& TPC = *iTPC;
304  this_tpc_volumes.push_back(TPC.ActiveBoundingBox());
305  iTPC++;
306  }
307  fTPCVolumes.push_back(std::move(this_tpc_volumes));
308  }
309 
310  // then combine them into active volumes
311  for (const std::vector<geo::BoxBoundedGeo> &tpcs: fTPCVolumes) {
312  double XMin = std::min_element(tpcs.begin(), tpcs.end(), [](auto &lhs, auto &rhs) { return lhs.MinX() < rhs.MinX(); })->MinX();
313  double YMin = std::min_element(tpcs.begin(), tpcs.end(), [](auto &lhs, auto &rhs) { return lhs.MinY() < rhs.MinY(); })->MinY();
314  double ZMin = std::min_element(tpcs.begin(), tpcs.end(), [](auto &lhs, auto &rhs) { return lhs.MinZ() < rhs.MinZ(); })->MinZ();
315 
316  double XMax = std::max_element(tpcs.begin(), tpcs.end(), [](auto &lhs, auto &rhs) { return lhs.MaxX() < rhs.MaxX(); })->MaxX();
317  double YMax = std::max_element(tpcs.begin(), tpcs.end(), [](auto &lhs, auto &rhs) { return lhs.MaxY() < rhs.MaxY(); })->MaxY();
318  double ZMax = std::max_element(tpcs.begin(), tpcs.end(), [](auto &lhs, auto &rhs) { return lhs.MaxZ() < rhs.MaxZ(); })->MaxZ();
319 
320  fActiveVolumes.emplace_back(XMin, XMax, YMin, YMax, ZMin, ZMax);
321  }
322 }

399  std::cout << "CAFMaker: unrecognized value for DetectorOverride parameter: '" << fParams.DetectorOverride() << "'" << std::endl;
400  abort();

404  const geo::GeometryCore* geom = lar::providerFrom<geo::Geometry>();
405 
406  std::string gdml = geom->GDMLFile();
407  gdml = basename(gdml.c_str()); // strip directory part
408  std::cout << "CAFMaker: Attempting to deduce detector from GDML file name: '" << gdml
409  << "' and configured detector name: '" << geom->DetectorName() << "'. ";
410  // Lowercase filename, in case it contains "SBND" or "Icarus" etc
411  for(unsigned int i = 0; i < gdml.size(); ++i) gdml[i] = std::tolower(gdml[i]);
412  // Do we find the string in either of the names?
413  const bool hasSBND = ((gdml.find("sbnd") != std::string::npos) ||
414  (geom->DetectorName().find("sbnd") != std::string::npos));
415  const bool hasIcarus = ((gdml.find("icarus") != std::string::npos) ||
416  (geom->DetectorName().find("icarus") != std::string::npos));
417 
418  // Either no evidence, or ambiguous evidence
419  if(hasSBND == hasIcarus){
420  std::cout << "Unable to automatically determine detector!" << std::endl;
421  if(override == kUNKNOWN) abort();
422  }
423  // Now must be one or the other
424  if(hasSBND){
425  fDet = kSBND;
426  std::cout << "Detected SBND" << std::endl;
427  }
428  if(hasIcarus){
429  fDet = kICARUS;
430  std::cout << "Detected Icarus" << std::endl;
431  }
432 
433  if(override != kUNKNOWN){
434  std::cout << "Detector set to ";

439  }

444  }

450
