212 void TPCNoiseFilter1DMC::configure(fhicl::ParameterSet const &pset)
213 {
214  fSigmaForTruncation = pset.get<float >("NSigmaForTrucation", 3.5);
215  fCoherentNoiseOffset = pset.get<size_t >("CoherentOffset", 0);
216  fStructuringElement = pset.get<size_t >("StructuringElement", 20);
217  fMorphWindow = pset.get<size_t >("FilterWindow", 10);
218  fThreshold = pset.get<std::vector<float> >("Threshold", std::vector<float>()={5.0,3.5,3.5});
219  fUseFFTFilter = pset.get<bool >("UseFFTFilter", true);
220  fDiagnosticOutput = pset.get<bool >("DiagnosticOutput", false);
221  fFilterModeVec = pset.get<std::vector<std::string>>("FilterModeVec", std::vector<std::string>()={"e","g","d"});
222 
223  fFFTSigmaValsVec = pset.get<FloatPairVec >("FFTSigmaVals", FloatPairVec()={{1.5,20.}, {1.5,20.}, {2.0,20.}});
224  fFFTCutoffValsVec = pset.get<FloatPairVec >("FFTCutoffVals", FloatPairVec()={{8.,800.}, {8.,800.}, {0.0,800.}});
225 
226  fGeometry = art::ServiceHandle<geo::Geometry const>{}.get();
227 
228  fFFTFilterFunctionVec.clear();
229 
230  if (fUseFFTFilter)
231  {
232  for(int plane = 0; plane < 3; plane++)
233  {
234  if (plane < 2) fFFTFilterFunctionVec.emplace_back(std::make_unique<icarus_signal_processing::WindowFFTFilter>(fFFTSigmaValsVec[plane], fFFTCutoffValsVec[plane]));
235  else fFFTFilterFunctionVec.emplace_back(std::make_unique<icarus_signal_processing::NoFFTFilter>());
236  }
237  }
238 
239  return;
240 }

242 void TPCNoiseFilter1DMC::process_fragment(detinfo::DetectorClocksData const&,
243  const daq::INoiseFilter::ChannelPlaneVec& channelPlaneVec,
244  const icarus_signal_processing::ArrayFloat& dataArray,
245  const size_t& coherentNoiseGrouping)
246 {
247  cet::cpu_timer theClockTotal;
248 
249  theClockTotal.start();
250 
251  // Recover the number of channels and ticks
252  unsigned int numChannels = dataArray.size();
253  unsigned int numTicks = dataArray[0].size();
254 
255  if (fSelectVals.size() < numChannels) fSelectVals.resize(numChannels, icarus_signal_processing::VectorBool(numTicks));
256  if (fROIVals.size() < numChannels) fROIVals.resize(numChannels, icarus_signal_processing::VectorBool(numTicks));
257  if (fRawWaveforms.size() < numChannels) fRawWaveforms.resize(numChannels, icarus_signal_processing::VectorFloat(numTicks));
258  if (fPedCorWaveforms.size() < numChannels) fPedCorWaveforms.resize(numChannels, icarus_signal_processing::VectorFloat(numTicks));
259  if (fIntrinsicRMS.size() < numChannels) fIntrinsicRMS.resize(numChannels, icarus_signal_processing::VectorFloat(numTicks));
260  if (fCorrectedMedians.size() < numChannels) fCorrectedMedians.resize(numChannels, icarus_signal_processing::VectorFloat(numTicks));
261  if (fWaveLessCoherent.size() < numChannels) fWaveLessCoherent.resize(numChannels, icarus_signal_processing::VectorFloat(numTicks));
262  if (fMorphedWaveforms.size() < numChannels) fMorphedWaveforms.resize(numChannels, icarus_signal_processing::VectorFloat(numTicks));
263 
264  if (fChannelIDVec.size() < numChannels) fChannelIDVec.resize(numChannels);
265  if (fPedestalVals.size() < numChannels) fPedestalVals.resize(numChannels);
266  if (fFullRMSVals.size() < numChannels) fFullRMSVals.resize(numChannels);
267  if (fTruncRMSVals.size() < numChannels) fTruncRMSVals.resize(numChannels);
268  if (fNumTruncBins.size() < numChannels) fNumTruncBins.resize(numChannels);
269  if (fRangeBins.size() < numChannels) fRangeBins.resize(numChannels);
270 
271  if (fThresholdVec.size() < numChannels) fThresholdVec.resize(numChannels / coherentNoiseGrouping);
272 
273  if (fFilterFunctionVec.size() < numChannels) fFilterFunctionVec.resize(numChannels);
274 
275  icarus_signal_processing::Denoiser1D denoiser;
276  icarus_signal_processing::WaveformTools<float> waveformTools;
277 
278  // Make a pass throught to do pedestal corrections and get raw waveform information
279  for(size_t idx = 0; idx < numChannels; idx++)
280  {
281  icarus_signal_processing::VectorFloat& pedCorDataVec = fPedCorWaveforms[idx];
282 
283  // Keep track of the channel
284  fChannelIDVec[idx] = channelPlaneVec[idx].first;
285 
286  // We need to recover info on which plane we have
287  std::vector<geo::WireID> widVec = fGeometry->ChannelToWire(fChannelIDVec[idx]);
288 
289  // Handle the filter function to use for this channel
290  // Note the modulus... this to enable a workaround for the wirecell 2D drift which misses the channels with no signal
291  unsigned int plane = channelPlaneVec[idx].second % 3;
292 
293  // Set the threshold which toggles between planes
294  fThresholdVec[idx / coherentNoiseGrouping] = fThreshold[plane];
295 
296  switch(fFilterModeVec[plane][0])
297  {
298  case 'd' :
299  fFilterFunctionVec[idx] = std::make_unique<icarus_signal_processing::Dilation1D>(fStructuringElement);
300  break;
301  case 'e' :
302  fFilterFunctionVec[idx] = std::make_unique<icarus_signal_processing::Erosion1D>(fStructuringElement);
303  break;
304  case 'g' :
305  fFilterFunctionVec[idx] = std::make_unique<icarus_signal_processing::Gradient1D>(fStructuringElement);
306  break;
307  case 'a' :
308  fFilterFunctionVec[idx] = std::make_unique<icarus_signal_processing::Average1D>(fStructuringElement);
309  break;
310  case 'm' :
311  fFilterFunctionVec[idx] = std::make_unique<icarus_signal_processing::Median1D>(fStructuringElement);
312  break;
313  default:
314  std::cout << "***** FOUND NO MATCH FOR TYPE: " << fFilterModeVec[plane] << ", plane " << plane << " DURING INITIALIZATION OF FILTER FUNCTIONS IN TPCNoiseFilter1DMC" << std::endl;
315  break;
316  }
317 
318  // Now determine the pedestal and correct for it
319  waveformTools.getPedestalCorrectedWaveform(dataArray[idx],
320  pedCorDataVec,
321  fSigmaForTruncation,
322  fPedestalVals[idx],
323  fFullRMSVals[idx],
324  fTruncRMSVals[idx],
325  fNumTruncBins[idx],
326  fRangeBins[idx]);
327 
328  // Convolve with a filter function
329  if (fUseFFTFilter) (*fFFTFilterFunctionVec[plane])(pedCorDataVec);
330 
331  // Make sure our selection and ROI arrays are initialized
332  std::fill(fSelectVals[idx].begin(),fSelectVals[idx].end(),false);
333  }
334 
335  denoiser(fWaveLessCoherent.begin(),
336  fPedCorWaveforms.begin(),
337  fMorphedWaveforms.begin(),
338  fIntrinsicRMS.begin(),
339  fSelectVals.begin(),
340  fROIVals.begin(),
341  fCorrectedMedians.begin(),
342  fFilterFunctionVec.begin(),
343  fThresholdVec,
344  numChannels,
345  coherentNoiseGrouping,
346  fCoherentNoiseOffset,
347  fMorphWindow);
348 
349  theClockTotal.stop();
350 
351  double totalTime = theClockTotal.accumulated_real_time();
352 
353  mf::LogDebug("TPCNoiseFilter1DMC") << " *totalTime: " << totalTime << std::endl;
354 
355  return;
356 }
