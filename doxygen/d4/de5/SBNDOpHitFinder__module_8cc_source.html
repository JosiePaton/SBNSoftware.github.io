104  SBNDOpHitFinder::SBNDOpHitFinder(const fhicl::ParameterSet & pset):
105  EDProducer{pset},
106  fPulseRecoMgr()
107  {
108  // Indicate that the Input Module comes from .fcl
109  fInputModule = pset.get< std::string >("InputModule");
110  fGenModule = pset.get< std::string >("GenModule");
111  fInputLabels = pset.get< std::vector< std::string > >("InputLabels");
112 
113  for (auto const& ch : pset.get< std::vector< unsigned int > >
114  ("ChannelMasks", std::vector< unsigned int >()))
115  fChannelMasks.insert(ch);
116 
117  _pd_to_use = pset.get< std::vector< std::string > >("PD", _pd_to_use);
118  fElectronics = pset.get< std::string >("Electronics");
119  _opch_to_use = this->PDNamesToList(_pd_to_use);
120 
121  fDaphne_Freq = pset.get< float >("DaphneFreq");
122  fHitThreshold = pset.get< float >("HitThreshold");
123  bool useCalibrator = pset.get< bool > ("UseCalibrator", false);
124 
125  auto const& geometry(*lar::providerFrom< geo::Geometry >());
126  fMaxOpChannel = geometry.MaxOpChannel();
127 
128  if (useCalibrator) {
129  // If useCalibrator, get it from ART
130  fCalib = lar::providerFrom<calib::IPhotonCalibratorService>();
131  }
132  else {
133  // If not useCalibrator, make an internal one based
134  // on fhicl settings to hit finder.
135  bool areaToPE = pset.get< bool > ("AreaToPE");
136  float SPEArea = pset.get< float >("SPEArea");
137  float SPEShift = pset.get< float >("SPEShift", 0.);
138 
139  // Reproduce behavior from GetSPEScales()
140  if (!areaToPE) SPEArea = 20;
141 
142  // Delete and replace if we are reconfiguring
143  if (fCalib) {
144  delete fCalib;
145  }
146 
147  fCalib = new calib::PhotonCalibratorStandard(SPEArea, SPEShift, areaToPE);
148  }
149 
150  // Initialize the hit finder algorithm
151  auto const hit_alg_pset = pset.get< fhicl::ParameterSet >("HitAlgoPset");
152  std::string threshAlgName = hit_alg_pset.get< std::string >("Name");
153  if (threshAlgName == "Threshold")
154  fThreshAlg = new pmtana::AlgoThreshold(hit_alg_pset);
155  else if (threshAlgName == "SiPM")
156  fThreshAlg = new pmtana::AlgoSiPM(hit_alg_pset);
157  else if (threshAlgName == "SlidingWindow")
158  fThreshAlg = new pmtana::AlgoSlidingWindow(hit_alg_pset);
159  else if (threshAlgName == "FixedWindow")
160  fThreshAlg = new pmtana::AlgoFixedWindow(hit_alg_pset);
161  else if (threshAlgName == "CFD" )
162  fThreshAlg = new pmtana::AlgoCFD(hit_alg_pset);
163  else throw art::Exception(art::errors::UnimplementedFeature)
164  << "Cannot find implementation for "
165  << threshAlgName << " algorithm.\n";
166 
167  auto const ped_alg_pset = pset.get< fhicl::ParameterSet >("PedAlgoPset");
168  std::string pedAlgName = ped_alg_pset.get< std::string >("Name");
169  if (pedAlgName == "Edges")
170  fPedAlg = new pmtana::PedAlgoEdges(ped_alg_pset);
171  else if (pedAlgName == "RollingMean")
172  fPedAlg = new pmtana::PedAlgoRollingMean(ped_alg_pset);
173  else if (pedAlgName == "UB" )
174  fPedAlg = new pmtana::PedAlgoUB(ped_alg_pset);
175  else throw art::Exception(art::errors::UnimplementedFeature)
176  << "Cannot find implementation for "
177  << pedAlgName << " algorithm.\n";
178 
179  produces< std::vector< recob::OpHit > >();
180 
181  fPulseRecoMgr.AddRecoAlgo(fThreshAlg);
182  fPulseRecoMgr.SetDefaultPedAlgo(fPedAlg);
183 
184  }

197  void SBNDOpHitFinder::produce(art::Event& evt)
198  {
199 
200  // These is the storage pointer we will put in the event
201  std::unique_ptr< std::vector< recob::OpHit > >
202  HitPtrFinal(new std::vector< recob::OpHit >);
203 
204  // These is a temporary storage pointer
205  std::unique_ptr< std::vector< recob::OpHit > >
206  HitPtr(new std::vector< recob::OpHit >);
207 
208  std::vector< const sim::BeamGateInfo* > beamGateArray;
209  try
210  {
211  evt.getView(fGenModule, beamGateArray);
212  }
213  catch (art::Exception const& err)
214  {
215  if ( err.categoryCode() != art::errors::ProductNotFound ) throw;
216  }
217 
218  auto const& geometry(*lar::providerFrom< geo::Geometry >());
219  auto const clockData_CAEN = art::ServiceHandle<detinfo::DetectorClocksService>()->DataFor(evt);
220  auto const& calibrator(*fCalib);
221  detinfo::ElecClock const optical_clock_daphne = detinfo::ElecClock(clockData_CAEN.OpticalClock().Time(),
222  clockData_CAEN.OpticalClock().FramePeriod(),
223  fDaphne_Freq);
224  detinfo::DetectorClocksData const& clockData_Daphne = detinfo::DetectorClocksData(-clockData_CAEN.G4ToElecTime(0),
225  clockData_CAEN.TriggerOffsetTPC(),
226  clockData_CAEN.TriggerTime(),
227  clockData_CAEN.BeamGateTime(),
228  clockData_CAEN.TPCClock(),
229  optical_clock_daphne,
230  clockData_CAEN.TriggerClock(),
231  clockData_CAEN.ExternalClock());
232  detinfo::DetectorClocksData const& clockData = (fElectronics=="Daphne") ? clockData_Daphne : clockData_CAEN;
233  // std::cout<<"@rodrigoa debug: frecuencies "<<clockData_Daphne.OpticalClock().Frequency()<<" "<<clockData.OpticalClock().Frequency()<<std::endl;
234  //
235  // Get the pulses from the event
236  //
237 
238  // Load pulses into WaveformVector
239  if(fChannelMasks.empty() && _opch_to_use.empty() && fInputLabels.size()<2) {
240  art::Handle< std::vector< raw::OpDetWaveform > > wfHandle;
241  if(fInputLabels.empty())
242  evt.getByLabel(fInputModule, wfHandle);
243  else
244  evt.getByLabel(fInputModule, fInputLabels.front(), wfHandle);
245  assert(wfHandle.isValid());
246  RunHitFinder(*wfHandle,
247  *HitPtr,
248  fPulseRecoMgr,
249  *fThreshAlg,
250  geometry,
251  fHitThreshold,
252  clockData,
253  calibrator);
254  } else {
255 
256  // Reserve a large enough array
257  int totalsize = 0;
258  for (auto label : fInputLabels)
259  {
260  art::Handle< std::vector< raw::OpDetWaveform > > wfHandle;
261  evt.getByLabel(fInputModule, label, wfHandle);
262  if (!wfHandle.isValid()) continue; // Skip non-existent collections
263  totalsize += wfHandle->size();
264  }
265 
266  std::vector< raw::OpDetWaveform > WaveformVector;
267  WaveformVector.reserve(totalsize);
268 
269  for (auto label : fInputLabels)
270  {
271  art::Handle< std::vector< raw::OpDetWaveform > > wfHandle;
272  evt.getByLabel(fInputModule, label, wfHandle);
273  if (!wfHandle.isValid()) continue; // Skip non-existent collections
274 
275  for(auto const& wf : *wfHandle)
276  {
277  // If this channel is in the channel mask, ingore it
278  if ( fChannelMasks.find(wf.ChannelNumber()) != fChannelMasks.end() ) continue;
279  // If this PDS in not in the list of PDS to use, ingore it
280  if ( std::find(_opch_to_use.begin(), _opch_to_use.end(), wf.ChannelNumber()) == _opch_to_use.end() ) continue;
281 
282  WaveformVector.push_back(wf);
283  }
284  }
285 
286  RunHitFinder(WaveformVector,
287  *HitPtr,
288  fPulseRecoMgr,
289  *fThreshAlg,
290  geometry,
291  fHitThreshold,
292  clockData,
293  calibrator);
294  // for (auto h : *HitPtr)
295  // std::cout << "> ophit time " << h.PeakTime()
296  // << ", corrected " << h.PeakTime() - clockData.TriggerTime()
297  // << ", area " << h.Area()
298  // << ", pe " << h.PE() << std::endl;
299  }
300 
301  // Now correct the time. Unfortunately, there are no setter methods for OpHits,
302  // so we have to make a new OpHit vector.
303  for (auto h : *HitPtr) {
304  (*HitPtrFinal).emplace_back(h.OpChannel(),
305  h.PeakTime() + clockData.TriggerTime(),
306  h.PeakTimeAbs(),
307  h.Frame(),
308  h.Width(),
309  h.Area(),
310  h.Amplitude(),
311  h.PE(),
312  0.0);
313  }
314  // Store results into the event
315  evt.put(std::move(HitPtrFinal));
316 
317  }
