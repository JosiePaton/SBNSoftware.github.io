362 void ROIFinder::processPlane(size_t idx,
363  art::Event& event,
364  const PlaneIDVec& planeIDVec,
365  const PlaneIDToDataPairMap& planeIDToDataPairMap,
366  std::vector<recob::Wire>& wireColVec,
367  std::vector<recob::Wire>& morphedVec) const
368 {
369  // Recover the planeID for this thread
370  const geo::PlaneID& planeID = planeIDVec[idx];
371 
372  // And the data array
373  PlaneIDToDataPairMap::const_iterator mapItr = planeIDToDataPairMap.find(planeID);
374 
375  if (mapItr == planeIDToDataPairMap.end())
376  {
377  std::cout << "We know this cannot happen" << std::endl;
378  return;
379  }
380 
381  const PlaneIDToDataPair& planeIDToDataPair = mapItr->second;
382 
383  const icarus_signal_processing::ArrayFloat& dataArray = planeIDToDataPair.second();
384  const std::vector<raw::ChannelID_t>& channelVec = planeIDToDataPair.first;
385 
386  // Keep track of our selected values
387  icarus_signal_processing::ArrayFloat outputArray(dataArray.size(),icarus_signal_processing::VectorFloat(dataArray[0].size(),0.));
388  icarus_signal_processing::ArrayBool selectedVals(dataArray.size(),icarus_signal_processing::VectorBool(dataArray[0].size(),false));
389 
390  fROIToolMap.at(planeID.Plane)->FindROIs(event, dataArray, mapItr->first, outputArray, selectedVals);
391 
392 // icarus_signal_processing::ArrayFloat morphedWaveforms(dataArray.size());
393 
394  // Copy the "morphed" array
395  if (fOutputMorphed)
396  {
397  for(size_t waveIdx = 0; waveIdx < outputArray.size(); waveIdx++)
398  {
399  // skip if a bad channbel
400  if (channelVec[idx] >= 100000) continue;
401 
402  // First get a lock to make sure we don't conflict
403  tbb::spin_mutex::scoped_lock lock(roifinderSpinMutex);
404 
405  recob::Wire::RegionsOfInterest_t ROIVec;
406 
407  ROIVec.add_range(0, std::move(outputArray[waveIdx]));
408 
409  raw::ChannelID_t channel = channelVec[waveIdx];
410  geo::View_t view = fGeometry->View(channel);
411 
412  std::vector<geo::WireID> chanIDVec = fGeometry->ChannelToWire(channel);
413 
414  morphedVec.push_back(recob::WireCreator(std::move(ROIVec),channel,view).move());
415  }
416  }
417 
418  // Ok, now go through the refined selected values array and find ROIs
419  // Define the ROI and its container
420  using CandidateROI = std::pair<size_t, size_t>;
421  using CandidateROIVec = std::vector<CandidateROI>;
422 
423  size_t leadTrail(0);
424 
425  for(size_t waveIdx = 0; waveIdx < selectedVals.size(); waveIdx++)
426  {
427  // Skip if a bad channel
428  if (channelVec[waveIdx] >= 100000)
429  {
430  std::cout << "==> found an unexpected channel number: " << channelVec[waveIdx] << std::endl;
431  continue;
432  }
433 
434  // Set up an object...
435  CandidateROIVec candidateROIVec;
436 
437  // Search for ROIs in current waveform
438  const icarus_signal_processing::VectorBool& selVals = selectedVals[waveIdx];
439 
440  size_t idx(2);
441 
442  while(idx < selVals.size())
443  {
444  if (selVals[idx])
445  {
446  Size_t startTick = idx >= leadTrail ? idx - leadTrail : 0;
447 
448  while(idx < selVals.size() && selVals[idx]) idx++;
449 
450  size_t stopTick = idx < selVals.size() - leadTrail ? idx + leadTrail : selVals.size();
451 
452  candidateROIVec.emplace_back(startTick, stopTick);
453  }
454 
455  idx++;
456  }
457 
458  // merge overlapping (or touching) ROI's
459  if(candidateROIVec.size() > 1)
460  {
461  // temporary vector for merged ROIs
462  CandidateROIVec tempRoiVec;
463 
464  // Loop through candidate roi's
465  size_t startRoi = candidateROIVec.front().first;
466  size_t stopRoi = startRoi;
467 
468  for(auto& roi : candidateROIVec)
469  {
470  // Should we merge roi's?
471  if (roi.first <= stopRoi)
472  {
473  // Make sure the merge gets the right start/end times
474  startRoi = std::min(startRoi,roi.first);
475  stopRoi = std::max(stopRoi,roi.second);
476  }
477  else
478  {
479  tempRoiVec.emplace_back(startRoi,stopRoi);
480 
481  startRoi = roi.first;
482  stopRoi = roi.second;
483  }
484  }
485 
486  // Make sure to get the last one
487  tempRoiVec.emplace_back(startRoi,stopRoi);
488 
489  candidateROIVec = tempRoiVec;
490  }
491 
492  // If no candidates no need for further effort
493  if (!candidateROIVec.empty())
494  {
495  // vector that will be moved into the Wire object
496  recob::Wire::RegionsOfInterest_t ROIVec;
497 
498  // Check if we have possible overlap wires where we need to merge the previous results with new results
499  if (planeID.Plane > 0)
500  {
501  raw::ChannelID_t channel = channelVec[waveIdx];
502 
503  std::vector<geo::WireID> wireIDVec = fGeometry->ChannelToWire(channel);
504 
505  if (wireIDVec.size() > 1)
506  {
507  std::vector<recob::Wire>::iterator wireItr = std::find_if(wireColVec.begin(),wireColVec.end(),[channel](const auto& wire){return wire.Channel() == channel;});
508 
509  if (wireItr != wireColVec.end())
510  {
511  ROIVec = wireItr->SignalROI();
512 
513  // Avoid duplicate entries by erasing the previous instance
514  wireColVec.erase(wireItr);
515  }
516  }
517  }
518 
519  const icarus_signal_processing::VectorFloat& waveform = dataArray[waveIdx];
520 
521  // We need to copy the deconvolved (and corrected) waveform ROI's
522  for(const auto& candROI : candidateROIVec)
523  {
524  // First up: copy out the relevent ADC bins into the ROI holder
525  size_t roiLen = candROI.second - candROI.first;
526  size_t firstBin = candROI.first;
527 
528  icarus_signal_processing::VectorFloat holder(roiLen);
529 
530  std::copy(waveform.begin()+candROI.first, waveform.begin()+candROI.second, holder.begin());
531 
532  // Now we do the baseline determination and correct the ROI
533  // For now we are going to reset to the minimum element
534  // Get slope/offset from first to last ticks
535  if (fCorrectROIBaseline && holder.size() > fMinSizeForCorrection && holder.size() < fMaxSizeForCorrection)
536  {
537  // Try to find the minimum value in the leading and trailing bins
538  size_t nBins = holder.size()/3;
539  icarus_signal_processing::VectorFloat::iterator firstItr = std::min_element(holder.begin(),holder.begin()+nBins);
540  icarus_signal_processing::VectorFloat::iterator lastItr = std::min_element(holder.end()-nBins,holder.end());
541 
542  size_t newSize = std::distance(firstItr,lastItr) + 1;
543  float dADC = (*lastItr - *firstItr) / float(newSize);
544  float offset = *firstItr;
545 
546  for(size_t binIdx = 0; binIdx < newSize; binIdx++)
547  {
548  holder[binIdx] = *(firstItr + binIdx) - offset;
549  offset += dADC;
550  }
551 
552  firstBin += std::distance(holder.begin(),firstItr);
553 
554  holder.resize(newSize);
555  }
556 
557  // add the range into ROIVec
558  ROIVec.add_range(firstBin, std::move(holder));
559  }
560 
561  // Check for emptiness
562  if (!ROIVec.empty())
563  {
564  // First get a lock to make sure we don't conflict
565  tbb::spin_mutex::scoped_lock lock(roifinderSpinMutex);
566 
567  raw::ChannelID_t channel = channelVec[waveIdx];
568  geo::View_t view = fGeometry->View(channel);
569 
570  // Since we process logical TPC images we need to watch for duplicating entries
571  // We can do that by checking to see if a channel has already been added...
572  std::vector<geo::WireID> wireIDVec = fGeometry->ChannelToWire(channel);
573 
574  wireColVec.push_back(recob::WireCreator(std::move(ROIVec),channel,view).move());
575  }
576  }
577  }
578 
579  return;

582 float ROIFinder::getMedian(icarus_signal_processing::VectorFloat vals, const unsigned int nVals) const
583 {
584  float median(0.);
585 
586  if (nVals > 2)
587  {
588  if (nVals % 2 == 0)
589  {
590  const auto m1 = vals.begin() + nVals / 2 - 1;
591  const auto m2 = vals.begin() + nVals / 2;
592  std::nth_element(vals.begin(), m1, vals.begin() + nVals);
593  const auto e1 = *m1;
594  std::nth_element(vals.begin(), m2, vals.begin() + nVals);
595  const auto e2 = *m2;
596  median = (e1 + e2) / 2.0;
597  }
598  else
599  {
600  const auto m = vals.begin() + nVals / 2;
601  std::nth_element(vals.begin(), m, vals.begin() + nVals);
602  median = *m;
603  }
604  }
605 
606  return median;
