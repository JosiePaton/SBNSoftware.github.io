193 void TrackHitAna::reconfigure(fhicl::ParameterSet const& p)

196  // to p.get<TYPE> must match names in the .fcl file.
197  fHitProducerLabelVec = p.get< std::vector<art::InputTag> >("HitModuleLabel", std::vector<art::InputTag>() = {"gauss"});
198  fPFParticleProducerLabelVec = p.get< std::vector<art::InputTag> >("PFParticleProducerLabel", std::vector<art::InputTag>() = {"cluster3d"});
199  fTrackProducerLabelVec = p.get< std::vector<art::InputTag> >("TrackProducerLabel", std::vector<art::InputTag>() = {"trackkalmanhit"});
200  fWireProducerLabelVec = p.get< std::vector<art::InputTag> >("WireProducerLabel", std::vector<art::InputTag>() = {"caldata"});
201 
202  // Implement the tools for handling the responses
203  const std::vector<fhicl::ParameterSet>& hitHistogramToolVec = p.get<std::vector<fhicl::ParameterSet>>("HitHistogramToolList");
204 
205  for(auto& hitHistogramTool : hitHistogramToolVec)
206  fHitHistogramToolVec.push_back(art::make_tool<IHitHistogramTool>(hitHistogramTool));
207 
208  return;

211 //-----------------------------------------------------------------------
212 void TrackHitAna::analyze(const art::Event& event)
213 {
214  // Start by fetching some basic event information for our n-tuple.
215  fEvent = event.id().event();
216  fRun = event.run();
217  fSubRun = event.subRun();
218 
219  fNumEvents++;
220 
221  for(const auto& hitLabel : fHitProducerLabelVec)
222  {
223  // Make a pass through all hits to make contrasting plots
224  art::Handle< std::vector<recob::Hit> > hitHandle;
225  event.getByLabel(hitLabel, hitHandle);
226 
227  if (hitHandle.isValid())
228  {
229  IHitHistogramTool::HitPtrVec allHitVec;
230  art::fill_ptr_vector(allHitVec, hitHandle);
231 
232  for(auto& hitHistTool : fHitHistogramToolVec) hitHistTool->fillHistograms(allHitVec);
233  }
234  }
235 
236  return;
237 }
