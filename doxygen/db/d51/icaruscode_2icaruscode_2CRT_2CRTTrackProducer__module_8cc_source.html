392  // Function to calculate the mean and rms from a vector of values
393  void vmanip(std::vector<float> v, float* ave, float* rms)
394  {
395  *ave=0.0; *rms =0.0;
396  if (v.size()>0) {
397  // find the mean and *rms of all the vector elements
398  double sum = std::accumulate(v.begin(), v.end(), 0.0);
399  double mean = sum / v.size();
400  *ave=mean;
401 
402  if (v.size()>1) {
403  double sq_sum = std::inner_product(v.begin(), v.end(), v.begin(), 0.0);
404  double stdev = std::sqrt(sq_sum / v.size() - mean * mean);
405  *rms=stdev;
406  }
407  }
408  } // vmanip()
