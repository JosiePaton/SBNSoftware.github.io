307 //-----------------------------------------------------------------------
308 void TPCPurityMonitor::produce(art::Event& event)
309 {
310  //setup output vector#include "lardataobj/RecoBase/TrackHitMeta.h"
311  std::unique_ptr< std::vector<anab::TPCPurityInfo> > outputPtrVector(new std::vector<anab::TPCPurityInfo>());
312 
313  fNumEvents++;
314 
315  // Recover the detector properties.
316  auto const clockData = art::ServiceHandle<detinfo::DetectorClocksService const>()->DataFor(event);
317 
318  fSamplingRate = 1.e-3 * sampling_rate(clockData); // Note sampling rate is in ns, convert to us
319 
320  for(const auto& trackLabel : fTrackLabelVec)
321  {
322  // Make a pass through all hits to make contrasting plots
323  art::Handle< std::vector<recob::Track>> trackHandle;
324  event.getByLabel(trackLabel, trackHandle);
325 
326  if (!trackHandle.isValid()) continue;
327 
328  // Recover the collection of associations between tracks and hits
329  art::FindManyP<recob::Hit,recob::TrackHitMeta> trackHitAssns(trackHandle, event, trackLabel);
330 
331  // Loop over tracks and recover hits
332  for(size_t trackIdx = 0; trackIdx < trackHandle->size(); trackIdx++)
333  {
334  art::Ptr<recob::Track> track(trackHandle,trackIdx);
335 
336  const std::vector<art::Ptr<recob::Hit>>& trackHitsVec(trackHitAssns.at(track.key()));
337 // const std::vector<const recob::TrackHitMeta*> metaHitsVec = trackHitAssns.data(track.key());
338 
339  // Focus on selected hits:
340  // 1) Pick out hits on a single plane given by fhicl parameter
341  // 2) multiplicity == 1 which should give us clean gaussian shaped pulses
342  std::vector<art::Ptr<recob::Hit>> selectedTrackHitsVec;
343 
344  for(auto& hit : trackHitsVec)
345  {
346  if (hit->WireID().Plane == fSelectedPlane && hit->Multiplicity() == 1) selectedTrackHitsVec.emplace_back(hit);
347  }
348 
349  // Need a minimum number of hits
350  if (selectedTrackHitsVec.size() < fMinNumHits) continue;
351 
352 // std::cout << "--> trackMeta index: " << metaHitsVec.front()->Index() << ", Dx: " << metaHitsVec.front()->Dx() << ", meta size: " << metaHitsVec.size() << ", track:" << trackHitsVec.size() << std::endl;
353 
354  // Sort hits by increasing time
355  std::sort(selectedTrackHitsVec.begin(),selectedTrackHitsVec.end(),[](const auto& left, const auto& right){return left->PeakTime() < right->PeakTime();});
356 
357  // Require track to have a minimum range in ticks
358  if (selectedTrackHitsVec.back()->PeakTime() - selectedTrackHitsVec.front()->PeakTime() < fMinTickRange) continue;
359 
360  // At this point we should have a vector of art::Ptrs to hits on the selected plane
361  // So we should be able to now transition to computing the attenuation
362  // Start by forming a vector of pairs of the time (in ticks) and the ln of charge derated by an assumed lifetime
363  HitStatusChargePairVec hitStatusChargePairVec;
364 
365  float firstHitTime(selectedTrackHitsVec.front()->PeakTime());
366 
367  for(const auto& hit : selectedTrackHitsVec)
368  {
369  float charge = fUseHitIntegral ? hit->Integral() : hit->SummedADC();
370 // float logCharge = charge * exp(fSamplingRate * (hit->PeakTime() - firstHitTime) / fAssumedELifetime);
371 
372  hitStatusChargePairVec.emplace_back(hit.get(),StatusChargePair(true,log(charge)));
373  }
374 
375  // Drop the smallest and largest charges
376 // std::sort(HitStatusChargePairVec.begin(),HitStatusChargePairVec.end(),[](const auto& left, const auto& right){return left.second.second < right.second.second;});
377 
378  size_t numOrig = hitStatusChargePairVec.size();
379  size_t lowCutIdx = fMinRejectFraction * numOrig;
380  size_t hiCutIdx = fMaxRejectFraction * numOrig;
381 
382  // Will require a minimum number of hits left over to proceed
383  if (lowCutIdx + 10 >= hiCutIdx)
384  {
385  std::cout << "*****>>>> lowCutIdx >= hiCutIdx: " << lowCutIdx << ", " << hiCutIdx << std::endl;
386  continue;
387  }
388 
389 // HitStatusChargePairVec(hitStatusChargePairVec.begin() + lowCutIdx, hitStatusChargePairVec.begin() + hiCutIdx).swap(hitStatusChargePairVec);
390 
391  // Put back in time order
392 // std::sort(hitStatusChargePairVec.begin(),hitStatusChargePairVec.end(),[](const auto& left, const auto& right){return left.first->PeakTime() < right.first->PeakTime();});
393 
394  // Tag the leading and trailing hits so as to not use them
395  std::transform(hitStatusChargePairVec.begin(),hitStatusChargePairVec.begin()+lowCutIdx,hitStatusChargePairVec.begin(), [](const auto& hitPair){return HitStatusChargePair(hitPair.first,StatusChargePair(false,hitPair.second.second));});
396  std::transform(hitStatusChargePairVec.begin()+hiCutIdx,hitStatusChargePairVec.end(),hitStatusChargePairVec.begin()+hiCutIdx,[](const auto& hitPair){return HitStatusChargePair(hitPair.first,StatusChargePair(false,hitPair.second.second));});
397 
398  PrincipalComponents pca;
399 
400  GetPrincipalComponents(hitStatusChargePairVec, pca);
401 
402  // Reject the outliers
403  RejectOutliers(hitStatusChargePairVec, pca);
404 
405  // Recompute the pca
406  GetPrincipalComponents(hitStatusChargePairVec, pca);
407 
408  const PrincipalComponents::EigenVectors& eigenVectors = pca.getEigenVectors();
409 
410  double attenuation = eigenVectors.row(1)[1] / eigenVectors.row(1)[0];
411 
412  // Want to find the wire range (or should it be the number of wires?)
413  unsigned maxWire(0);
414  unsigned minWire(100000);
415 
416  std::set<unsigned> usedWiresSet;
417 
418  for(const auto& hitPair : hitStatusChargePairVec)
419  {
420  unsigned wire = hitPair.first->WireID().Wire;
421 
422  usedWiresSet.insert(wire);
423 
424  if (wire > maxWire) maxWire = wire;
425  if (wire < minWire) minWire = wire;
426  }
427 
428  geo::WireID wireID = hitStatusChargePairVec.front().first->WireID();
429 
430  anab::TPCPurityInfo purityInfo;
431 
432  purityInfo.Run = event.run();
433  purityInfo.Subrun = event.subRun();
434  purityInfo.Event = event.event();
435  purityInfo.Cryostat = wireID.Cryostat;
436  purityInfo.TPC = wireID.TPC;
437  purityInfo.Wires = usedWiresSet.size(); //maxWire - minWire;
438  purityInfo.Ticks = hitStatusChargePairVec.back().first->PeakTime() - firstHitTime;
439  purityInfo.Attenuation = -attenuation;
440  purityInfo.FracError = std::sqrt(pca.getEigenValues()[0] / pca.getEigenValues()[1]);
441 
442  outputPtrVector->emplace_back(purityInfo);
443 
444  if (fDiagnosticTuple)
445  {
446  fRunNumber = event.run();
447  fSubRunNumber = event.subRun();
448  fEventNumber = event.event();
449  fCryostat = wireID.Cryostat;
450  fTPC = wireID.TPC;
451  fTrackIdx = trackIdx;
452  fWireRange = maxWire - minWire;
453  fWires = usedWiresSet.size();
454  fTicks = hitStatusChargePairVec.back().first->PeakTime() - firstHitTime;
455  fAttenuation = -attenuation;
456  fError = std::sqrt(pca.getEigenValues()[0] / pca.getEigenValues()[1]);
457 
458  const geo::Point_t& trackStart = track->Start();
459  const geo::Vector_t trackDir = track->StartDirection();
460 
461  fTrackStartXVec.emplace_back(trackStart.X());
462  fTrackStartYVec.emplace_back(trackStart.Y());
463  fTrackStartZVec.emplace_back(trackStart.Z());
464  fTrackDirXVec.emplace_back(trackDir.X());
465  fTrackDirYVec.emplace_back(trackDir.Y());
466  fTrackDirZVec.emplace_back(trackDir.Z());
467 
468  for(size_t rowIdx = 0; rowIdx < 2; rowIdx++)
469  {
470  for(size_t colIdx = 0; colIdx < 2; colIdx++) fPCAAxes.emplace_back(eigenVectors.row(rowIdx)[colIdx]);
471  }
472 
473  fEigenValues.emplace_back(pca.getEigenValues()[0]);
474  fEigenValues.emplace_back(pca.getEigenValues()[1]);
475 
476  fMeanPosition.emplace_back(pca.getAvePosition()[0]);
477  fMeanPosition.emplace_back(pca.getAvePosition()[1]);
478 
479  for(const auto& hitPair : hitStatusChargePairVec)
480  {
481  double charge = fUseHitIntegral ? hitPair.first->Integral() : hitPair.first->SummedADC();
482 
483  fTickVec.emplace_back(hitPair.first->PeakTime());
484  fChargeVec.emplace_back(charge);
485  fGoodHitVec.emplace_back(hitPair.second.first);
486  }
487 
488  fDiagnosticTree->Fill();
489 
490  fTrackStartXVec.clear();
491  fTrackStartYVec.clear();
492  fTrackStartZVec.clear();
493  fTrackDirXVec.clear();
494  fTrackDirYVec.clear();
495  fTrackDirZVec.clear();
496  fPCAAxes.clear();
497  fEigenValues.clear();
498  fMeanPosition.clear();
499  fTickVec.clear();
500  fChargeVec.clear();
501  fGoodHitVec.clear();
502  }
503  }
504  }
505 
506  //put info onto the event
507  event.put(std::move(outputPtrVector));
508 
509  return;
510 }

577 void TPCPurityMonitor::GetPrincipalComponents(const HitStatusChargePairVec& hitPairVector, PrincipalComponents& pca) const
578 {
579  // Run through the HitPairList and get the mean position of all the hits
580  Eigen::Vector2d meanPos(Eigen::Vector2d::Zero());
581  double meanWeightSum(0.);
582  int numPairsInt(0);
583 
584  float startTime = hitPairVector.front().first->PeakTime();
585 
586  for (const auto& hitPair : hitPairVector)
587  {
588  if (!hitPair.second.first) continue;
589 
590  const recob::Hit* hit = hitPair.first;
591 
592  // Weight the hit by the peak time difference significance
593  double weight = fWeightByChiSq ? 1./hit->GoodnessOfFit() : 1.;
594  double charge = fUseHitIntegral ? hit->Integral() : hit->SummedADC();
595 
596  meanPos(0) += fSamplingRate * (hit->PeakTime() - startTime) * weight;
597  meanPos(1) += std::log(charge) * weight;
598  numPairsInt++;
599 
600  meanWeightSum += weight;
601  }
602 
603  meanPos /= meanWeightSum;
604 
605  // Define elements of our covariance matrix
606  double xi2(0.);
607  double xiyi(0.);
608  double yi2(0.0);
609  double weightSum(0.);
610 
611  // Back through the hits to build the matrix
612  for (const auto& hitPair : hitPairVector)
613  {
614  if (!hitPair.second.first) continue;
615 
616  const recob::Hit* hit = hitPair.first;
617 
618  double weight = fWeightByChiSq ? 1./hit->GoodnessOfFit() : 1.;
619  double charge = fUseHitIntegral ? hit->Integral() : hit->SummedADC();
620 
621  double x = (fSamplingRate * (hit->PeakTime() - startTime) - meanPos(0)) * weight;
622  double y = (std::log(charge) - meanPos(1)) * weight;
623 
624  weightSum += weight * weight;
625 
626  xi2 += x * x;
627  xiyi += x * y;
628  yi2 += y * y;
629  }
630 
631  // Using Eigen package
632  Eigen::Matrix2d sig;
633 
634  sig << xi2, xiyi, xiyi, yi2;
635 
636  sig *= 1. / weightSum;
637 
638  Eigen::SelfAdjointEigenSolver<Eigen::Matrix2d> eigenMat(sig);
639 
640  if (eigenMat.info() == Eigen::ComputationInfo::Success)
641  {
642  // Now copy output
643  // The returned eigen values and vectors will be returned in an xyz system where x is the smallest spread,
644  // y is the next smallest and z is the largest. Adopt that convention going forward
645  PrincipalComponents::EigenValues eigenVals = eigenMat.eigenvalues();
646  PrincipalComponents::EigenVectors eigenVecs = eigenMat.eigenvectors().transpose();
647 
648  // Store away
649  // NOTE: the major axis will be the second entry, the minor axis will be the first
650  pca = PrincipalComponents(true, numPairsInt, eigenVals, eigenVecs, meanPos);
651  }
652  else
653  {
654  mf::LogDebug("Cluster3D") << "PCA decompose failure, numPairs = " << numPairsInt << std::endl;
655  pca = PrincipalComponents();
656  }
657 
658  return;
659 }

661 void TPCPurityMonitor::RejectOutliers(HitStatusChargePairVec& hitPairVector, const PrincipalComponents& pca) const
662 {
663  double slope = pca.getEigenVectors().row(1)[1] / pca.getEigenVectors().row(1)[0];
664  const Eigen::Vector2d& avePos = pca.getAvePosition();
665 
666  // We assume the input vector has been time ordered
667  double firstHitTime = hitPairVector.front().first->PeakTime();
668 
669  for(auto& hitPair : hitPairVector)
670  {
671  // We are only interested in the "good" hits here
672  if (hitPair.second.first)
673  {
674  double charge = fUseHitIntegral ? hitPair.first->Integral() : hitPair.first->SummedADC();
675 
676  double predLogCharge = (fSamplingRate * (hitPair.first->PeakTime() - firstHitTime) - avePos[0]) * slope + avePos[1];
677  double deltaLogCharge = std::log(charge) - predLogCharge;
678 
679  hitPair.second.second = deltaLogCharge;
680  }
681  else hitPair.second.second = 0.; // These hits already rejected, don't double count them.
682  }
683 
684  // Sort hits by their deviation from the prediction
685  std::sort(hitPairVector.begin(),hitPairVector.end(),[](const auto& left, const auto& right){return left.second.second < right.second.second;});
686 
687  // Go through and tag those we are rejecting
688  size_t loRejectIdx = 0.01 * hitPairVector.size();
689  size_t hiRejectIdx = fOutlierRejectFrac * hitPairVector.size();
690 
691  const double outlierReject = 0.75;
692 
693  for(size_t idx = 0; idx < hitPairVector.size(); idx++)
694  {
695  if (idx < loRejectIdx || idx > hiRejectIdx) hitPairVector[idx].second.first = false;
696  if (hitPairVector[idx].second.second < -outlierReject) hitPairVector[idx].second.first = false;
697  }
698 
699  // Put back in time order
700  std::sort(hitPairVector.begin(), hitPairVector.end(), [](const auto& left, const auto& right){return left.first->PeakTime() < right.first->PeakTime();});
701 
702  return;
703 }
