120 //----------------------------------------------------------------------------
121 /// Produce method.
122 ///
123 /// Arguments:
124 ///
125 /// evt - Art event.
126 ///
127 /// This is the primary method. The goal is to produce a list of recob::Hit
128 /// objects which are a "clean" subset of all hits and which are believed to
129 /// be due to a neutrino interaction. It does this by considering input CosmicTag
130 /// objects, relating them to PFParticles/Tracks and removing the hits
131 /// associated to those objects which are believed to be Cosmic Rays.
132 ///
133 void HitSelector::produce(art::Event & evt)
134 {
135  ++fNumEvent;
136 
137  // Start by looking up the original hits
138  art::Handle< std::vector<recob::Hit> > hitHandle;
139  evt.getByLabel(fHitProducerLabel, hitHandle);
140 
141  // also get the associated wires and raw digits;
142  // we assume they have been created by the same module as the hits
143  art::FindOneP<raw::RawDigit> hitToRawDigitAssns(hitHandle, evt, fHitProducerLabel);
144  art::FindOneP<recob::Wire> hitToWireAssns( hitHandle, evt, fHitProducerLabel);
145 
146  // this object contains the hit collection
147  // and its associations to wires and raw digits:
148  recob::HitCollectionCreator hcol(evt,
149  hitToWireAssns.isValid(),
150  hitToRawDigitAssns.isValid()
151  );
152 
153  // If there are no hits then there should be no output
154  if (hitHandle.isValid())
155  {
156  HitPtrVector hitPtrVec;
157  HitPtrVector hitSnippetVec;
158 
159  art::fill_ptr_vector(hitPtrVec, hitHandle);
160 
161  int lastHitIndex = 0;
162 
163  // Loop the hits and make some plots
164  for(const auto& curHitPtr : hitPtrVec)
165  {
166  // Have we collected all of the hits on the same snippet?
167  if (!hitSnippetVec.empty() && lastHitIndex >= curHitPtr->LocalIndex())
168  {
169  // Recover plane
170  int plane = hitSnippetVec.front()->WireID().Plane;
171 
172  // Only worried about multi hit snippets
173  if (hitSnippetVec.size() > 1)
174  {
175  // Sort in order of largest to smallest pulse height
176  std::sort(hitSnippetVec.begin(),hitSnippetVec.end(),[](const auto& left, const auto& right){return left->PeakAmplitude() > right->PeakAmplitude();});
177 
178  float maxPulseHeight = hitSnippetVec.front()->PeakAmplitude();
179 
180  // Require the largest pulse in the train to be above threshold
181  if (maxPulseHeight > fMinMaxPulseHeighMulti[plane])
182  {
183  // Loop over all hits...
184  for(size_t idx = 0; idx < hitSnippetVec.size(); idx++)
185  {
186  art::Ptr<recob::Hit> hitPtr = hitSnippetVec[idx];
187 
188  float pulseHeight = hitPtr->PeakAmplitude();
189  float pulseWid = hitPtr->RMS();
190  int numDOF = hitPtr->DegreesOfFreedom();
191 
192  // Watch for case of long pulse trains (numDOF == 1) and keep all of these no matter
193  // Also always keep the largest (first) pulse
194  // Otherwise, select on minimum pulse height and width
195  if (idx == 0 || numDOF == 1 || (pulseHeight > fMinPulseHeightMulti[plane] && pulseWid > fMinPulseWidthMulti[plane]))
196  {
197  art::Ptr<recob::Wire> wire = hitToWireAssns.at(hitPtr.key());
198  art::Ptr<raw::RawDigit> rawdigits = hitToRawDigitAssns.at(hitPtr.key());
199 
200  hcol.emplace_back(*hitPtr, wire, rawdigits);
201  }
202  }
203  }
204  }
205  // Here we handle the case of single hits
206  else
207  {
208  art::Ptr<recob::Hit> hitPtr = hitSnippetVec.front();
209 
210  float pulseHeight = hitPtr->PeakAmplitude();
211  float pulseWid = hitPtr->RMS();
212 
213  if (pulseHeight > fMinPulseHeightSingle[plane] && pulseWid > fMinPulseWidthSingle[plane])
214  {
215  art::Ptr<recob::Wire> wire = hitToWireAssns.at(hitPtr.key());
216  art::Ptr<raw::RawDigit> rawdigits = hitToRawDigitAssns.at(hitPtr.key());
217 
218  hcol.emplace_back(*hitPtr, wire, rawdigits);
219  }
220  }
221 
222  hitSnippetVec.clear();
223  }
224 
225  lastHitIndex = curHitPtr->LocalIndex();
226  hitSnippetVec.push_back(curHitPtr);
227  }
228  }
229 
230  // put the hit collection and associations into the event
231  hcol.put_into(evt);
232 
233  return;
234 }
