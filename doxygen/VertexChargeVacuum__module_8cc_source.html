119 float Vert2HitDistance(const recob::Hit &hit, const recob::Vertex &vert, const geo::GeometryCore *geo, const detinfo::DetectorPropertiesData &dprop) {
120  std::array<float, 2> vert_v = VertexVector(vert, hit.WireID(), geo, dprop);
121  std::array<float, 2> hit_v = HitVector(hit, geo, dprop);
122 
123  return sqrt((vert_v[0] - hit_v[0]) * (vert_v[0] - hit_v[0]) + (vert_v[1] - hit_v[1]) * (vert_v[1] - hit_v[1]));

126 // local helper function: Get the mean direction of a set of hits away from a vertex (projected on a plane)
127 std::array<float, 2> HitDirection(const std::vector<art::Ptr<recob::Hit>> &hits, const recob::Vertex &vert,
128  const geo::GeometryCore *geo, const detinfo::DetectorPropertiesData &dprop) {
129  if (!hits.size()) return {0., 0.};
130 
131  geo::Vector_t avg(0., 0., 0.);
132 
133  std::array<float, 2> vert_v = VertexVector(vert, hits[0]->WireID(), geo, dprop);
134  geo::Point_t vert_p(vert_v[0], vert_v[1], 0.);
135 
136  for (const art::Ptr<recob::Hit> &h: hits) {
137  std::array<float, 2> hit_v = HitVector(*h, geo, dprop);
138  geo::Point_t hit_p(hit_v[0], hit_v[1], 0.);
139  avg += (hit_p - vert_p).Unit();
140  }
141 
142  avg = avg.Unit();
143 
144  return {(float)avg.X(), (float)avg.Y()};
145 }

147 float TrackDirectionParallel(const recob::Track &trk, const geo::PlaneID &plane,
148  const geo::GeometryCore *geo, const detinfo::DetectorPropertiesData &dprop) {
149  double angleToVert = geo->WireAngleToVertical(geo->View(plane), plane) - 0.5*::util::pi<>();
150  double cosgamma = std::abs(std::sin(angleToVert)*trk.StartDirection().y() + std::cos(angleToVert)*trk.StartDirection().z());
151 
152  float ret = sqrt(cosgamma * cosgamma + trk.StartDirection().x() * trk.StartDirection().x());
153 
154  return ret;

158 geo::Point_t PlaceHitAlongTrack(const recob::Track &trk, const recob::Vertex &vert, const recob::Hit &hit,
159  const geo::GeometryCore *geo, const detinfo::DetectorPropertiesData &dprop) {
160  // project the vertex onto the hit Plane
161  std::array<float, 2> v_plane = VertexVector(vert, hit.WireID(), geo, dprop);
162 
163  // also get the 2d hit location
164  std::array<float, 2> h_plane = HitVector(hit, geo, dprop);
165 
166  float t_dir_parallel_plane = TrackDirectionParallel(trk, hit.WireID(), geo, dprop);
167 
168  // Get the distance from the vertex to the hit on the plane.
169  //
170  // The track "should" be along the same direction as the vertex to the hit in the plane,
171  // so this distance is the same as the distance along the track
172  float plane_dist = (h_plane[0] - v_plane[0]) + (h_plane[1] - v_plane[1]);
173 
174  // Use this distance to place the hit along the track trajectory in 3D
175  float dist_3d = plane_dist / t_dir_parallel_plane;
176 
177  // std::cout << "Placing Hit\n";
178  // std::cout << "Hit wire: " << hit.WireID() << std::endl;
179  // std::cout << "Hit plane coord: " << h_plane[0] << " " << h_plane[1] << std::endl;
180  // std::cout << "Vtx plane coord: " << v_plane[0] << " " << v_plane[1] << std::endl;
181  // std::cout << "Trk dir: " << trk.StartDirection().x() << " " << trk.StartDirection().y() << " " << trk.StartDirection().z() << std::endl;
182  // std::cout << "Vtx pos: " << vert.position().x() << " " << vert.position().y() << " " << vert.position().z() << std::endl;
183 
184  // std::cout << "Trk parallel dir: " << t_dir_parallel_plane << std::endl;
185  // std::cout << "Hit plane dist: " << plane_dist << std::endl;
186  // std::cout << "Hit 3d dist: " << dist_3d << std::endl;
187 
188  geo::Vector_t trk_dir(trk.StartDirection().x(), trk.StartDirection().y(), trk.StartDirection().z());
189 
190  return vert.position() + trk_dir * dist_3d;

207 void sbn::VertexChargeVacuum::produce(art::Event& evt)
208 {
209  // output stuff
210  std::unique_ptr<std::vector<sbn::VertexHit>> outVHit(new std::vector<sbn::VertexHit>);
211  std::unique_ptr<art::Assns<recob::Slice, sbn::VertexHit>> assn(new art::Assns<recob::Slice, sbn::VertexHit>);
212  std::unique_ptr<art::Assns<recob::Vertex, sbn::VertexHit>> vtxAssn(new art::Assns<recob::Vertex, sbn::VertexHit>);
213  std::unique_ptr<art::Assns<recob::Hit, sbn::VertexHit>> hitAssn(new art::Assns<recob::Hit, sbn::VertexHit>);
214 
215  art::PtrMaker<sbn::VertexHit> vhitPtrMaker {evt};
216 
217  // collect services
218  const geo::GeometryCore *geo = lar::providerFrom<geo::Geometry>();
219  auto const clock_data = art::ServiceHandle<detinfo::DetectorClocksService const>()->DataFor(evt);
220  auto const dprop =
221  art::ServiceHandle<detinfo::DetectorPropertiesService const>()->DataFor(evt, clock_data);
222  auto const* sce = lar::providerFrom<spacecharge::SpaceChargeService>();
223  // If we are not correcting for SCE, then blank out the service
224  if (!fCorrectSCE) sce = nullptr;
225 
226  // get the PFParticle's and the associated data
227  art::Handle<std::vector<recob::PFParticle>> pfparticle_handle;
228  evt.getByLabel(fPFParticleLabel, pfparticle_handle);
229 
230  std::vector<art::Ptr<recob::PFParticle>> pfparticles;
231  art::fill_ptr_vector(pfparticles, pfparticle_handle);
232 
233  art::FindManyP<recob::Vertex> pfparticleVertices(pfparticles, evt, fPFParticleLabel);
234  art::FindManyP<recob::Cluster> pfparticleClusters(pfparticles, evt, fPFParticleLabel);
235  art::FindManyP<recob::Slice> pfparticleSlices(pfparticles, evt, fPFParticleLabel);
236  art::FindManyP<recob::Track> pfparticleTracks(pfparticles, evt, fTrackLabel);
237 
238  // organize the PFPlist into a map
239  std::map<unsigned, art::Ptr<recob::PFParticle>> id_to_pfp;
240  for (unsigned i = 0; i < pfparticles.size(); i++) {
241  id_to_pfp[pfparticles[i]->Self()] = pfparticles[i];
242  }
243 
244  // get all the hits
245  // art::Handle<std::vector<recob::Hit>> hit_handle;
246  // evt.getByLabel(fHitLabel, hit_handle);
247 
248  // std::vector<art::Ptr<recob::Hit>> hits;
249  // art::fill_ptr_vector(hits, hit_handle);
250 
251  // map to space-points
252  // art::FindManyP<recob::SpacePoint> hitSPs(hits, evt, fPFParticleLabel);
253 
254  // iterate over the "primary" PFParticles
255  for (unsigned i_pfp = 0; i_pfp < pfparticles.size(); i_pfp++) {
256  const recob::PFParticle &pfp = *pfparticles[i_pfp];
257  if (!pfp.IsPrimary()) continue;
258  // Ignore PFP's with no vertex
259  if (!pfparticleVertices.at(i_pfp).size()) continue;
260 
261  // If configured, require this to be a PFP neutrino
262  unsigned pfpPDGC = std::abs(pfp.PdgCode());
263  if(fSelectNeutrino &&
264  (pfpPDGC != 12) && (pfpPDGC != 14) && (pfpPDGC != 16) ) continue;
265 
266  // we found a primary PFP! Get its vertex.
267  const art::Ptr<recob::Vertex> &vtx_ptr = pfparticleVertices.at(i_pfp).at(0);
268  const recob::Vertex &vert = *vtx_ptr;
269 
270  // The presence of space charge creates two different positions: the position as
271  // "seen" by the wires (post-space charge) and the true position (pre-space charge)
272  //
273  // In some cases we want the wire-position and in other we want the true-position.
274  // To be explicit, create two different vertexes for these two different cases
275  recob::Vertex vert_absolute(PositionAbsolute(vert.position(), geo, sce),
276  vert.covariance(),
277  vert.chi2(),
278  vert.ndof(),
279  vert.ID());
280 
281  recob::Vertex vert_atwires(PositionAtWires(vert.position(), geo, sce),
282  vert.covariance(),
283  vert.chi2(),
284  vert.ndof(),
285  vert.ID());
286 
287  // also get all the daughter PFParticles
288  const std::vector<size_t> &daughters = pfp.Daughters();
289  std::vector<art::Ptr<recob::PFParticle>> daughterPFPs;
290  for (size_t d: daughters) {
291  daughterPFPs.push_back(id_to_pfp.at(d));
292  }
293 
294  // look up the hits of each daughter
295  std::array<std::vector<std::vector<art::Ptr<recob::Hit>>>, 3> daughterPlaneHits;
296  for (const art::Ptr<recob::PFParticle> &d: daughterPFPs) {
297  for (unsigned i_plane = 0; i_plane < 3; i_plane++) {
298  daughterPlaneHits[i_plane].emplace_back();
299  }
300  const std::vector<art::Ptr<recob::Cluster>> &d_clusters = pfparticleClusters.at(d.key());
301  art::FindManyP<recob::Hit> d_cluster_hits(d_clusters, evt, fPFParticleLabel);
302  for (unsigned i = 0; i < d_clusters.size(); i++) {
303  const std::vector<art::Ptr<recob::Hit>> &this_cluster_hits = d_cluster_hits.at(i);
304  daughterPlaneHits[d_clusters[i]->Plane().Plane].back().insert(
305  daughterPlaneHits[d_clusters[i]->Plane().Plane].back().end(),
306  this_cluster_hits.begin(), this_cluster_hits.end());
307  }
308  }
309 
310  // Get the Slice associated with the primary PFP
311  art::Ptr<recob::Slice> thisSlc = pfparticleSlices.at(i_pfp).at(0);
312  // look up the hits
313  art::FindManyP<recob::Hit> thisSlcHits({thisSlc}, evt, fPFParticleLabel);
314  const std::vector<art::Ptr<recob::Hit>> &hits = thisSlcHits.at(0);
315 
316  // work on each plane
317  for (unsigned i_plane = 0; i_plane < 3; i_plane++) {
318  // vacuum up all the hits within the radius
319  std::vector<art::Ptr<recob::Hit>> nearbyHits;
320  for (unsigned i_hit = 0; i_hit < hits.size(); i_hit++) {
321  const recob::Hit &hit = *hits[i_hit];
322  if (hit.WireID().Plane == i_plane) {
323  if (Vert2HitDistance(hit, vert_atwires, geo, dprop) < fHitVacuumRadius) {
324  nearbyHits.push_back(hits[i_hit]);
325  }
326  }
327  }
328 
329  // and find the hit SP's
330  art::FindManyP<recob::SpacePoint> hitSPs(nearbyHits, evt, fPFParticleLabel);
331 
332  // Compute all needed information for each hit
333  for (unsigned i_hit = 0; i_hit < nearbyHits.size(); i_hit++) {
334  const recob::Hit &hit = *nearbyHits[i_hit];
335 
336  sbn::VertexHit vhit;
337  vhit.wire = hit.WireID();
338  vhit.charge = fCaloAlg.ElectronsFromADCArea(hit.Integral(), hit.WireID().Plane) * fCaloAlg.LifetimeCorrection(clock_data, dprop, hit.PeakTime(), 0.);
339  vhit.proj_dist_to_vertex = Vert2HitDistance(hit, vert_atwires, geo, dprop);
340  vhit.vtxw = geo->WireCoordinate(vert_atwires.position(), hit.WireID());
341  vhit.vtxx = vert_atwires.position().x();
342  vhit.vtxXYZ = vert_absolute.position();
343 
344  // lookup the spacepoint location
345  const std::vector<art::Ptr<recob::SpacePoint>> &hit_sp = hitSPs.at(i_hit);
346 
347  bool has_xyz = false;
348  int spID = -1;
349  geo::Point_t spXYZ;
350  // Space-Point!
351  if (hit_sp.size()) {
352  const recob::SpacePoint sp = *hit_sp.at(0);
353  spID = sp.ID();
354  spXYZ = PositionAbsolute(sp.position(), geo, sce);
355  has_xyz = true;
356  }
357  // No Space-Point. If configured, see if we can look up a point along the assigned track
358  else if (fUseTrackSPRecovery) {
359  unsigned plane = hit.WireID().Plane;
360  art::Ptr<recob::PFParticle> matchingPFP;
361  for (unsigned i_pfp_chk = 0; i_pfp_chk < daughterPlaneHits[plane].size(); i_pfp_chk++) {
362  for (unsigned i_hit_chk = 0; i_hit_chk < daughterPlaneHits[plane][i_pfp_chk].size(); i_hit_chk++) {
363  if (daughterPlaneHits[plane][i_pfp_chk][i_hit_chk] == nearbyHits[i_hit]) {
364  matchingPFP = daughterPFPs[i_pfp_chk];
365  break;
366  }
367  }
368  }
369  if (matchingPFP) {
370  const std::vector<art::Ptr<recob::Track>> &pfptrack = pfparticleTracks.at(matchingPFP.key());
371  if (pfptrack.size()) {
372  const recob::Track &thisTrack = *pfptrack.at(0);
373  spXYZ = PositionAbsolute(PlaceHitAlongTrack(thisTrack, vert_atwires, hit, geo, dprop), geo, sce);
374  has_xyz = true;
375  }
376  }
377  }
378 
379  if (has_xyz) {
380  vhit.spID = spID;
381  vhit.spXYZ = spXYZ;
382 
383  geo::Vector_t dir = (spXYZ - vert_absolute.position()).Unit();
384 
385  // Compute the pitch. Since we have used the corrected vertex and Space-Point position, the
386  // pt and dir here are space-charge corrected regardless of the input configuration
387  vhit.pitch = sbn::GetPitch(geo, sce, spXYZ, dir, hit.View(), hit.WireID(), fCorrectSCE, true);
388 
389  vhit.dqdx = fCaloAlg.ElectronsFromADCArea((hit.Integral() / vhit.pitch), hit.WireID().Plane) * fCaloAlg.LifetimeCorrection(clock_data, dprop, hit.PeakTime(), 0.);
390 
391  // Same here -- input position is already corrected
392  float EField = sbn::GetEfield(dprop, sce, spXYZ, hit.WireID(), false);
393 
394  vhit.dedx = fCaloAlg.dEdx_AREA(clock_data, dprop, vhit.dqdx, hit.PeakTime(), hit.WireID().Plane, 0., EField);
395  }
396  else {
397  vhit.spID = -1;
398  vhit.pitch = -1;
399  vhit.dqdx = -1;
400  vhit.dedx = -1;
401  }
402 
403  // Save!
404  outVHit->push_back(vhit);
405  art::Ptr<sbn::VertexHit> thisVHitPtr = vhitPtrMaker(outVHit->size()-1);
406  assn->addSingle(thisSlc, thisVHitPtr);
407  vtxAssn->addSingle(vtx_ptr, thisVHitPtr);
408  hitAssn->addSingle(nearbyHits[i_hit], thisVHitPtr);
409 
410  } // end iterate over hits
411  } // end iterate over planes
412  } // end iterate over pfparticle's
413 
414  evt.put(std::move(outVHit));
415  evt.put(std::move(vtxAssn));
416  evt.put(std::move(assn));
417  evt.put(std::move(hitAssn));
418 
419 }
