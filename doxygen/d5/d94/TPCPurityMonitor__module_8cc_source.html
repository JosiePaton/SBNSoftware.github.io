373 void TPCPurityMonitor::produce(art::Event& event)
374 {
375  //setup output vector#include "lardataobj/RecoBase/TrackHitMeta.h"
376  std::unique_ptr< std::vector<anab::TPCPurityInfo> > outputPtrVector(new std::vector<anab::TPCPurityInfo>());
377 
378  fNumEvents++;
379 
380  // Recover the detector properties.
381  auto const clockData = art::ServiceHandle<detinfo::DetectorClocksService const>()->DataFor(event);
382 
383  fSamplingRate = 1.e-3 * sampling_rate(clockData); // Note sampling rate is in ns, convert to us
384 
385  for(const auto& trackLabel : fTrackLabelVec)
386  {
387  // Make a pass through all hits to make contrasting plots
388  art::Handle< std::vector<recob::Track>> trackHandle;
389  event.getByLabel(trackLabel, trackHandle);
390 
391  if (!trackHandle.isValid()) continue;
392 
393  // I don't know another way to do this... but we need to build a map from hit to spacepoint
394  // since we don't seem to have a way to go that direction with what we have for kalman fit
395  // tracks.
396  using HitToSpacePointMap = std::unordered_map<const recob::Hit*,const recob::SpacePoint*>;
397 
398  HitToSpacePointMap hitToSpacePointMap;
399 
400  using PointCloud = std::vector<geo::Point_t>;
401 
402  PointCloud pointCloud;
403 
404  // Recover the collection of associations between tracks and hits and hits and spacepoints
405  art::FindManyP<recob::Hit,recob::TrackHitMeta> trackHitAssns(trackHandle, event, trackLabel);
406 
407  // Loop over tracks and recover hits
408  for(size_t trackIdx = 0; trackIdx < trackHandle->size(); trackIdx++)
409  {
410  art::Ptr<recob::Track> track(trackHandle,trackIdx);
411 
412  const std::vector<art::Ptr<recob::Hit>>& trackHitsVec(trackHitAssns.at(track.key()));
413  const std::vector<const recob::TrackHitMeta*> metaHitsVec = trackHitAssns.data(track.key());
414 
415  // Focus on selected hits:
416  // 1) Pick out hits on a single plane given by fhicl parameter
417  // 2) multiplicity == 1 which should give us clean gaussian shaped pulses
418  using TPCToHitMetaPairVecMap = std::unordered_map<unsigned int,HitMetaPairVec>;
419 
420  TPCToHitMetaPairVecMap selectedHitMetaVecMap;
421 
422  for(size_t idx=0; idx<trackHitsVec.size(); idx++)
423  {
424  art::Ptr<recob::Hit> hit(trackHitsVec.at(idx));
425 
426  if (hit->WireID().Plane == fSelectedPlane && hit->Multiplicity() == 1) selectedHitMetaVecMap[hit->WireID().TPC].emplace_back(hit,metaHitsVec.at(idx));
427  }
428 
429  if (selectedHitMetaVecMap.empty()) continue;
430 
431  // Currently we need to limit the analysis to a single TPC and we have tracks which may have been stitched across the cathode...
432  // For now, we search and find the TPC with the most hits
433  TPCToHitMetaPairVecMap::iterator bestMapItr = selectedHitMetaVecMap.begin();
434 
435  for(TPCToHitMetaPairVecMap::iterator mapItr = selectedHitMetaVecMap.begin(); mapItr != selectedHitMetaVecMap.end(); mapItr++)
436  {
437  if (mapItr->second.size() > bestMapItr->second.size()) bestMapItr = mapItr;
438  }
439 
440  HitMetaPairVec& selectedHitMetaVec = bestMapItr->second;
441 
442  // Need a minimum number of hits
443  if (selectedHitMetaVec.size() < fMinNumHits) continue;
444 
445  // Sort hits by increasing time
446  std::sort(selectedHitMetaVec.begin(),selectedHitMetaVec.end(),[](const auto& left, const auto& right){return left.first->PeakTime() < right.first->PeakTime();});
447 
448  // Require track to have a minimum range in ticks
449  if (selectedHitMetaVec.back().first->PeakTime() - selectedHitMetaVec.front().first->PeakTime() < fMinTickRange) continue;
450 
451  // At this point we should have a vector of art::Ptrs to hits on the selected plane
452  // So we should be able to now transition to computing the attenuation
453  // Start by forming a vector of pairs of the time (in ticks) and the ln of charge derated by an assumed lifetime
454  HitStatusChargePairVec hitStatusChargePairVec;
455  HitPointDirTupleMap hitPointDirTupleMap;
456 
457  float firstHitTime(selectedHitMetaVec.front().first->PeakTime());
458  double maxDeltaX(1.5); // Assume a "long hit" would be no more than 1.5 cm in length
459  double wirePitch(0.3);
460 
461  for(const auto& hitMetaPair: selectedHitMetaVec)
462  {
463  unsigned int trkHitIndex = hitMetaPair.second->Index();
464  double deltaX = 0.3; // Set this to 3 mm just in case no corresponding point
465  double cosTheta = -100.;
466 
467  if (trkHitIndex != std::numeric_limits<unsigned int>::max() && track->HasValidPoint(trkHitIndex))
468  {
469  geo::Point_t hitPos = track->LocationAtPoint(trkHitIndex);
470  geo::Vector_t hitDir = track->DirectionAtPoint(trkHitIndex);
471  const geo::WireGeo& wireGeo = fGeometry->Wire(hitMetaPair.first->WireID());
472  geo::Vector_t wireDir(wireGeo.Direction()[0],wireGeo.Direction()[1],wireGeo.Direction()[2]);
473 
474  pointCloud.emplace_back(hitPos);
475 
476  cosTheta = std::abs(hitDir.Dot(wireDir));
477 
478  if (cosTheta < 1.)
479  {
480  deltaX = std::min(wirePitch / (1. - cosTheta), maxDeltaX);
481  }
482  else deltaX = maxDeltaX;
483 
484  double charge = fUseHitIntegral ? hitMetaPair.first->Integral() : hitMetaPair.first->SummedADC();
485 
486  hitStatusChargePairVec.emplace_back(hitMetaPair,StatusChargePair(true,charge/deltaX));
487  hitPointDirTupleMap[hitMetaPair.first.get()] = PointDirTuple(hitPos,hitDir,wireDir);
488  }
489  }
490 
491  size_t numOrig = hitStatusChargePairVec.size();
492  size_t lowCutIdx = fMinRejectFraction * numOrig;
493  size_t hiCutIdx = fMaxRejectFraction * numOrig;
494 
495  // Will require a minimum number of hits left over to proceed
496  if (lowCutIdx + 10 >= hiCutIdx)
497  {
498  std::cout << "*****>>>> lowCutIdx >= hiCutIdx: " << lowCutIdx << ", " << hiCutIdx << std::endl;
499  continue;

754 void TPCPurityMonitor::GetPrincipalComponents2D(const HitStatusChargePairVec& hitPairVector, PrincipalComponents2D& pca) const
755 {
756  // Run through the HitPairList and get the mean position of all the hits
757  Eigen::Vector2d meanPos(Eigen::Vector2d::Zero());
758  double meanWeightSum(0.);
759  int numPairsInt(0);
760 
761  float startTime = 0.; //hitPairVector.front().first->PeakTime();
762 
763  for (const auto& hitPair : hitPairVector)
764  {
765  if (!hitPair.second.first) continue;
766 
767  const recob::Hit* hit = hitPair.first.first.get();
768 
769  // Weight the hit by the peak time difference significance
770  double weight = fWeightByChiSq ? 1./hit->GoodnessOfFit() : 1.;
771 
772  meanPos(0) += fSamplingRate * (hit->PeakTime() - startTime) * weight;
773  meanPos(1) += std::log(hitPair.second.second) * weight;
774  numPairsInt++;
775 
776  meanWeightSum += weight;
777  }
778 
779  meanPos /= meanWeightSum;
780 
781  // Define elements of our covariance matrix
782  double xi2(0.);
783  double xiyi(0.);
784  double yi2(0.0);
785  double weightSum(0.);
786 
787  // Back through the hits to build the matrix
788  for (const auto& hitPair : hitPairVector)
789  {
790  if (!hitPair.second.first) continue;
791 
792  const recob::Hit* hit = hitPair.first.first.get();
793 
794  double weight = fWeightByChiSq ? 1./hit->GoodnessOfFit() : 1.;
795 
796  double x = (fSamplingRate * (hit->PeakTime() - startTime) - meanPos(0)) * weight;
797  double y = (std::log(hitPair.second.second) - meanPos(1)) * weight;
798 
799  weightSum += weight * weight;
800 
801  xi2 += x * x;
802  xiyi += x * y;
803  yi2 += y * y;
804  }
805 
806  // Using Eigen package
807  Eigen::Matrix2d sig;
808 
809  sig << xi2, xiyi, xiyi, yi2;
810 
811  sig *= 1. / weightSum;
812 
813  Eigen::SelfAdjointEigenSolver<Eigen::Matrix2d> eigenMat(sig);
814 
815  if (eigenMat.info() == Eigen::ComputationInfo::Success)
816  {
817  // Now copy outputPrincipalCo
818  // The returned eigen values and vectors will be returned in an xyz system where x is the smallest spread,
819  // y is the next smallest and z is the largest. Adopt that convention going forward
820  PrincipalComponents2D::EigenValues eigenVals = eigenMat.eigenvalues();
821  PrincipalComponents2D::EigenVectors eigenVecs = eigenMat.eigenvectors().transpose();
822 
823  // Store away
824  // NOTE: the major axis will be the second entry, the minor axis will be the first
825  pca = PrincipalComponents2D(true, numPairsInt, eigenVals, eigenVecs, meanPos);
826  }
827  else
828  {
829  mf::LogDebug("Cluster3D") << "PCA decompose failure, numPairs = " << numPairsInt << std::endl;
830  pca = PrincipalComponents2D();
831  }
832 
833  return;
834 }

925 {

940  double deltaLogCharge = std::log(hitPair.second.second) - predLogCharge;
941

958  if (hitPairDeltaLogChargePairVec[idx].second < -outlierReject) hitPairDeltaLogChargePairVec[idx].first->second.first = false;
