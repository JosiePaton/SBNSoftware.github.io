58  /*
59  * Currently util::Quantity objects have serious issues with FHiCL validation,
60  * so we are not reading them directly as such. This pulls in a number of
61  * workarounds for:
62  * * every `Quantity` parameter: its FHiCL parameter must be declared as
63  * the base type `Quantity::value_t`
64  * * every optional parameter must be then read indirectly since a reference
65  * to the exact type of the parameter is required in such reading
66  * * for sequences, direct vector assignment
67  * (`vector<Quantity> = vector<Quantity::value_t>`) won't work, and since
68  * `Sequence::operator()` returns a temporary, this needs to be wraped too
69  */
70  template <typename SeqValueType>
71  struct FHiCLsequenceWrapper {
72  SeqValueType seqValue;
73 
74  FHiCLsequenceWrapper(SeqValueType seqValue): seqValue(seqValue) {}
75 
76  template <typename T>
77  operator std::vector<T>() const
78  { return { seqValue.cbegin(), seqValue.cend() }; }
79 
80  }; // FHiCLsequenceWrapper

91 icarus::trigger::SingleChannelOpticalTriggerGate&
92 icarus::trigger::TriggerGateBuilder::TriggerGates::gateFor
93  (raw::OpDetWaveform const& waveform)
94 {
95  // keeping the gates sorted by channel
96  auto iGate = std::lower_bound
97  (fGates.begin(), fGates.end(), waveform, ::ChannelComparison<>());
98  if (iGate != fGates.end()) { // found, it's there already
99  MF_LOG_TRACE(details::TriggerGateDebugLog)
100  << "Appending waveform to trigger gate (thr=" << threshold()
101  << ") of channel " << waveform.ChannelNumber();
102  iGate->add(waveform);
103  return *iGate;
104  }
105  // add the new gate in channel order
106  MF_LOG_TRACE(details::TriggerGateDebugLog)
107  << "Creating a new trigger gate (thr=" << threshold()
108  << ") for channel " << waveform.ChannelNumber();
109  iGate = fGates.emplace(iGate, waveform);
110  return *iGate;
111 } // icarus::trigger::TriggerGateBuilder::TriggerGates::gateFor()
