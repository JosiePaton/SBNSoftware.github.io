283  bool GenFilter::UsesCRTAuxDet(const simb::MCParticle &particle, geo::AuxDetGeo const& crt){
284  //We need to prepare the particle's position and direction and construct a bounding box from the CRT for the ray-box collision algorithm
285  //Start with the particle
286  double particle_position_array[3], particle_direction_array[3], particle_local_position_array[3], particle_local_direction_array[3];
287  particle_position_array[0] = particle.Position(0).X();
288  particle_position_array[1] = particle.Position(0).Y();
289  particle_position_array[2] = particle.Position(0).Z();
290  particle_direction_array[0] = particle.Momentum(0).X()/particle.Momentum(0).Vect().Mag();
291  particle_direction_array[1] = particle.Momentum(0).Y()/particle.Momentum(0).Vect().Mag();
292  particle_direction_array[2] = particle.Momentum(0).Z()/particle.Momentum(0).Vect().Mag();
293  crt.WorldToLocal(particle_position_array,particle_local_position_array);
294  crt.WorldToLocalVect(particle_direction_array,particle_local_direction_array);
295  TVector3 particle_local_position(particle_local_position_array[0],particle_local_position_array[1],particle_local_position_array[2]);
296  TVector3 particle_local_direction(particle_local_direction_array[0],particle_local_direction_array[1],particle_local_direction_array[2]);
297  double norm[3], lnorm[3];
298  double center[3];
299  crt.GetNormalVector(norm);
300  crt.WorldToLocalVect(norm,lnorm);
301  crt.GetCenter(center);
302 
303  //Now make the bounding box min and max extent from the CRT
304  //In local coordinates, the normal of the CRT is parallel to the z-axis, the length is parallel to z, width parallel to x and height parallel to y
305  //A gotchya: AuxDets pass half widths and half heights but FULL lengths. So, divide length by 2
306  TVector3 crt_min_extent(-1.*crt.HalfWidth1(),-1.*crt.HalfHeight(),-1.*crt.Length()/2.);
307  //Scale the dimensions if the user wants them scaling
308  crt_min_extent*=fCRTDimensionScaling;
309  //Now make the max extent
310  //Use symmetry to reduce change of a typo
311  TVector3 crt_max_extent = crt_min_extent*-1.;
312 
313  //Now run the algorithm
314  return RayIntersectsBox(particle_local_position,particle_local_direction,crt_min_extent,crt_max_extent);
315  }
