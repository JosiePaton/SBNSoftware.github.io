140  static_assert(value(EnumType::NBits) >= 0);

144  constexpr std::size_t MaxBits = sizeof(mask) * 8U;
145  constexpr std::size_t NSupportedBits = value(EnumType::NBits);
146 
147  std::vector<std::string> names;
148  for (std::size_t bit = 0U; bit < MaxBits; ++bit) {
149  auto const typedBit = static_cast<EnumType>(bit);
150  if (!hasBitSet(mask, typedBit)) continue;
151  names.push_back((bit > NSupportedBits)
152  ? "<unsupported ["s + std::to_string(bit) + "]>"s: name(typedBit)
153  );
154  } // for
