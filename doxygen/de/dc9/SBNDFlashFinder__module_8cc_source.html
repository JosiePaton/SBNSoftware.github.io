139  void SBNDFlashFinder::GetFlashLocation(std::vector<double> pePerOpChannel,
140  double& Ycenter,
141  double& Zcenter,
142  double& Ywidth,
143  double& Zwidth)
144  {
145 
146  // Reset variables
147  Ycenter = Zcenter = 0.;
148  Ywidth = Zwidth = -999.;
149  double totalPE = 0.;
150  double sumy = 0., sumz = 0., sumy2 = 0., sumz2 = 0.;
151  for (unsigned int opch = 0; opch < pePerOpChannel.size(); opch++) {
152  // Get physical detector location for this opChannel
153  double PMTxyz[3];
154  ::lightana::OpDetCenterFromOpChannel(opch, PMTxyz);
155  // Add up the position, weighting with PEs
156  sumy += pePerOpChannel[opch]*PMTxyz[1];
157  sumy2 += pePerOpChannel[opch]*PMTxyz[1]*PMTxyz[1];
158  sumz += pePerOpChannel[opch]*PMTxyz[2];
159  sumz2 += pePerOpChannel[opch]*PMTxyz[2]*PMTxyz[2];
160  totalPE += pePerOpChannel[opch];
161  }
162 
163  Ycenter = sumy/totalPE;
164  Zcenter = sumz/totalPE;
165 
166  // This is just sqrt(<x^2> - <x>^2)
167  if ( (sumy2*totalPE - sumy*sumy) > 0. )
168  Ywidth = std::sqrt(sumy2*totalPE - sumy*sumy)/totalPE;
169 
170  if ( (sumz2*totalPE - sumz*sumz) > 0. )
171  Zwidth = std::sqrt(sumz2*totalPE - sumz*sumz)/totalPE;
172  }
