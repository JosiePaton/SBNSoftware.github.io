535  for (unsigned j = 0; j < particle.NumberTrajectoryPoints(); j++) {
536  for (unsigned i = 0; i < active_volumes.size(); i++) {
537  if (active_volumes.at(i).ContainsPosition(particle.Position(j).Vect())) {
538  entry_point = j;
539  cryostat_index = i;
540  break;
541  }
542  }
543  if (entry_point != -1) break;
544  }
545 
546  // get the wall
547  if (entry_point > 0) {
548  trueparticle.wallin = (int)caf::GetWallCross(active_volumes.at(cryostat_index), particle.Position(entry_point).Vect(), particle.Position(entry_point-1).Vect());
549  }
550 
551  int exit_point = -1;
552 
553  // now setup the cryostat the particle is in
554  std::vector<geo::BoxBoundedGeo> volumes;
555  if (entry_point >= 0) {
556  volumes = tpc_volumes.at(cryostat_index);
557  for (unsigned i = 0; i < volumes.size(); i++) {
558  if (volumes[i].ContainsPosition(particle.Position(entry_point).Vect())) {
559  tpc_index = i;
560  trueparticle.cont_tpc = entry_point == 0;
561  break;
562  }
563  }
564  trueparticle.contained = entry_point == 0;
565  }
566  // if we couldn't find the initial point, set not contained
567  else {
568  trueparticle.contained = false;
569  }
570 
571  if (tpc_index < 0) {
572  trueparticle.cont_tpc = false;
573  }
574 
575  // Get the length and determine if any point leaves the active volume
576  // Use every trajectory point if possible
577  if (entry_point >= 0) {
578  // particle trajectory
579  const simb::MCTrajectory &trajectory = particle.Trajectory();
580  TVector3 pos = trajectory.Position(entry_point).Vect();
581  for (unsigned i = entry_point+1; i < particle.NumberTrajectoryPoints(); i++) {
582  TVector3 this_point = trajectory.Position(i).Vect();
583  // get the exit point
584  // update if particle is contained
585  // check if particle has crossed TPC
586  if (!trueparticle.crosses_tpc) {
587  for (unsigned j = 0; j < volumes.size(); j++) {
588  if (volumes[j].ContainsPosition(this_point) && tpc_index >= 0 && j != ((unsigned)tpc_index)) {
589  trueparticle.crosses_tpc = true;
590  break;
591  }
592  }
593  }
594  // check if particle has left tpc
595  if (trueparticle.cont_tpc) {
596  trueparticle.cont_tpc = volumes[tpc_index].ContainsPosition(this_point);
597  }
598 
599  if (trueparticle.contained) {
600  trueparticle.contained = active_volumes.at(cryostat_index).ContainsPosition(this_point);
601  }
602 
603  trueparticle.length += (this_point - pos).Mag();
604 
605  if (!active_volumes.at(cryostat_index).ContainsPosition(this_point) && active_volumes.at(cryostat_index).ContainsPosition(pos)) {
606  exit_point = i-1;
607  }
608 
609  pos = trajectory.Position(i).Vect();
610  }
611  }
612  if (exit_point < 0 && entry_point >= 0) {
613  exit_point = particle.NumberTrajectoryPoints() - 1;
614  }
615  if (exit_point >= 0 && ((unsigned)exit_point) < particle.NumberTrajectoryPoints() - 1) {
616  trueparticle.wallout = (int)caf::GetWallCross(active_volumes.at(cryostat_index), particle.Position(exit_point).Vect(), particle.Position(exit_point+1).Vect());
617  }
618 
619  // other truth information
620  trueparticle.pdg = particle.PdgCode();
621 
622  trueparticle.gen = ConvertTVector(particle.NumberTrajectoryPoints() ? particle.Position().Vect() : TVector3(-9999, -9999, -9999));
623  trueparticle.genT = particle.NumberTrajectoryPoints() ? particle.Position().T() / 1000. /* ns -> us*/: -9999;
624  trueparticle.genp = ConvertTVector(particle.NumberTrajectoryPoints() ? particle.Momentum().Vect(): TVector3(-9999, -9999, -9999));
625  trueparticle.genE = particle.NumberTrajectoryPoints() ? particle.Momentum().E(): -9999;
626 
627  trueparticle.start = ConvertTVector((entry_point >= 0) ? particle.Position(entry_point).Vect(): TVector3(-9999, -9999, -9999));
628  trueparticle.startT = (entry_point >= 0) ? particle.Position(entry_point).T() / 1000. /* ns-> us*/: -9999;
629  trueparticle.end = ConvertTVector((exit_point >= 0) ? particle.Position(exit_point).Vect(): TVector3(-9999, -9999, -9999));
630  trueparticle.endT = (exit_point >= 0) ? particle.Position(exit_point).T() / 1000. /* ns -> us */ : -9999;
631 
632  trueparticle.startp = ConvertTVector((entry_point >= 0) ? particle.Momentum(entry_point).Vect() : TVector3(-9999, -9999, -9999));
633  trueparticle.startE = (entry_point >= 0) ? particle.Momentum(entry_point).E() : -9999.;
634  trueparticle.endp = ConvertTVector((exit_point >= 0) ? particle.Momentum(exit_point).Vect() : TVector3(-9999, -9999, -9999));
635  trueparticle.endE = (exit_point >= 0) ? particle.Momentum(exit_point).E() : -9999.;
636 
637  trueparticle.start_process = (int)caf::GetG4ProcessID(particle.Process());
638  trueparticle.end_process = (int)caf::GetG4ProcessID(particle.EndProcess());
639 
640  trueparticle.G4ID = particle.TrackId();
641  trueparticle.parent = particle.Mother();
642 
643  // Organize deposition info into per-wire true "Hits" -- key is the Channel Number
644  std::map<unsigned, sbn::TrueHit> truehits;
645 
646  for (auto const &ide_pair: particle_ides) {
647  const geo::WireID &w = ide_pair.first;
648  unsigned c = geo->PlaneWireToChannel(w);
649  const sim::IDE *ide = ide_pair.second;
650 
651  // Set stuff
652  truehits[c].cryo = w.Cryostat;
653  truehits[c].tpc = w.TPC;
654  truehits[c].plane = w.Plane;
655  truehits[c].wire = w.Wire;
656  truehits[c].channel = c;
657 
658  // Average stuff using charge-weighting
659  float old_elec = truehits[c].nelec;
660  float new_elec = old_elec + ide->numElectrons;
661  truehits[c].p.x = (truehits[c].p.x*old_elec + ide->x*ide->numElectrons) / new_elec;
662  truehits[c].p.y = (truehits[c].p.y*old_elec + ide->y*ide->numElectrons) / new_elec;
663  truehits[c].p.z = (truehits[c].p.z*old_elec + ide->z*ide->numElectrons) / new_elec;
664 
665  // Also get the position with space charge un-done
666  geo::Point_t ide_p(ide->x, ide->y, ide->z);
667  geo::Point_t ide_p_scecorr = WireToTrajectoryPosition(ide_p, w);
668 
669  truehits[c].p_scecorr.x = (truehits[c].p_scecorr.x*old_elec + ide_p_scecorr.x()*ide->numElectrons) / new_elec;
670  truehits[c].p_scecorr.y = (truehits[c].p_scecorr.y*old_elec + ide_p_scecorr.y()*ide->numElectrons) / new_elec;
671  truehits[c].p_scecorr.z = (truehits[c].p_scecorr.z*old_elec + ide_p_scecorr.z()*ide->numElectrons) / new_elec;
672 
673  // Sum stuff
674  truehits[c].nelec += ide->numElectrons;
675  truehits[c].e += ide->energy;
676  truehits[c].ndep += 1;
677  }
678 
679  // Compute widths
680  for (auto const &ide_pair: particle_ides) {
681  const geo::WireID &w = ide_pair.first;
682  unsigned c = geo->PlaneWireToChannel(w);
683  const sim::IDE *ide = ide_pair.second;
684 
685  geo::Point_t ide_p(ide->x, ide->y, ide->z);
686  geo::Point_t ide_p_scecorr = WireToTrajectoryPosition(ide_p, w);
687 
688  // Average stuff using charge-weighting
689  float this_elec = ide->numElectrons;
690 
691  truehits[c].p_width.x += (ide_p.x() - truehits[c].p.x) * (ide_p.x() - truehits[c].p.x) * this_elec / truehits[c].nelec;
692  truehits[c].p_width.y += (ide_p.y() - truehits[c].p.y) * (ide_p.y() - truehits[c].p.y) * this_elec / truehits[c].nelec;
693  truehits[c].p_width.z += (ide_p.z() - truehits[c].p.z) * (ide_p.z() - truehits[c].p.z) * this_elec / truehits[c].nelec;
694 
695  truehits[c].p_scecorr_width.x += (ide_p_scecorr.x() - truehits[c].p_scecorr.x) * (ide_p_scecorr.x() - truehits[c].p_scecorr.x) * this_elec / truehits[c].nelec;
696  truehits[c].p_scecorr_width.y += (ide_p_scecorr.y() - truehits[c].p_scecorr.y) * (ide_p_scecorr.y() - truehits[c].p_scecorr.y) * this_elec / truehits[c].nelec;
697  truehits[c].p_scecorr_width.z += (ide_p_scecorr.z() - truehits[c].p_scecorr.z) * (ide_p_scecorr.z() - truehits[c].p_scecorr.z) * this_elec / truehits[c].nelec;
698  }
699 
700  // Convert to vector
701  std::vector<sbn::TrueHit> truehits_v;
702  for (auto const &p: truehits) {
703  truehits_v.push_back(p.second);
704  }
705 
706  // Compute the time of each hit
707  for (sbn::TrueHit &h: truehits_v) {
708  h.time = dprop.ConvertXToTicks(h.p.x, h.plane, h.tpc, h.cryo);
709 
710  double xdrift = abs(h.p.x - geo->TPC(h.tpc, h.cryo).PlaneLocation(0)[0]);
711  h.tdrift = xdrift / dprop.DriftVelocity();
712  }
713 
714  // Compute the pitch of each hit and order it in the trajectory
715  for (sbn::TrueHit &h: truehits_v) {
716  // Use the SCE-undone hit since this matches to the Trajectory
717  TVector3 h_p(h.p_scecorr.x, h.p_scecorr.y, h.p_scecorr.z);
718 
719  TVector3 direction;
720  float closest_dist = -1.;
721  int traj_index = -1;
722  for (unsigned i_traj = 0; i_traj < particle.NumberTrajectoryPoints(); i_traj++) {
723  if (closest_dist < 0. || (particle.Position(i_traj).Vect() - h_p).Mag() < closest_dist) {
724  direction = particle.Momentum(i_traj).Vect().Unit();
725  closest_dist = (particle.Position(i_traj).Vect() - h_p).Mag();
726  traj_index = i_traj;
727  }
728  }
729 
730  // If we got a direction, get the pitch
731  if (closest_dist >= 0. && direction.Mag() > 1e-4) {
732  geo::PlaneID plane(h.cryo, h.tpc, h.plane);
733  float angletovert = geo->WireAngleToVertical(geo->View(plane), plane) - 0.5*::util::pi<>();
734  float cosgamma = abs(cos(angletovert) * direction.Z() + sin(angletovert) * direction.Y());
735  float pitch = geo->WirePitch(plane) / cosgamma;
736  h.pitch = pitch;
737  }
738  else {
739  h.pitch = -1.;
740  }
741  // And the pitch induced by SCE
742  if (closest_dist >= 0. && direction.Mag() > 1e-4) {
743  geo::PlaneID plane(h.cryo, h.tpc, h.plane);
744  float angletovert = geo->WireAngleToVertical(geo->View(plane), plane) - 0.5*::util::pi<>();
745 
746  TVector3 loc_mdx_v = h_p - direction * (geo->WirePitch(geo->View(plane) / 2.));
747  TVector3 loc_pdx_v = h_p + direction * (geo->WirePitch(geo->View(plane) / 2.));
748 
749  // Convert types for helper functions
750  geo::Point_t loc_mdx(loc_mdx_v.X(), loc_mdx_v.Y(), loc_mdx_v.Z());
751  geo::Point_t loc_pdx(loc_pdx_v.X(), loc_pdx_v.Y(), loc_pdx_v.Z());
752  geo::Point_t h_p_point(h_p.X(), h_p.Y(), h_p.Z());
753 
754  loc_mdx = TrajectoryToWirePosition(loc_mdx, plane);
755  loc_pdx = TrajectoryToWirePosition(loc_pdx, plane);
756 
757  // Direction at wires
758  geo::Vector_t dir = (loc_pdx - loc_mdx) / (loc_mdx - loc_pdx).r();
759 
760  // Pitch at wires
761  double cosgamma = std::abs(std::sin(angletovert)*dir.Y() + std::cos(angletovert)*dir.Z());
762  double pitch;
763  if (cosgamma) {
764  pitch = geo->WirePitch(geo->View(plane))/cosgamma;
765  }
766  else {
767  pitch = 0.;
768  }
769 
770  // Now bring that back to the particle trajectory
771  geo::Point_t loc_w = TrajectoryToWirePosition(h_p_point, plane);
772 
773  geo::Point_t locw_pdx_traj = WireToTrajectoryPosition(loc_w + pitch*dir, plane);
774  geo::Point_t loc = WireToTrajectoryPosition(loc_w, plane);
775 
776  h.pitch_sce = (locw_pdx_traj - loc).R();
777  }
778  else {
779  h.pitch_sce = -1.;
780  }
781 
782  // And the trajectory location
783  h.itraj = traj_index;
784 
785  // And the residual range of the hit
786  h.rr = 0.;
787  if (traj_index >= 0) {
788  for (int i_traj = traj_index+1; i_traj < (int)particle.NumberTrajectoryPoints(); i_traj++) {
789  h.rr += (particle.Position(i_traj).Vect() - particle.Position(i_traj-1).Vect()).Mag();
790  }
791 
792  // Also account for the distance from the Hit point to the matched trajectory point
793  double hit_distance_along_particle = (h_p - particle.Position(traj_index).Vect()).Dot(particle.Momentum(traj_index).Vect().Unit());
794  h.rr += -hit_distance_along_particle;
795 
796  }
797  }
798 
799  // Order the hits by their location along the trajectory, start to end
800  std::sort(truehits_v.begin(), truehits_v.end(),
801  [](auto const &lhs, auto const &rhs) {
802  return lhs.itraj < rhs.itraj;
803  });
804 
805  // Save depositions into the True Particle
806  for (sbn::TrueHit &h: truehits_v) {
807  if (h.plane == 0) {
808  trueparticle.truehits0.push_back(h);
809  }
810  else if (h.plane == 1) {
811  trueparticle.truehits1.push_back(h);
812  }
813  else if (h.plane == 2) {
814  trueparticle.truehits2.push_back(h);
815  }
816  }
817 
818  return trueparticle;
819 }
