16  double OscCalcSterileApprox::GetSinSq2ThetaEE() const
17  {
18  // The three angles are coupled via
19  //
20  // ss2thmm = 4*Um4^2*(1-Um4^2)
21  // ss2thee = 4*Ue4^2*(1-Ue4^2)
22  // ss2thme = 4*Um4^2*Ue4^2
23  //
24  // Solve for the final (nue survival angle). Emperically, choosing the
25  // negative sign in the expression works well.
26 
27  // A couple of limits
28  if(fSinSq2ThetaMuMu <= 0) return fSinSq2ThetaMuE / 4;
29  if(fSinSq2ThetaMuMu >= 1) return fSinSq2ThetaMuE / 2;
30 
31  // General case
32  const double Ue4sq = fSinSq2ThetaMuE/(2*fSinSq2ThetaMuMu)*(1-sqrt(1-fSinSq2ThetaMuMu));
33  assert(!isinf(Ue4sq) && !isnan(Ue4sq));
34  const double sinsq2thetaee = Ue4sq*(1-Ue4sq);
35 
36  assert(sinsq2thetaee >= 0 && sinsq2thetaee <= 1);
37 
38  // Cross-check that we actually solved the equations correctly
39  if(Ue4sq > 0){
40  const double Umu4sq = fSinSq2ThetaMuE/(4*Ue4sq);
41  assert(fabs(4*Umu4sq*(1-Umu4sq) - fSinSq2ThetaMuMu) < 1e-6);
42  }
43 
44  return sinsq2thetaee;
45  }

55  double AvgSinSq(double k, double a, double b)
56  {
57  // This function shows up high in profiles, and is often called with the same masses/baselines/energies
58  static std::map<std::tuple<double, double, double>, double> cache;
59  const std::tuple<double, double, double> key = {k, a, b};
60 
61  // energies times masses times baselines plus some slack
62  if(cache.size() > 100*100*10) cache.clear();
63 
64  auto it = cache.find(key);
65  if(it != cache.end()) return it->second;
66 
67  double ret = 0;
68  if(a == b){
69  ret = util::sqr(sin(k/a));
70  }
71  else{
72  // https://www.wolframalpha.com/input/?i=integral+sin%5E2(k%2Fx)+from+a+to+b
73  ret = k*(Si(2*k/a)-Si(2*k/b));
74  assert(!isnan(ret));
75  if(a) ret -= a * util::sqr(sin(k/a));
76  assert(!isnan(ret));
77  if(b) ret += b * util::sqr(sin(k/b));
78  assert(!isnan(ret));
79 
80  ret /= (b-a);
81  }
82 
83  cache[key] = ret;
84  return ret;
