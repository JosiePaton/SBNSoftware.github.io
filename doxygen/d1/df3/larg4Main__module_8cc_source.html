140  // We need all of the services to run @produces@ on the data they will store. We do this
141  // by retrieving the holder services.
142  art::ServiceHandle<artg4tk::ActionHolderService> actionHolder;
143  art::ServiceHandle<artg4tk::DetectorHolderService> detectorHolder;
144 
145  if (initializeDetectors_) {
146  detectorHolder->initialize();
147  // Build the detectors' logical volumes
148  detectorHolder->constructAllLVs();
149  initializeDetectors_ = false;
150  }
151 
152  // And running @callArtProduces@ on each
153  actionHolder->callArtProduces(producesCollector());
154  detectorHolder->callArtProduces(producesCollector());
155 
156  // -- Check for invalid seed value
157  if (seed_ > 900000000) {
158  throw cet::exception("largeant:BadSeedValue")
159  << "The provided largeant seed value: " << seed_ << " is invalid! Maximum seed value is 9E8.";
160  }
161  // Set up the random number engine.
162  // -- D.R.: Use the NuRandomService engine for additional control over the seed generation policy
163  (void)art::ServiceHandle<rndm::NuRandomService>()->createEngine(*this, "G4Engine", p, "seed");

165  // Handle the afterEvent setting

189 void
190 larg4::larg4Main::beginRun(art::Run& r)
191 {
192  if (processingRun_++ != 0) {
193  return;
194  }
195 
196  // Get the physics list and pass it to Geant and initialize the list if necessary
197  art::ServiceHandle<artg4tk::PhysicsListHolderService const> physicsListHolder;
198  runManager_->SetUserInitialization(physicsListHolder->makePhysicsList());
199 
200  // Get all of the detectors and initialize them
201  // Declare the detector construction to Geant
202  auto detectorHolder = art::ServiceHandle<artg4tk::DetectorHolderService>().get();
203  runManager_->SetUserInitialization(
204  new artg4tk::ArtG4DetectorConstruction{detectorHolder->worldPhysicalVolume()});
205 
206  // Get all of the actions and initialize them
207  auto actionHolder = art::ServiceHandle<artg4tk::ActionHolderService>().get();
208  actionHolder->initialize();
209 
210  // Store the run in the action holder
211  actionHolder->setCurrArtRun(r);
212 
213  // Declare the primary generator action to Geant
214  runManager_->SetUserAction(new artg4tk::ArtG4PrimaryGeneratorAction{actionHolder});
215 
216  // Note that these actions (and ArtG4PrimaryGeneratorAction above) are all
217  // generic actions that really don't do much on their own. Rather, to
218  // use the power of actions, one must create action objects (derived from
219  // @ActionBase@) and register them with the Art @ActionHolder@ service.
220  // See @ActionBase@ and/or @ActionHolderService@ for more information.
221  runManager_->SetUserAction(new artg4tk::ArtG4SteppingAction{actionHolder});
222  runManager_->SetUserAction(new artg4tk::ArtG4StackingAction{actionHolder});
223  runManager_->SetUserAction(new artg4tk::ArtG4EventAction{actionHolder, detectorHolder});
224  runManager_->SetUserAction(new artg4tk::ArtG4TrackingAction{actionHolder});
225 
226  runManager_->Initialize();
227  physicsListHolder->initializePhysicsList();
228 
229  //get the pointer to the User Interface manager
230  UI_ = G4UImanager::GetUIpointer();
231 
232  // Find the macro (or try to) along the directory path.
233  std::string macroLocation = "";
234  bool macroWasFound = pathFinder_.find_file(g4MacroFile_, macroLocation);
235  mf::LogInfo("larg4Main") << "Finding path for " << g4MacroFile_ << "...\nSearch "
236  << (macroWasFound ? "successful " : "unsuccessful ") << "and path is: \n"
237  << macroLocation;
238 
239  // Execute the macro if we were able to find it
240  if (macroWasFound) {
241  // Create the string containing the execution command
242  mf::LogInfo("larg4Main") << "Executing macro: " << g4MacroFile_;
243  std::string commandToExecute = "/control/execute ";
244  commandToExecute.append(macroLocation);
245  UI_->ApplyCommand(commandToExecute);
246  }
247  else {
248  mf::LogInfo("larg4Main") << "Unable to find " << g4MacroFile_ << " in the path(s) "
249  << macroPath_;
250  }
251 
252  // Open a UI if asked
253  if (uiAtBeginRun_) {
254  session_ = new G4UIterminal;
255  session_->SessionStart();
256  delete session_;
257  }
258 
259  // Start the Geant run!
260  runManager_->BeamOnBeginRun(r.id().run());
261 }

266 {
267  // The holder services need the event
268  art::ServiceHandle<artg4tk::ActionHolderService>()->setCurrArtEvent(e);
269  art::ServiceHandle<artg4tk::DetectorHolderService>()->setCurrArtEvent(e);
270 
271  auto const mclists = inputCollections(e);
272  art::ServiceHandle<larg4::MCTruthEventActionService>()->setInputCollections(mclists);
273 
274  art::ServiceHandle<larg4::ParticleListActionService> pla;
275  pla->setInputCollections(mclists);
276  auto const pid = e.getProductID<std::vector<simb::MCParticle>>();
277  pla->setPtrInfo(pid, e.productGetter(pid));
278 
279  runManager_->BeamOnDoOneEvent(e.id().event());
280  runManager_->BeamOnEndEvent();
281 
282  e.put(pla->ParticleCollection());
283  e.put(pla->AssnsMCTruthToMCParticle());
284  e.put(pla->DroppedTracksCollection());
