197 //-----------------------------------------------------------------------
198 void MCTruthTestAna::analyze(const art::Event& event)
199 {
200  // Recover the backtracker
201  art::ServiceHandle<cheat::BackTrackerService> backTracker;
202  art::ServiceHandle<cheat::ParticleInventoryService> partInventory;
203 
204  // "Rebuild" the maps used by the parallel backtracker
205  fMCTruthMatching->Rebuild(event);
206 
207  // Begin the comparisons by simply checking that the number of MCParticles agree between the two
208  const sim::ParticleList& particleList = fMCTruthMatching->ParticleList();
209  const sim::ParticleList& btPartList = partInventory->ParticleList();
210 
211  if (particleList.size() != btPartList.size())
212  {
213  mf::LogInfo("MCTruthTestAna") << "*** ParticleList size mismatch, BackTracker says: " << btPartList.size() << ", alternate says: " << particleList.size() << std::endl;
214  }
215 
216  // Flush with the success of passing that test, let's recover the hits and then loop through them
217  // and do hit-by-hit comparisons...
218  art::Handle<std::vector<recob::Hit> > hitHandle;
219  event.getByLabel(fHitProducerLabel, hitHandle);
220 
221  auto const clockData = art::ServiceHandle<detinfo::DetectorClocksService const>()->DataFor(event);
222  if (hitHandle.isValid())
223  {
224  int nTotalMisMatches(0);
225  int nNegTrackIDs(0);
226 
227  for(const auto& hit : *hitHandle)
228  {
229  int nIDEMisMatches(0);
230 
231  // Check the claimed parentage of the current hit
232  std::vector<sim::TrackIDE> trackIDEVec = fMCTruthMatching->HitToTrackID(clockData, hit);
233  std::vector<sim::TrackIDE> btTrkIDEVec = backTracker->HitToTrackIDEs(clockData, hit);
234 
235  int deltaIDs = int(btTrkIDEVec.size()) - int(trackIDEVec.size());
236 
237  fNumIDEHist->Fill(trackIDEVec.size(),1.);
238  fDeltaIDEHist->Fill(deltaIDs, 1.);
239 
240  // It can be that the BackTracker is returning IDE's with negative Track ID's and these don't
241  // appear in the associations, so bypass them here...
242  if (deltaIDs != 0)
243  {
244  int numNeg = std::count_if(btTrkIDEVec.begin(),btTrkIDEVec.end(),[](const auto& trackIDE){return trackIDE.trackID < 0;});
245 
246  deltaIDs -= numNeg;
247 
248  std::sort(btTrkIDEVec.begin(),btTrkIDEVec.end(),[](const auto& left, const auto& right){return left.trackID > right.trackID;});
249 
250  while(btTrkIDEVec.back().trackID < 0)
251  {
252  fNegEnergyHist->Fill(std::min(9.98,double(btTrkIDEVec.back().energy)), 1.);
253  fNegNElecHist->Fill(std::min(99.9,double(btTrkIDEVec.back().numElectrons)), 1.);
254 
255  btTrkIDEVec.pop_back();
256  nNegTrackIDs++;
257  }
258  }
259 
260  // Currently assume BackTracker is "correct" so loop over its output and look for match
261  for(const auto& btTrkID : btTrkIDEVec)
262  {
263  std::vector<sim::TrackIDE>::const_iterator trackIDItr = std::find_if(trackIDEVec.begin(),trackIDEVec.end(),[btTrkID](const auto& id){return btTrkID.trackID == id.trackID && btTrkID.energy == id.energy && btTrkID.numElectrons == id.numElectrons;});
264 
265  if (trackIDItr == trackIDEVec.end()) nIDEMisMatches++;
266  }
267 
268  fIDEMisMatchHist->Fill(nIDEMisMatches, 1.);
269  nTotalMisMatches += nIDEMisMatches;
270  }
271 
272  mf::LogInfo("MCTruthTestAna") << "==> Found " << nTotalMisMatches << " between BackTracker and Associations, BT reported " << nNegTrackIDs << " negative track ids \n";
273  }
274 
275  // We're on a roll now!
276  // So let's see if we can recover information about tracks
277  art::Handle<std::vector<recob::Track>> trackHandle;
278  event.getByLabel(fTrackProducerLabel, trackHandle);
279 
280  if (trackHandle.isValid())
281  {
282  // We are going to want a vector of art pointers to hits, so build here
283  std::vector<art::Ptr<recob::Hit>> hitPtrVector;
284 
285  art::fill_ptr_vector(hitPtrVector, hitHandle);
286 
287  // Recover the associations between tracks and hits
288  art::FindManyP<recob::Hit> hitTrackAssns(trackHandle, event, fTrackProducerLabel);
289 
290  int nBadEffMatches(0);
291  int nBadPurMatches(0);
292 
293  for(size_t trackIdx = 0; trackIdx < trackHandle->size(); trackIdx++)
294  {
295  art::Ptr<recob::Track> track(trackHandle,trackIdx);
296 
297  std::vector<art::Ptr<recob::Hit>> trackHitVec = hitTrackAssns.at(track.key());
298 
299  std::set<int> trackIDSet = fMCTruthMatching->GetSetOfTrackIDs(clockData, trackHitVec);
300  std::set<int> btTrkIDSet = backTracker->GetSetOfTrackIds(clockData, trackHitVec);
301 
302  // Check for case were we might have negative track IDs from the BackTracker
303  if (btTrkIDSet.size() > trackIDSet.size())
304  {
305  // Note that the container here is a std::set which will be ordered smallest to largest.
306  // In this case we want to remove elements from the front until they are positive
307  for(std::set<int>::iterator btTrkIDSetItr = btTrkIDSet.begin(); btTrkIDSetItr != btTrkIDSet.end();)
308  {
309  if (!(*btTrkIDSetItr < 0)) break;
310 
311  btTrkIDSetItr = btTrkIDSet.erase(btTrkIDSetItr);
312  }
313  }
314 
315  if (btTrkIDSet.size() != trackIDSet.size())
316  mf::LogDebug("MCTruthTestAna") << "Mismatch in associated track ids, backtracker: " << btTrkIDSet.size() << ", associations: " << trackIDSet.size() << ", track id: " << track.key() << "\n";
317 
318  // Ok, turn the "set" of track ID's into a vector of track ID's
319  std::vector<int> btTrkIDVec;
320 
321  for(auto& trackID : btTrkIDSet) btTrkIDVec.emplace_back(trackID);
322 
323  // And then use this to recover the vectors of hits associated to each MC track
324  std::vector<std::vector<art::Ptr<recob::Hit>>> trkHitVecVec;
325 
326  for(const auto& tkID : btTrkIDVec)
327  {
328  std::vector<art::Ptr<recob::Hit>> hitVec = backTracker->TrackIdToHits_Ps(clockData, tkID, hitPtrVector);
329  trkHitVecVec.push_back(hitVec);
330  }
331  // Apply majority logic - we declare the MCParticle with the most hits to be the "winner"
332  std::vector<std::vector<art::Ptr<recob::Hit>>>::iterator bestTrkHitVecItr = std::max_element(trkHitVecVec.begin(),trkHitVecVec.end(),[](const auto& a, const auto& b){return a.size() < b.size();});
333 
334  if (bestTrkHitVecItr == trkHitVecVec.end())
335  {
336  // I have been watching way too much Star Trek (the original!)
337  mf::LogDebug("MCTruthTestAna") << ">>>>>>> ERROR! >>>>>>> ERROR! >>>>>> MUST PURIFY! >>>>>> ERROR!" << "\n";
338  continue;
339  }
340 
341  int indexToBestMC = std::distance(trkHitVecVec.begin(),bestTrkHitVecItr);
342  int bestMCTrackID = btTrkIDVec.at(indexToBestMC);
343 
344  std::vector<art::Ptr<recob::Hit>>& bestMCTrackHitVec = *bestTrkHitVecItr;
345 
346  std::set<int> mcTrackIdxSet = {bestMCTrackID};
347 
348  double btTrkEffic = backTracker->HitCollectionEfficiency(clockData, mcTrackIdxSet, trackHitVec, bestMCTrackHitVec, geo::k3D);
349  double trackEffic = fMCTruthMatching->HitCollectionEfficiency(clockData, mcTrackIdxSet, trackHitVec, bestMCTrackHitVec, geo::k3D);
350 
351  if (btTrkEffic != trackEffic)
352  {
353  mf::LogDebug("MCTruthTestAna") << "Efficiency mismatch, track ID: " << bestMCTrackID << ", track: " << track.key() << ", # hits: " << trackHitVec.size() << ", # MC hits: " << bestMCTrackHitVec.size() << ", btTrkEff: " << btTrkEffic << ", trackEff: " << trackEffic << "\n|";
354  nBadEffMatches++;
355  }
356 
357  fBTEfficiencyHist->Fill(std::min(1.01,btTrkEffic), 1.);
358  fAssocEfficiencyHist->Fill(std::min(1.01,trackEffic), 1.);
359  fEfficiencyCompHist->Fill(std::min(1.01,btTrkEffic), std::min(1.01,trackEffic), 1.);
360 
361  double btTrkPurity = backTracker->HitCollectionPurity(clockData, mcTrackIdxSet, trackHitVec);
362  double trackPurity = fMCTruthMatching->HitCollectionPurity(clockData, mcTrackIdxSet, trackHitVec);
363 
364  if (btTrkPurity != trackPurity)
365  {
366  mf::LogDebug("MCTruthTestAna") << "Purity mismatch, track ID: " << bestMCTrackID << ", track: " << track.key() << ", # hits: " << trackHitVec.size() << ", # MC hits: " << bestMCTrackHitVec.size() << ", btTrkPurity: " << btTrkPurity << ", trackPurity: " << trackPurity << "\n|";
367  nBadPurMatches++;
368  }
369 
370  fBTPurityHist->Fill(std::min(1.01,btTrkPurity), 1.);
371  fAssocPurityHist->Fill(std::min(1.01,trackPurity), 1.);
372  fPurityCompHist->Fill(std::min(1.01,btTrkPurity),std::min(1.01,trackPurity), 1.);
373  }
374 
375  mf::LogDebug("MCTruthTestAna") << "Event with " << trackHandle->size() << " reconstructed tracks, found " << nBadEffMatches << " efficiency mismatches, " << nBadPurMatches << " purity mismatchs \n";
376  }
377 
378  return;
379 }
