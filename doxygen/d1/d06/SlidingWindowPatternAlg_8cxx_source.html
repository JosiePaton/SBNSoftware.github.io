59 auto icarus::trigger::SlidingWindowPatternAlg::simulateResponse
60  (TriggerGates_t const& gates) const -> AllTriggerInfo_t
61 {
62 
63  // ensures input gates are in the same order as the configured windows
64  verifyInputTopology(gates);
65 
66  auto const& inBeamGates = fBeamGate? fBeamGate->applyToAll(gates): gates;
67 
68  //
69  // 2. apply pattern:
70  //
71  std::size_t const nWindows = fWindowTopology.nWindows();
72 
73  //
74  // 2.1. for each main window, apply the pattern
75  //
76  WindowTriggerInfo_t triggerInfo; // start empty
77  for (std::size_t const iWindow: util::counter(nWindows)) {
78 
79  TriggerInfo_t const windowResponse
80  = applyWindowPattern(fWindowPattern, iWindow, inBeamGates);
81 
82  if (!windowResponse) continue;
83 
84  assert(windowResponse.hasLocation());
85  assert(windowResponse.location() == iWindow);
86  if (windowResponse.nTriggers() == 1U) {
87  mfLogTrace() << "Pattern fired on window #" << iWindow
88  << " at tick " << windowResponse.atTick();
89  }
90  else {
91  std::vector<TriggerInfo_t::OpeningInfo_t> const& triggers
92  = windowResponse.all();
93  TriggerInfo_t::OpeningInfo_t const* pMain = &(windowResponse.main());
94  auto log = mfLogTrace();
95  log << "Pattern fired " << triggers.size()
96  << " times on window #" << iWindow << ":";
97  for (auto const& [ iTrigger, trigger ]: util::enumerate(triggers)) {
98  log << " [#" << iTrigger << "] " << " on " << trigger.tick
99  << " (";
100  if (&trigger == pMain) log << "main; ";
101  log << "lvl=" << trigger.level << ")";
102  } // for
103  }
104 
105  //
106  // 2.2. pick the main window with the earliest successful response, if any;
107  // that defines location and time of the trigger
108  //
109  if (!triggerInfo || triggerInfo.info.atTick() > windowResponse.atTick()) {
110  if (!triggerInfo) mfLogTrace() << " (new global trigger)";
111  triggerInfo.emplace(iWindow, windowResponse);
112  }
113 
114  } // main window choice
115 
116  return { std::move(triggerInfo.info), MoreInfo_t{ triggerInfo.windowIndex } };
117 } // icarus::trigger::SlidingWindowPatternAlg::simulateResponse()

136 //------------------------------------------------------------------------------
137 void icarus::trigger::SlidingWindowPatternAlg::verifyInputTopology(TriggerGates_t const& gates) const {
138  /*
139  * Verifies that the `gates` are in the expected order and have
140  * the expected channel content.
141  */
142 
143  std::string errorMsg; // if this stays `empty()` there is no error
144  for (auto const& [ iWindow, gate ]: util::enumerate(gates)) {
145 
146  std::string windowError; // if this stays `empty()` there is no error
147 
148  // more input gates than windows?
149  if (iWindow >= fWindowTopology.nWindows()) {
150  windowError = "unexpected input gate #" + std::to_string(iWindow) + " (";
151  for (raw::Channel_t const channel: gate.channels())
152  windowError += " " + std::to_string(channel);
153  windowError += " )";
154  errorMsg += windowError + '\n';
155  continue; // we collect info of all spurious gates
156  }
157 
158  WindowTopology_t::WindowInfo_t const& windowInfo
159  = fWindowTopology.info(iWindow);
160 
161  auto const channelInWindow
162  = [begin=windowInfo.channels.cbegin(),end=windowInfo.channels.cend()]
163  (raw::Channel_t channel)
164  { return std::binary_search(begin, end, channel); }
165  ;
166 
167  for (raw::Channel_t const channel: gate.channels()) {
168  if (channelInWindow(channel)) continue;
169  if (windowError.empty()) {
170  windowError =
171  "channels not in window #" + std::to_string(windowInfo.index)
172  + ":";
173  } // if first error
174  windowError += " " + std::to_string(channel);
175  } // for all channels in gate
176 
177  if (!windowError.empty()) errorMsg += windowError + '\n';
178  } // for gates
179 
180  // more input gates than windows?
181  if (fWindowTopology.nWindows() > size(gates)) {
182  errorMsg +=
183  "Not enough input gates: " + std::to_string(size(gates)) + " gates for "
184  + std::to_string(fWindowTopology.nWindows()) + " windows\n";
185  }
186 
187  if (errorMsg.empty()) return;
188 
189  // put together the exception message and throw it.
190  throw cet::exception("SlidingWindowPatternAlg")
191  << "Some channels from trigger gates do not match"
192  " the configured window allocation:\n"
193  << errorMsg
194  << "\n" // empty line
195  << "Window configuration: "
196  << fWindowTopology << "\n";
197 
198 } // icarus::trigger::SlidingWindowPatternAlg::verifyInputTopology()

201 //------------------------------------------------------------------------------
202 auto icarus::trigger::SlidingWindowPatternAlg::applyWindowPattern(
203  WindowTopology_t::WindowInfo_t const& windowInfo,
204  WindowPattern_t const& pattern,
205  TriggerGates_t const& gates
206  ) const -> TriggerInfo_t
207 {
208 
209  /*
210  * 1. check that the pattern can be applied; if not, return no trigger
211  * 2. discriminate all the relevant gates against their required minimum count
212  * 3. combine them in AND
213  * 4. find the trigger time, fill the trigger information accordingly
214  */
215  TriggerInfo_t res; // no trigger by default
216  assert(!res);
217 
218  //
219  // 1. check that the pattern can be applied; if not, return no trigger
220  //
221 
222  // check that the pattern centered into iWindow has all it needs:
223  if (pattern.requireUpstreamWindow && !windowInfo.hasUpstreamWindow())
224  return res;
225  if (pattern.requireDownstreamWindow && !windowInfo.hasDownstreamWindow())
226  return res;
227 
228 
229  //
230  // 2. discriminate all the relevant gates against their required minimum count
231  // 3. combine them in AND
232  //
233 
234  mfLogTrace()
235  << "Window info #" << windowInfo.index << " pattern " << pattern.tag();
236 
237  // we have two different "modes" depending on whether we are constraining
238  // the sum of main and opposite window, or not;
239  std::optional<TriggerGateData_t> const mainPlusOpposite
240  = (pattern.minSumInOppositeWindows > 0U)
241  ? std::optional{
242  windowInfo.hasOppositeWindow()
243  ? sumGates(gates[windowInfo.index], gates[windowInfo.opposite])
244  : gates[windowInfo.index]
245  }
246  : std::nullopt
247  ;
248 
249  // the basic trigger primitive gate has the levels of the main or
250  // main+opposite window depending on the requirements
251  TriggerGateData_t trigPrimitive
252  = mainPlusOpposite? *mainPlusOpposite: gates[windowInfo.index];
253 
254  mfLogTrace() << " base: " << compactdump(trigPrimitive);
255 
256  // main window
257  if (pattern.minInMainWindow > 0U) {
258  trigPrimitive.Mul
259  (discriminate(gates[windowInfo.index], pattern.minInMainWindow));
260  mfLogTrace()
261  << " main >= " << pattern.minInMainWindow << ": "
262  << compactdump(trigPrimitive);
263  } // if
264 
265  // add opposite window requirement (if any)
266  if ((pattern.minInOppositeWindow > 0U) && windowInfo.hasOppositeWindow()) {
267  trigPrimitive.Mul
268  (discriminate(gates[windowInfo.opposite], pattern.minInOppositeWindow));
269  mfLogTrace() << " opposite [#" << windowInfo.opposite << "]: "
270  << compactdump(gates[windowInfo.opposite])
271  << "\n => " << compactdump(trigPrimitive);
272  } // if
273 
274  // add main plus opposite window requirement (if any)
275  if (pattern.minSumInOppositeWindows > 0U) {
276  assert(mainPlusOpposite.has_value());
277  trigPrimitive.Mul
278  (discriminate(*mainPlusOpposite, pattern.minSumInOppositeWindows));
279  mfLogTrace() << " sum [+ #" << windowInfo.opposite << "]: "
280  << compactdump(*mainPlusOpposite)
281  << "\n => " << compactdump(trigPrimitive);
282 
283  } // if
284 
285  // add upstream window requirement (if any)
286  if ((pattern.minInUpstreamWindow > 0U) && windowInfo.hasUpstreamWindow()) {
287  trigPrimitive.Mul
288  (discriminate(gates[windowInfo.upstream], pattern.minInUpstreamWindow));
289  } // if
290 
291  // add downstream window requirement (if any)
292  if ((pattern.minInDownstreamWindow > 0U) && windowInfo.hasDownstreamWindow())
293  {
294  trigPrimitive.Mul(
295  discriminate(gates[windowInfo.downstream], pattern.minInDownstreamWindow)
296  );
297  } // if
298 
299  mfLogTrace() << " final: " << compactdump(trigPrimitive);
300 
301  //
302  // 4. find the trigger time, fill the trigger information accordingly
303  //
304  icarus::trigger::details::GateOpeningInfoExtractor extractOpeningInfo
305  { trigPrimitive };
306 
307  extractOpeningInfo.setLocation
308  (TriggerInfo_t::LocationID_t{ windowInfo.index });
309 
310  while (extractOpeningInfo) {
311  auto info = extractOpeningInfo();
312  if (info) res.add(info.value());
313  } // while
314 
315  return res;
316 
317 } // icarus::trigger::SlidingWindowTriggerEfficiencyPlots::applyWindowPattern()
