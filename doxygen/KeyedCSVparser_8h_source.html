266 std::optional<T> icarus::details::KeyValuesData::Item::getOptionalNumber
267  (std::size_t index, bool ignoreFormatErrors /* = false */) const
268 {
269  if (index < values.size()) return std::nullopt;

273  return (number || ignoreFormatErrors)

282  (std::string const& valueStr)
283 {
284  T number {}; // useless initialization to avoid GCC complains
285  char const *b = valueStr.data(), *e = b + valueStr.length();
286  return (std::from_chars(b, e, number).ptr == e)
287  ? std::make_optional(number): std::nullopt;
