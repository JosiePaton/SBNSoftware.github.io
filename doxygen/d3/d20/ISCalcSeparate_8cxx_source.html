52  // fNumIonElectrons returns a value that is not corrected for life time effects
53  double
54  ISCalcSeparate::CalcIon(detinfo::DetectorPropertiesData const& detProp,
55  sim::SimEnergyDeposit const& edep)
56  {
57  float e = edep.Energy();
58  float ds = edep.StepLength();
59 
60  double recomb = 0.;
61  double dEdx = (ds <= 0.0) ? 0.0 : e / ds;
62  double EFieldStep = EFieldAtStep(detProp.Efield(), edep);
63 
64  // Guard against spurious values of dE/dx. Note: assumes density of LAr
65  if (dEdx < 1.) { dEdx = 1.; }
66 
67  if (fUseModBoxRecomb) {
68  if (ds > 0) {
69  double const scaled_modboxb = fModBoxB / detProp.Density(detProp.Temperature());
70  double const Xi = scaled_modboxb * dEdx / EFieldStep;
71  recomb = log(fModBoxA + Xi) / Xi;
72  }
73  else {
74  recomb = 0;
75  }
76  }
77  else {
78  double const scaled_recombk = fRecombk / detProp.Density(detProp.Temperature());
79  recomb = fRecombA / (1. + dEdx * scaled_recombk / EFieldStep);
80  }
81 
82  // 1.e-3 converts fEnergyDeposit to GeV
83  auto const numIonElectrons = fGeVToElectrons * 1.e-3 * e * recomb;
84 
85  MF_LOG_DEBUG("ISCalcSeparate")
86  << " Electrons produced for " << edep.Energy() << " MeV deposited with " << recomb
87  << " recombination: " << numIonElectrons << std::endl;
88  return numIonElectrons;
89  }

93  ISCalcSeparate::CalcScint(sim::SimEnergyDeposit const& edep)
94  {
95  float const e = edep.Energy();

97

102 
103  switch (pdg) {

105  case 13:
106  case -13: scintYield = fLArProp->MuonScintYield(true); break;
107  case 211:
108  case -211: scintYield = fLArProp->PionScintYield(true); break;
109  case 321:
110  case -321: scintYield = fLArProp->KaonScintYield(true); break;
111  case 1000020040: scintYield = fLArProp->AlphaScintYield(true); break;
112  case 11:
113  case -11:
114  case 22: scintYield = fLArProp->ElectronScintYield(true); break;
115  default: scintYield = fLArProp->ElectronScintYield(true);
116  }
117 
118  numScintPhotons = scintYield * e;
119  }
120  else {
121  numScintPhotons = scint_yield_factor * scintYield * e;
122  }
123 
124  return {numScintPhotons, GetScintYieldRatio(edep)};
125  }
