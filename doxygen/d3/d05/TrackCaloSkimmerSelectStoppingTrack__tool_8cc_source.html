118 bool TrackCaloSkimmerSelectStoppingTrack::Select(const TrackInfo &t) {
119  bool downwards = (t.dir.y < 0.) || !fRequireDownwards;
120 
121  bool end_is_fid = false;
122  for (const geo::BoxBoundedGeo &g: fFiducialVolumes) {
123  geo::Point_t end {t.end.x, t.end.y, t.end.z};
124 
125  if (g.ContainsPosition(end)) {
126  end_is_fid = true;
127  break;
128  }
129  }
130 
131  // Collection plane times need to be fiducial
132  bool time_is_fid = \
133  (t.hit_min_time_p2_tpcE < 0. || t.hit_min_time_p2_tpcE > fFidTickMin) &&
134  (t.hit_max_time_p2_tpcE < 0. || t.hit_max_time_p2_tpcE < fFidTickMax) &&
135  (t.hit_min_time_p2_tpcW < 0. || t.hit_min_time_p2_tpcW > fFidTickMin) &&
136  (t.hit_max_time_p2_tpcW < 0. || t.hit_max_time_p2_tpcW < fFidTickMax);
137 
138  // compute the median dqdx of the last few cm -- using fMediandQdxRRMax
139  std::vector<double> endp_dqdx;
140  for (const sbn::TrackHitInfo &h: t.hits2) {
141  if (h.oncalo && h.rr < fMediandQdxRRMax) endp_dqdx.push_back(h.dqdx);
142  }
143  double med_dqdx = -1;
144  if (endp_dqdx.size()) {
145  unsigned middle = endp_dqdx.size() / 2;
146  std::nth_element(endp_dqdx.begin(), endp_dqdx.begin() + middle, endp_dqdx.end());
147  med_dqdx = endp_dqdx[middle];
148 
149  // for even case
150  if (endp_dqdx.size() % 2 == 0) {
151  unsigned other_middle = middle - 1;
152  std::nth_element(endp_dqdx.begin(), endp_dqdx.begin() + other_middle, endp_dqdx.end());
153  med_dqdx = (med_dqdx + endp_dqdx[other_middle]) / 2.;
154  }
155  }
156 
157  bool valid_med_dqdx = ((med_dqdx > 0.) && (med_dqdx > fEndMediandQdxCut)) || (fEndMediandQdxCut < 0.);
158 
159  return downwards && end_is_fid && time_is_fid && valid_med_dqdx;
160 }
