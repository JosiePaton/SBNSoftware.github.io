43 /**
44  * @brief Extracts and saves the time coverage of optical detector waveforms.
45  *
46  * This module writes a list of `sbn::OpDetWaveformMeta` objects matching the
47  * information of each optical detector waveform.
48  *
49  * It may be used as input to modules which require to operate on beam gates,
50  * to select time(s) around the reconstructed (and selected) tracks.
51  *
52  *
53  * Input data products
54  * ====================
55  *
56  * This module acts on a _selection_ of tracks, which implies that the input is
57  * a set of _pointers_ to tracks rather than to an actual track collection.
58  * For each track, an associated time is required.
59  *
60  * * `std::vector<raw::OpDetWaveform>` (tag from `Waveforms`):
61  * all optical detector waveforms to extract the information from
62  *
63  *
64  * Output data products
65  * =====================
66  *
67  * * `std::vector<sbn::OpDetWaveformMeta>`: a collection parallel to the input
68  * one (from data product configured by `Waveforms`) with the summary
69  * information on each of them; also an explicit courtesy association
70  * `art::Assns<sbn::OpDetWaveformMeta, raw::OpDetWaveform>` for uses where
71  * order is not preserved.
72  * The times in these objects are on the same scale as the ones in the source
73  * data product, which is expected to be
74  * @ref DetectorClocksElectronicsStartTime "electronics time scale [us]".
75  *
76  *
77  * Configuration parameters
78  * =========================
79  *
80  * A terse online description of the parameters is printed by running
81  * `lar --print-description OpDetWaveformMetaMaker`.
82  *
83  * * `Waveforms` (input tag, mandatory): the list of optical detector waveforms
84  * to be processed.
85  * * `LogCategory` (string, default: `OpDetWaveformMetaMaker`): name of the
86  * output stream category for console messages (managed by MessageFacility
87  * library).
88  *
89  *
90  */

193 //------------------------------------------------------------------------------
194 void icarus::trigger::OpDetWaveformMetaMaker::produce
195  (art::Event& event, art::ProcessingFrame const&)
196 {
197 
198  using detinfo::timescales::electronics_time;
199 
200  //
201  // get the timing information for this event
202  //
203  detinfo::DetectorTimings const detTimings = detinfo::makeDetectorTimings
204  (art::ServiceHandle<detinfo::DetectorClocksService const>()->DataFor(event))
205  ;
206 
207  //
208  // fetch input
209  //
210  auto const& waveformHandle
211  = event.getValidHandle<std::vector<raw::OpDetWaveform>>(fWaveformTag);
212 
213  mf::LogDebug(fLogCategory) << "Event " << event.id()
214  << " has beam gate starting at " << detTimings.BeamGateTime()
215  << " and trigger at " << detTimings.TriggerTime() << "."
216  << "\nNow extracting information from " << waveformHandle->size()
217  << " waveforms."
218  ;
219 
220  //
221  // create the content
222  //
223  sbn::OpDetWaveformMetaMaker makeOpDetWaveformMeta{ detTimings };
224 
225  std::vector<sbn::OpDetWaveformMeta> PMTinfo;
226  art::Assns<sbn::OpDetWaveformMeta, raw::OpDetWaveform> infoToWaveform;
227 
228  art::PtrMaker<sbn::OpDetWaveformMeta> const makeInfoPtr { event };
229  art::PtrMaker<raw::OpDetWaveform> const makeWaveformPtr
230  { event, waveformHandle.id() };
231 
232  for (auto const& [ iWaveform, waveform ]: util::enumerate(*waveformHandle)) {
233  assert(iWaveform == PMTinfo.size());
234 
235  PMTinfo.push_back(makeOpDetWaveformMeta(waveform));
236  infoToWaveform.addSingle
237  (makeInfoPtr(iWaveform), makeWaveformPtr(iWaveform));
238 
239  {
240  sbn::OpDetWaveformMeta const& info = PMTinfo.back();
241  mf::LogTrace log{ fLogCategory };
242  log << "Coverage for waveform #" << iWaveform
243  << " on channel " << waveform.ChannelNumber() << ": "
244  << info.startTime << " -- " << info.endTime;
245  if (info.withTrigger()) log << "; includes trigger";
246  if (info.withBeamGate()) log << "; includes beam gate start";
247  }
248 
249  } // for
250 
251  //
252  // store output
253  //
254  event.put(moveToUniquePtr(PMTinfo));
255  event.put(moveToUniquePtr(infoToWaveform));
256 
257 } // icarus::trigger::OpDetWaveformMetaMaker::produce()
