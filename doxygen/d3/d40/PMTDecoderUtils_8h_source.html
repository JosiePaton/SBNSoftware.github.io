162 template <typename Coll, typename Key, typename KeyExtractor>
163 typename Coll::value_type const* daq::details::binarySearch
164  (Coll const& coll, Key const& key, KeyExtractor&& extractKey)
165 {
166  using std::begin, std::end;
167  auto const b = begin(coll);
168  auto const e = end(coll);
169  auto it = std::lower_bound(
170  b, e, key,
171  [&extractKey](auto const& a, Key const& key){ return extractKey(a) < key; }
172  );
173  return ((it != e) && (extractKey(*it) == key))? &*it: nullptr;
174 } // daq::details::binarySearch()
