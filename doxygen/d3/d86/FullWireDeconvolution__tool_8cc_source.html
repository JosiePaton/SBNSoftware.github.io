71 {
72  // Start by recovering the parameters
73  //wire-by-wire calibration
74  fDodQdxCalib = pset.get< bool >("DodQdxCalib", false);
75 
76  if (fDodQdxCalib)

78  fdQdxCalibFileName = pset.get< std::string >("dQdxCalibFileName");

80  cet::search_path sp("FW_SEARCH_PATH");

82 
83  if (fullname.empty())
84  {
85  std::cout << "Input file " << fdQdxCalibFileName << " not found" << std::endl;
86  throw cet::exception("File not found");
87  }
88  else
89  std::cout << "Applying wire-by-wire calibration using file " << fdQdxCalibFileName << std::endl;
90 
91  std::ifstream inFile(fullname, std::ios::in);
92  std::string line;
93 
94  while (std::getline(inFile,line))
95  {
96  unsigned int channel;
97  float constant;
98  std::stringstream linestream(line);
99  linestream >> channel >> constant;
100  fdQdxCalib[channel] = constant;
101  if (channel%1000==0) std::cout<<"Channel "<<channel<<" correction factor "<<fdQdxCalib[channel]<<std::endl;
102  }
103  }
104 
105  // Get signal shaping service.
106  fSignalShaping = art::ServiceHandle<icarusutil::SignalShapingICARUSService>();
107 
108  // Now set up our plans for doing the convolution
109  auto const detProp = art::ServiceHandle<detinfo::DetectorPropertiesService const>()->DataForJob();

115 void FullWireDeconvolution::Deconvolve(IROIFinder::Waveform const& waveform,
116  double const samplingRate,
117  raw::ChannelID_t channel,
118  IROIFinder::CandidateROIVec const& roiVec,
119  recob::Wire::RegionsOfInterest_t& ROIVec) const
120 {
121  // The goal of this function is to reproduce "exactly" the operation of the deconvolution process in MCC7
122  // hence the copying over of some of the code that has been pushed into external tools.
123 
124  // The size of the input waveform **should** be the raw buffer size
125  size_t dataSize = waveform.size();
126 
127  // Make sure the deconvolution size is set correctly (this will probably be a noop after first call)
128  fSignalShaping->SetDecon(samplingRate, dataSize, channel);
129 
130  // now make a buffer to contain the waveform which will be of the right size
131  icarusutil::TimeVec rawAdcLessPedVec(dataSize,0.);
132 
133  size_t binOffset = 0; //transformSize > dataSize ? (transformSize - dataSize) / 2 : 0;
134  float deconNorm = fSignalShaping->GetDeconNorm();
135  float normFactor = 1. / deconNorm; // This is what we had previously: (samplingRate * deconNorm);
136  bool applyNormFactor = std::abs(normFactor - 1.) > std::numeric_limits<float>::epsilon() ? true : false;
137 
138  // Copy the input (assumed pedestal subtracted) waveforms into our zero padded deconvolution buffer
139  std::copy(waveform.begin(),waveform.end(),rawAdcLessPedVec.begin()+binOffset);
140 
141  // Strategy is to run deconvolution on the entire channel and then pick out the ROI's we found above
142  fFFT->deconvolute(rawAdcLessPedVec, fSignalShaping->GetResponse(channel).getDeconvKernel(), fSignalShaping->ResponseTOffset(channel));
143 
144  std::vector<float> holder;
145 
146  for(size_t roiIdx = 0; roiIdx < roiVec.size(); roiIdx++)
147  {
148  const auto& roi = roiVec[roiIdx];
149 
150  // First up: copy out the relevent ADC bins into the ROI holder
151  size_t roiLen = roi.second - roi.first + 1;
152 
153  holder.resize(roiLen);
154 
155  std::copy(rawAdcLessPedVec.begin()+binOffset+roi.first, rawAdcLessPedVec.begin()+binOffset+roiLen, holder.begin());
156  if (applyNormFactor) std::transform(holder.begin(),holder.end(),holder.begin(), std::bind(std::multiplies<float>(),std::placeholders::_1,normFactor));
157 
158  // Get the truncated mean and rms
159  float truncMean;
160  int nTrunc;
161  int range;
162 
163  fWaveformTool.getTruncatedMean(holder, truncMean, nTrunc, range);
164 
165  std::transform(holder.begin(),holder.end(),holder.begin(), std::bind(std::minus<float>(),std::placeholders::_1,truncMean));
166 
167  // apply wire-by-wire calibration
168  if (fDodQdxCalib){
169  if(fdQdxCalib.find(channel) != fdQdxCalib.end()){
170  float constant = fdQdxCalib.at(channel);
171  //std::cout<<channel<<" "<<constant<<std::endl;
172  for (size_t iholder = 0; iholder < holder.size(); ++iholder){
173  holder[iholder] *= constant;
174  }
175  }
176  }
177 
178  // add the range into ROIVec
179  ROIVec.add_range(roi.first, std::move(holder));
180  }
181 
182  return;
183 }
