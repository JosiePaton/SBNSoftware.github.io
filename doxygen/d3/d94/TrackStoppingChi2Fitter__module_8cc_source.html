118 StoppingChi2Fit TrackStoppingChi2Fitter::RunFit(const anab::Calorimetry& calo) const
119 {
120 
121  std::vector<float> dEdxVec, resRangeVec;
122  // Fill the dEdx vs res range vectors, ignoring the first/last points
123  for (size_t i = 1; i < calo.dEdx().size() - 1; i++) {
124  const float thisdEdx(calo.dEdx()[i]);
125  const float thisResRange(calo.ResidualRange()[i]);
126  if (thisResRange > fFitRange || thisdEdx > fMaxdEdx)
127  continue;
128 
129  dEdxVec.push_back(thisdEdx);
130  resRangeVec.push_back(thisResRange);
131  }
132 
133  if (dEdxVec.size() != resRangeVec.size())
134  throw cet::exception("TrachStoppingChi2Fitter") << "dEdx and Res Range do not have same length: " << dEdxVec.size() << " and " << resRangeVec.size() << std::endl;
135 
136  if (dEdxVec.size() < fMinHits)
137  return StoppingChi2Fit();
138 
139  const auto graph(std::make_unique<TGraph>(dEdxVec.size(), &resRangeVec[0], &dEdxVec[0]));
140 
141  // Try and fit a flat polynomial
142  graph->Fit("pol0", "Q");
143  const TF1* polFit = graph->GetFunction("pol0");
144  const float pol0Chi2(polFit ? polFit->GetChisquare() : -5.f);
145  const float pol0Fit(polFit ? polFit->GetParameter(0) : -5.f);
146 
147  // Try to fit an exponential
148  graph->Fit("expo", "Q");
149  const TF1* expFit = graph->GetFunction("expo");
150  const float expChi2(expFit ? expFit->GetChisquare() : -5.f);
151 
152  return StoppingChi2Fit(pol0Chi2, expChi2, pol0Fit);
