320  /// Definition of a range along one dimension.
321  template <typename Data = double>
322  struct Range {
323  using Data_t = Data; ///< Numeric type for boundaries.
324  using Range_t = Range<Data>; ///< This range type.
325 
326  Data_t lower = 1.0; ///< Starting coordinate.
327  Data_t upper = 0.0; ///< Ending coordinate.
328 
329  /// Default constructor: empty range.
330  Range() = default;
331 
332  /// Constructor from lower and upper bounds.
333  Range(Data_t lower, Data_t upper, bool doSort = false)
334  : lower(lower), upper(upper)
335  { if (doSort) sort(); }
336 
337  /// Returns whether the range is empty.
338  bool isNull() const { return lower >= upper; }
339 
340  /// Returns the distance between upper and lower bounds.
341  Data_t length() const { return std::max(upper - lower, Data_t(0.0)); }
342 
343  /// Returns whether the specified value is within the range.
344  bool contains(Data_t v) const { return (v >= lower) && (v <= upper); }
345 
346  /// Returns whether the specified range overlaps this range.
347  bool overlaps(Range_t const& r) const;
348 
349  /// Returns a value that, added to v, makes it fall within a margin in
350  /// the range.
351  Data_t delta(Data_t v, Data_t margin = 0.0) const;
352 
353  /// Extends the range to include the specified point.
354  void extendToInclude(Data_t);
355 
356  /// Extends the range to include the specified point.
357  void extendToInclude(Range_t const& r);
358 
359  /// Shortens this range to include only points also in `r`.
360  void intersect(Range_t const& r);
361 
362  private:
363  /// Ensures order of boundaries. Corrupts invalid ranges.
364  void sort() { if (lower > upper) std::swap(lower, upper); }
