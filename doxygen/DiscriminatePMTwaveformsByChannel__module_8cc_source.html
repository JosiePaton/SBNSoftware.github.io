63 /**
64  * @brief Produces discriminated optical waveforms.
65  *
66  * This module produces a "discriminated waveform" for each optical detector
67  * channel and for a single set of discrimination thresholds.
68  * The thresholds are specified one per channel.
69  *
70  *
71  * Output data products
72  * =====================
73  *
74  * The output data products can be assigned an instance name
75  * (`OutputInstanceName` configuration parameter) if desired. They are:
76  *
77  * * `std::vector<icarus::trigger::OpticalTriggerGate::GateData_t>`,
78  * with one entry per optical channel, including _all_ optical channels
79  * (even when no signal above threshold is ever present); each gate object
80  * has as index in the vector the number of the optical channel;
81  * * an _art_ association, one trigger gate data to many optical waveforms,
82  * of each trigger gate data (as above) with all the optical waveforms
83  * which contributed to it; the order of the association pairs is the same
84  * as the one of the trigger gate data, i.e. by channel, and some trigger
85  * gate data entries may be not associated to any waveform and therefore
86  * they can not be listed in the association; association pairs within
87  * the same optical channel are sorted by optical waveform timestamp;
88  * the type of the association is
89  * `art::Assns<icarus::trigger::OpticalTriggerGate::GateData_t, raw::OpDetWaveform>`.
90  *
91  *
92  * Input data products
93  * ====================
94  *
95  * * `std::vector<raw::OpDetWaveform>`: a single waveform for each recorded
96  * optical detector activity; the activity belongs to a single channel, but
97  * there may be multiple waveforms on the same channel. The time stamp is
98  * expected to be from the
99  * @anchor DetectorClocksElectronicsTime "electronics time scale"
100  * and therefore expressed in microseconds.
101  *
102  *
103  * Service requirements
104  * ---------------------
105  *
106  * The following services are _required_:
107  *
108  * * an implementation of `detinfo::DetectorClocksService`
109  * * _art_ message facility
110  *
111  *
112  * Configuration parameters
113  * =========================
114  *
115  * There are several (too many?) ways to specify the discrimination thresholds.
116  * In the end, all PMT channels must be assigned a threshold: if at any time,
117  * on any event, a channel is found whose threshold was not set, an exception
118  * is thrown.
119  * All threshold specifications are relative to the baseline. There are
120  * currently four levels of threshold setting:
121  *
122  * 1. if `NChannels` (positive integer) and `DefaultThreshold` (integral number)
123  * are both specified, the first `NChannels` PMT channels are assigned the
124  * `DefaultThreshold`; note that they both need to be specified at the same
125  * time, and when they are, channels with ID equal or larger than
126  * `NChannels` are still not configured, i.e. they will not be assigned the
127  * `DefaultThreshold` and if encountered an exception will be thrown;
128  * 2. `ThresholdList` (list of integers) will define a threshold for each
129  * channel starting with ID `0` on; thresholds already set with the previous
130  * step will be overridden;
131  * 3. `ThresholdsFromPMTconfig` (input tag) points to a data product with
132  * a PMT configuration (`sbn::PMTconfiguration`) which contains
133  * information on each channel, including the thresholds; thresholds are set
134  * for each channel from the value returned by
135  * `sbn::V1730channelConfiguration::relativeThreshold()` for that channel;
136  * only channel configuration entries that have a offline channel ID
137  * (`sbn::V1730channelConfiguration::hasChannelID()`) are configured in this
138  * way; thresholds already set with the previous steps will be overridden;
139  * when using this parameter, `NChannels` is also mandatory and channels
140  * with ID greater or equal than `NChannels` will trigger an exception.
141  * 4. `Thresholds` (list of tables): in each entry, a threshold is specified for
142  * a single channel; the format of the entry is:
143  * * `Channel` (integral, mandatory): channel number;
144  * * `Threshold` (integral, mandatory): the threshold for that channel.
145  * .
146  * Thresholds already set with the previous steps will be overridden.
147  *
148  * All these are optional (but if the configuration has none, not much can be
149  * done since there will be no channel configured).
150  *
151  * A terse description of the remaining parameters follows.
152  *
153  * * `OpticalWaveforms` (input tag): the data product containing all optical
154  * detector waveforms.
155  * * `Baselines` (input tag): the data product containing one baseline per
156  * waveform in data product (from `OpticalWaveforms`).
157  * * `NChannels` (integral): if specified, only channels with ID from `0` to
158  * `NChannels - 1` are processed; waveforms from channels with higher ID will
159  * be ignored. If `DefaultThreshold` or `ThresholdsFromConfiguration`
160  * parameters are specified (see above), then `NChannels` is mandatory;
161  * otherwise, its value is deduced from the highest channel configured with
162  * one of the other threshold parameters.
163  * * `TriggerGateBuilder` (tool configuration): template configuration of the
164  * _art_ tool used to discriminate the optional waveforms; the tool interface
165  * is `icarus::trigger::TriggerGateBuilder`. Note that the threshold settings
166  * of this configuration is going to be ignored.
167  * * `OutputCategory` (string, default: `"DiscriminatePMTwaveformsByChannel"`):
168  * label for the category of messages in the console output; this is the label
169  * that can be used for filtering messages via MessageFacility service.
170  * configuration.
171  *
172  *
173  */

552 //------------------------------------------------------------------------------
553 void icarus::trigger::DiscriminatePMTwaveformsByChannel::produce(art::Event& event) {
554 
555  //
556  // set up the algorithm to create the trigger gates
557  //
558  detinfo::DetectorTimings const detTimings {
559  art::ServiceHandle<detinfo::DetectorClocksService const>()->DataFor(event)
560  };
561 
562  //
563  // fetch input
564  //
565  auto const& waveformHandle
566  = event.getValidHandle<std::vector<raw::OpDetWaveform>>(fOpDetWaveformTag);
567  auto const& waveforms = *waveformHandle;
568 
569  // map address of waveform to art pointer to that waveform
570  auto const& opDetWavePtrs
571  = util::mapDataProductPointers(event, waveformHandle);
572 
573  //
574  // retrieve the baseline information if provided event by event
575  //
576  std::vector<icarus::WaveformBaseline> baselines;
577  if (fBaselineTag) {
578  // read and assign from the data product; configured baselines are ignored
579 
580  baselines = event.getProduct<std::vector<icarus::WaveformBaseline>>
581  (fBaselineTag.value());
582 
583  } // if baselines from event
584  else {
585  // assign baselines from configuration
586  baselines.reserve(waveforms.size());
587  for (auto const& waveform: waveforms) {
588  auto const channelSlot
589  = static_cast<std::size_t>(waveform.ChannelNumber());
590  if (channelSlot >= fCurrentBaselines.size()) {
591  throw cet::exception("DiscriminatePMTwaveformsByChannel")
592  << "Waveforms on channel " << waveform.ChannelNumber()
593  << " have no baseline configured.\n";
594  }
595  baselines.emplace_back(fCurrentBaselines[channelSlot].value.value());
596  } // for all waveforms
597  } // if ... else
598 
599  {
600  mf::LogDebug log { fLogCategory };
601  log << "Discrimination of " << waveforms.size() << " PMT waveforms from '"
602  << fOpDetWaveformTag.encode() << "'";
603  if (fBaselineTag)
604  log << " with baselines from '" << fBaselineTag->encode() << "'";
605  }
606 
607 
608  // We use an algorithm designed for applying to all channels the same
609  // thresholds, and many of them. But here we have a different threshold per
610  // channel, and a single threshold in that.
611  // So we'll reconfigure the algorithm for each channel, and process only
612  // the waveforms (and all of them) on that channel
613  // (a possible alternative is to set all the thresholds that appear in any
614  // of the channels, and after processing pick the right one for each channel;
615  // this normally works well because the threshold values are actually very few
616  // (a nominal one, and a few exceptions) but it may not scale.
617 
618  // group the waveforms by channel; ignore the channels we'll not process:
619  // channel ID -> all input waveforms and baselines on that channel
620  std::vector<std::vector<icarus::trigger::WaveformWithBaseline>>
621  waveformInfoPerChannel(fNOpDetChannels);
622 
623  for (auto const& [ waveform, baseline ]: util::zip(waveforms, baselines)) {
624 
625  auto const channelSlot = static_cast<std::size_t>(waveform.ChannelNumber());
626  if (channelSlot >= waveformInfoPerChannel.size()) continue; // ignore
627 
628  waveformInfoPerChannel[channelSlot].emplace_back
629  (&waveform, &baseline);
630 
631  } // for all channels
632
