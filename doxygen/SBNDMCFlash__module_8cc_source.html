350 void SBNDMCFlash::GetFlashLocation(std::vector<double> pePerOpChannel,
351  double& Ycenter,
352  double& Zcenter,
353  double& Ywidth,
354  double& Zwidth)
355 {
356 
357  // Reset variables
358  Ycenter = Zcenter = 0.;
359  Ywidth = Zwidth = -999.;
360  double totalPE = 0.;
361  double sumy = 0., sumz = 0., sumy2 = 0., sumz2 = 0.;
362 
363  for (unsigned int opch = 0; opch < pePerOpChannel.size(); opch++) {
364 
365  // Get physical detector location for this opChannel
366  double PMTxyz[3];
367  ::art::ServiceHandle<geo::Geometry> geo;
368  geo->OpDetGeoFromOpChannel(opch).GetCenter(PMTxyz);
369 
370  // Add up the position, weighting with PEs
371  sumy += pePerOpChannel[opch]*PMTxyz[1];
372  sumy2 += pePerOpChannel[opch]*PMTxyz[1]*PMTxyz[1];
373  sumz += pePerOpChannel[opch]*PMTxyz[2];
374  sumz2 += pePerOpChannel[opch]*PMTxyz[2]*PMTxyz[2];
375 
376  totalPE += pePerOpChannel[opch];
377  }
378 
379  Ycenter = sumy/totalPE;
380  Zcenter = sumz/totalPE;
381 
382  // This is just sqrt(<x^2> - <x>^2)
383  if ( (sumy2*totalPE - sumy*sumy) > 0. )
384  Ywidth = std::sqrt(sumy2*totalPE - sumy*sumy)/totalPE;
385 
386  if ( (sumz2*totalPE - sumz*sumz) > 0. )
387  Zwidth = std::sqrt(sumz2*totalPE - sumz*sumz)/totalPE;
388 }
