148  for(unsigned int univIdx = 0; univIdx < Nuniv; ++univIdx){

150  }
151  }

163

165 
166  std::vector<std::vector<double>> coords_sub(Nuniv);
167 
168  for(unsigned int matSize = 1; matSize <= 500/*std::min(N, Nuniv)*/; ++matSize){
169  // Try adding each potential variable while holding the coefficients of
170  // the rest fixed. This is a fast way to estimate which variable will be
171  // the best when added to the full fit.
172  int bestVarIdx = -1;
173  double bestSqErr = std::numeric_limits<double>::infinity();
174 
175  for(unsigned int varIdx = 0; varIdx < N; ++varIdx){
176  if(already[varIdx]) continue;
177 
178  const std::vector<double>& c = coords[varIdx];
179 
180  double num = 0, denom = 0;
181 
182  for(unsigned int univIdx = 0; univIdx < Nuniv; ++univIdx){
183  num += residual[univIdx]*c[univIdx];
184  denom += util::sqr(c[univIdx]);
185  }
186 
187  // Best coefficient for this variable
188  const double x = num/denom;
189 
190  // Evaluate this solution
191  double sqErr = 0;
192  for(unsigned int univIdx = 0; univIdx < Nuniv; ++univIdx){
193  sqErr += util::sqr(residual[univIdx] - c[univIdx] * x);
194  }
195 
196  if(sqErr < bestSqErr){
197  bestSqErr = sqErr;
198  bestVarIdx = varIdx;
199  }
200  } // end for varIdx
201 
202  // std::cout << matSize << ": chose to add var " << bestVarIdx << " predicting MSE = " << sqrt(bestSqErr/Nuniv) << std::endl;
203 
204  // Now that it's certain, we can shorten the variable name
205  const int varIdx = bestVarIdx;
206  const std::vector<double>& c = coords[varIdx];
207 
208  // Make sure not to use it again
209  already[varIdx] = true;
210 
211  // Expand the problem with the new variable
212  vsub.push_back(v[varIdx]);
213  vars.push_back(varIdx);
214  icd.AddRow(M[varIdx], vars);
215 
216  // And solve it exactly
217  const std::vector<double> a = icd.Solve(vsub);
218 
219  double sqErr = 0;
220  for(unsigned int univIdx = 0; univIdx < Nuniv; ++univIdx){
221  std::vector<double>& cs = coords_sub[univIdx];
222  cs.push_back(c[univIdx]);
223 
224  // Update the residuals for the new solution
225  residual[univIdx] = ds[univIdx];
226  for(unsigned int i = 0; i < matSize; ++i){
227  residual[univIdx] -= a[i] * cs[i];
228  }
229 
230  // As a side effect, it's easy to compute the new squared error
231  sqErr += util::sqr(residual[univIdx]);
232  }
233 
234  // Can remove sqErr if we don't want this printout
235  // std::cout << " optimized by matrix to " << sqrt(sqErr/Nuniv) << std::endl;
236  } // end for matSize
237 
238  // Solve one last time, reuse vsub to store output
239  vsub = icd.Solve(vsub);
240 
241  // Package up results
242  std::vector<double> ret(N);
243  for(unsigned int i = 0; i < vars.size(); ++i) ret[vars[i]] = vsub[i];
