242  return false;

244

247 
248  assert(commonVersion >= 1);

251  mf::LogWarning("Geometry") << "Geometry::CompareConfigurationInfo(): "
252  "detector name mismatch: '" << A.detectorName << "' vs. '"
253  << B.detectorName << "'";
