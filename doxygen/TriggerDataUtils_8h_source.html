43  /**
44  * @brief Returns the trigger gates in serializable format.
45  * @tparam Gates type of the source of trigger gate data
46  * @param gates the data to be reformatted (*data will be stolen!*)
47  * @return gate data product
48  *
49  * This function transfers the data from the original structured `gates` into
50  * a data collection suitable for serialization with _art_, but
51  * *not including the association of each gate with all its contributing
52  * waveforms*.
53  * It must be stressed that this causes information loss, because the trigger
54  * gates will not be associated any more not only to the contributing
55  * waveforms, but also to the number of the optical detector channel(s).
56  * This happens because the trigger gates do not store that information,
57  * which is instead conveyed by the associated waveforms.
58  *
59  * The return value is a collection of trigger gate data
60  * (`icarus::trigger::OpticalTriggerGateData_t`), with data _stolen_
61  * from `gates`.
62  *
63  * The trigger gates are processed in the same order as they are in `gates`.
64  *
65  * After the function returns, `gates` will have been depleted of all the gate
66  * data; the waveform information will be still associated to each gate, whose
67  * gate data will be in an invalid state anyway, only good for destruction.
68  *
69  * The object `gates` is a iterable collection with each element required to
70  * respond to:
71  * * `waveforms()` returning a collection of pointers to the waveforms
72  * associated to the gate;
73  * * `gateLevels()` returning a moveable container with the data of the
74  * trigger gate.
75  * Collections of objects of type `icarus::trigger::OpticalTriggerGate` or
76  * derived from it (`icarus::trigger::SingleChannelOpticalTriggerGate`,
77  * `icarus::trigger::MultiChannelOpticalTriggerGate`) satisfy this
78  * requirement.
79  *
80  * In the following example, we start with trigger gates already serialized
81  * in a _art_ event. The serialization splits a trigger gate object in two
82  * components: the gate levels, and the associated waveforms.
83  * In the first part of the example we recover the information from the event
84  * and we assemble it into the standard trigger gate objects (of type
85  * `icarus::trigger::SingleChannelOpticalTriggerGate`).
86  * After some unspecified and optional processing, `gates` are disassembled
87  * to be saved into the event: this is achieved by a call to
88  * `transformIntoOpticalTriggerGate()` which produces the trigger gate data
89  * and their associations to the waveforms.
90  * In the last part, these components are both stored into the event.
91  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~{.cpp}
92  * using icarus::trigger::OpticalTriggerGateData_t; // for convenience
93  *
94  * //
95  * // somehow get/build a collection of trigger gates;
96  * // here we use `FillTriggerGates()` to read existing data from the event
97  * //
98  * auto gates = icarus::trigger::FillTriggerGates<icarus::trigger::SingleChannelOpticalTriggerGate>
99  * (
100  * *(event.getValidHandle<std::vector<OpticalTriggerGateData_t>>("orig")),
101  * *(event.getValidHandle<art::Assns<OpticalTriggerGateData_t, raw::OpDetWaveform>>("orig"))
102  * );
103  *
104  *
105  * // ...
106  *
107  * //
108  * // use the created vector and associations (e.g. put them into art event)
109  * //
110  * event.put(std::make_unique<std::vector<TriggerGateData_t>>
111  * (icarus::trigger::transformIntoOpticalTriggerGate(std::move(gates))));
112  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
113  * Also note that in the omitted processing part the trigger gates might have
114  * been combined into a different type of gates instead, e.g. into a
115  * collection of `icarus::trigger::MultiChannelOpticalTriggerGate`.
116  * To serialize that collection, the exact same procedure would be used.
117  * Once more, remember that the modules reading those newly put trigger gates
118  * will now know the channel or channels associated with them.
119  */
120  template <typename Gates>
121  std::vector<icarus::trigger::OpticalTriggerGateData_t>
122  transformIntoOpticalTriggerGate(Gates&& gates);

126  /**
127  * @brief Returns the trigger gates in serializable format.
128  * @tparam Gates type of the source of trigger gate data
129  * @param gates the data to be reformatted (*data will be stolen!*)
130  * @param makeGatePtr _art_ pointer maker for the gate data
131  * @param opDetWavePtrs map of art pointers to optical waveforms
132  * @return a pair: gate data product and associations to optical waveforms
133  *
134  * This function transfers the data from the original structured `gates` into
135  * a data collection suitable for serialization with _art_, including the
136  * association of each gate with all its contributing waveforms.
137  *
138  * The return value is a tuple of two elements:
139  * * `0`: collection of trigger gate data
140  * (`icarus::trigger::OpticalTriggerGateData_t`), with data _stolen_
141  * from `gates`;
142  * * `1`: association between trigger gate data and their optical waveforms.
143  *
144  * The trigger gates are processed in the same order as they are in `gates`,
145  * and the associations to the waveforms are set gate by gate, in the same
146  * order as they are reported by the `waveforms()` method of the input gate.
147  *
148  * After the function returns, `gates` will have been depleted of all the data
149  * and left in an undefined state only good for destruction.
150  *
151  * The object `gates` is a iterable collection with each element required to
152  * respond to:
153  * * `waveforms()` returning a collection of pointers to the waveforms
154  * associated to the gate;
155  * * `gateLevels()` returning a moveable container with the data of the
156  * trigger gate.
157  * Collections of objects of type `icarus::trigger::OpticalTriggerGate` or
158  * derived from it (`icarus::trigger::SingleChannelOpticalTriggerGate`,
159  * `icarus::trigger::MultiChannelOpticalTriggerGate`) satisfy this
160  * requirement.
161  *
162  *
163  * In the following example, we start with trigger gates already serialized
164  * in a _art_ event. The serialization splits a trigger gate object in two
165  * components: the gate levels, and the associated waveforms.
166  * In the first part of the example we recover the information from the event
167  * and we assemble it into the standard trigger gate objects (of type
168  * `icarus::trigger::SingleChannelOpticalTriggerGate`).
169  * After some unspecified and optional processing, `gates` are disassembled
170  * to be saved into the event: this is achieved by a call to
171  * `transformIntoOpticalTriggerGate()` which produces the trigger gate data
172  * and their associations to the waveforms.
173  * In the last part, these components are both stored into the event.
174  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~{.cpp}
175  * using icarus::trigger::OpticalTriggerGateData_t; // for convenience
176  *
177  * //
178  * // somehow get/build a collection of trigger gates;
179  * // here we use `FillTriggerGates()` to read existing data from the event
180  * //
181  * auto gates = icarus::trigger::FillTriggerGates<icarus::trigger::SingleChannelOpticalTriggerGate>
182  * (
183  * *(event.getValidHandle<std::vector<OpticalTriggerGateData_t>>("orig")),
184  * *(event.getValidHandle<art::Assns<OpticalTriggerGateData_t, raw::OpDetWaveform>>("orig"))
185  * );
186  *
187  *
188  * // ...
189  *
190  * //
191  * // optical waveform to pointer map is required to create associations
192  * // between the trigger gates and their waveforms
193  * //
194  * auto const& opDetWavePtrs = util::mapDataProductPointers
195  * (event, event.getValidHandle<std::vector<raw::OpDetWaveform>>("opdaq"));
196  * // transform the data; after this line, `gates` is not usable any more
197  * auto thresholdData = icarus::trigger::transformIntoOpticalTriggerGate
198  * (std::move(gates), makeGatePtr, opDetWavePtrs);
199  *
200  * //
201  * // use the created vector and associations (e.g. put them into art event)
202  * //
203  * event.put(
204  * std::make_unique<std::vector<TriggerGateData_t>>
205  * (std::move(std::get<0U>(thresholdData)))
206  * );
207  * event.put(
208  * std::make_unique<art::Assns<TriggerGateData_t, raw::OpDetWaveform>>
209  * (std::move(std::get<1U>(thresholdData)))
210  * );
211  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
212  * Also note that in the omitted processing part the trigger gates might have
213  * been combined into a different type of gates instead, e.g. into a
214  * collection of `icarus::trigger::MultiChannelOpticalTriggerGate`.
215  * To serialize that collection, the exact same procedure would be used,
216  * with the different outcome that now each trigger gate may be associated
217  * to waveforms from different optical detector channels.
218  *
219  */

233  /**
234  * @brief Creates a gate object out of trigger gate data products.
235  * @tparam GateObject type of gate object to create
236  * @tparam TriggerGateData the type of data stored
237  * @param gates collection of data from all the gates
238  * @param gateToWaveforms association between each gate and its waveforms
239  * @return a STL vector of `GateObject` with copy of data from `gates`
240  *
241  * Objects like `icarus::trigger::OpticalTriggerGate` are complex enough that
242  * they are not saved directly into an _art_ event. Rather, they are diced
243  * into pieces and the pieces are stored.
244  * This function stitches the pieces and returns back an object like
245  * `icarus::trigger::OpticalTriggerGate`.
246  *
247  * The supported types for `GateObject` need to expose a
248  * `icarus::trigger::OpticalTriggerGate`-like interface (including also e.g.
249  * `icarus::trigger::SingleChannelOpticalTriggerGate` and
250  * `icarus::trigger::MultiChannelOpticalTriggerGate`).
251  * They must accept:
252  * * to be constructed with a `raw::OpDetWaveform` reference;
253  * * to be added a `raw::OpDetWaveform` reference via `add()` call;
254  * * to be assigned a `TriggerGateData` type.
255  *
256  * Example:
257  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~{.cpp}
258  * icarus::trigger::TriggerGateBuilder::TriggerGates readTriggerGates(
259  * art::Event const& event,
260  * icarus::trigger::ADCCounts_t const threshold,
261  * art::InputTag const& dataTag
262  * ) {
263  *
264  * auto const& gates
265  * = *(event.getValidHandle<std::vector<TriggerGateData_t>>(dataTag));
266  * auto const& gateToWaveforms = *(
267  * event.getValidHandle<art::Assns<TriggerGateData_t, raw::OpDetWaveform>>
268  * (dataTag)
269  * );
270  * return {
271  * threshold,
272  * icarus::trigger::FillTriggerGates
273  * <icarus::trigger::SingleChannelOpticalTriggerGate>
274  * (gates, gateToWaveforms)
275  * };
276  *
277  * } // readTriggerGates()
278  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
279  * will return a `icarus::trigger::TriggerGateBuilder::TriggerGates` object
280  * for the specified threshold, reading the information from `dateTag` data
281  * products in an _art_ `event`.
282  *
283  *
284  * Return value
285  * -------------
286  *
287  * The returned collection contains one element for each `TriggerGateData`
288  * object in `gates`, in the same order. Each of these elements is of type
289  * `GateObject`, contains a copy of the data of the corresponding gate,
290  * and a list of optical waveforms (`raw::OpDetWaveform` objects) it is
291  * associated to.
292  *
293  *
294  * Requirements
295  * -------------
296  *
297  * The requirements bind the gates to their association to waveforms:
298  * * each gate must be associated to at least one waveform
299  * * the associations must be grouped so that all the association pairs
300  * pertaining a gate are contiguous
301  * * within each of these groups, which is made of at least one waveform,
302  * the waveforms must be ordered by increasing timestamp
303  * * the groups must be in the same order as their associated gates
304  * This constitutes the requirement of
305  * @ref LArSoftProxyDefinitionOneToManySeqAssn "one-to-many sequential association"
306  * with the addition that each element in `gates` must have at least one
307  * associated waveform.
308  *
309  */
310  template <typename GateObject, typename TriggerGateData>
311  std::vector<GateObject> FillTriggerGates(
312  std::vector<TriggerGateData> const& gates,
313  art::Assns<TriggerGateData, raw::OpDetWaveform> const& gateToWaveforms
314  );

371 template <typename Gates>
372 std::vector<icarus::trigger::OpticalTriggerGateData_t>
373 icarus::trigger::transformIntoOpticalTriggerGate(Gates&& gates) {
374  using TriggerGateData_t = icarus::trigger::OpticalTriggerGateData_t;
375 
376  //
377  // create objects for the data products
378  //
379  std::vector<TriggerGateData_t> gateData;
380 
381  for (auto& gate: gates) {
382  assert(gate.hasChannels());
383 
384  // we steal the data from the gate
385  gateData.push_back(std::move(gate.gateLevels()));
386 
387  // now we add all channels;
388  // the cast is a cheat for objects like `SingleChannelOpticalTriggerGate`
389  // which hide the `channels()` method
390  auto& newGate = gateData.back();
391  for (auto channel: static_cast<TriggerGateData_t const&>(gate).channels())
392  newGate.addChannel(channel);
393 
394  } // for all channels
395 
396  return gateData;
397 
398 } // icarus::trigger::transformIntoOpticalTriggerGate()

403 std::tuple<
404  std::vector<icarus::trigger::OpticalTriggerGateData_t>,
405  art::Assns<icarus::trigger::OpticalTriggerGateData_t, raw::OpDetWaveform>
406  >
407 icarus::trigger::transformIntoOpticalTriggerGate(
408  Gates&& gates,
409  art::PtrMaker<icarus::trigger::OpticalTriggerGateData_t> const& makeGatePtr,
410  OpDetWaveformDataProductMap_t const& opDetWavePtrs
411  )
412 {
413  using TriggerGateData_t = icarus::trigger::OpticalTriggerGateData_t;
414 
415  //
416  // create objects for the data products
417  //
418  std::vector<TriggerGateData_t> gateData
419  = transformIntoOpticalTriggerGate(std::move(gates));
420 
421  // note that we rely on the fact that `gates` still contains trigger gates,
422  // and those trigger gates still know about the associated waveforms;
423  // attempting to access their levels, though, would be an error
424  // (if really needed, we can still find them in `gateData`)
425  art::Assns<TriggerGateData_t, raw::OpDetWaveform> gateToWaveforms;
426 
427  for (auto const& [ iGate, gate ]: util::enumerate(gates)) {
428 
429  if (gate.waveforms().empty()) continue;
430 
431  //
432  // produce the associations
433  //
434 
435  // pointer to the gate data we have just added:
436  art::Ptr<TriggerGateData_t> const& gatePtr = makeGatePtr(iGate);
437  for (raw::OpDetWaveform const* waveform: gate.waveforms()) {
438 
439  gateToWaveforms.addSingle(gatePtr, opDetWavePtrs.at(waveform));
440 
441  } // for waveforms
442 
443  } // for all channels
444 
445  return { std::move(gateData), std::move(gateToWaveforms) };
446

451 template <typename GateObject, typename TriggerGateData>
452 std::vector<GateObject> icarus::trigger::FillTriggerGates(
453  std::vector<TriggerGateData> const& gates,
454  art::Assns<TriggerGateData, raw::OpDetWaveform> const& gateToWaveforms
455 ) {
456  using GateData_t = GateObject; ///< Type of object being returned,
457 
458  std::vector<GateData_t> allGates;
459 
460  auto iGateToWaveform = gateToWaveforms.begin();
461  auto const gwend = gateToWaveforms.end();
462 
463  for (auto const& [ iGate, gate ]: util::enumerate(gates)) {
464 
465  //
466  // find the relevant waveforms for this gate;
467  // match by index of the element in the data product collection
468  //
469  while (iGateToWaveform != gwend) {
470  if (iGateToWaveform->first.key() == iGate) break;
471  ++iGateToWaveform;
472  } // while
473  auto const iFirstWaveform = iGateToWaveform;
474  while (iGateToWaveform != gwend) {
475  if (iGateToWaveform->first.key() != iGate) break;
476  ++iGateToWaveform;
477  } // while
478 
479  typename GateData_t::ChannelID_t firstChannel;
480 
481  if constexpr(icarus::trigger::isReadoutTriggerGate_v<decltype(gate)>) {
482  assert(gate.hasChannels());
483  firstChannel = *(begin(gate.channels()));
484  }
485  else {
486  // input has no channels, and we need to set at least one
487  // (by requirement; the object could live without); but...
488  if (iFirstWaveform == iGateToWaveform) {
489  throw cet::exception("FillTriggerGates")
490  << " Could not find any waveform associated to trigger gate #" << iGate
491  << ", needed to assign it channel numbers.\n";
492  }
493  firstChannel = iFirstWaveform->second->ChannelNumber();
494  }
495 
496  /*
497  * this is quite crazy code attempting to serve many different objects;
498  * the constructor that at this time all these objects share takes a single
499  * channel number, and we use it here;
500  * copy constructors are complicate (work only as in `base = derived` and
501  * most often we go the other way around here).
502  * The effect of the assignment below is also not so-well defined,
503  * but it is guaranteed to copy the gate data; it _might_ copy channels too,
504  * which at this point we do not consider harmful.
505  */
506  GateData_t gateData(firstChannel);
507 
508  gateData = gate; // copy data and possibly more
509 
510  // add all the waveforms; this also registers their channels,
511  // and skips duplicate waveforms (and channels)
512  auto iEndWaveform = iFirstWaveform;
513  while (iEndWaveform != iGateToWaveform)
514  gateData.add(*(iEndWaveform++)->second);
515 
516  allGates.push_back(std::move(gateData));
517 
518  } // for gates
519 
520  return allGates;
521 } // icarus::trigger::FillTriggerGates()
