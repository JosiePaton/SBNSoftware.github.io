100 uint64_t crt::DecoderICARUSCRT::CalculateTimestamp(icarus::crt::BernCRTTranslator& hit) {
101  /**
102  * Calculate timestamp based on nanosecond from FEB and poll times measured by server
103  * see: https://sbn-docdb.fnal.gov/cgi-bin/private/DisplayMeeting?sessionid=7783
104  */
105  int32_t ts0 = hit.ts0; //must be signed int
106 
107  //add PPS cable length offset modulo 1s
108  ts0 = (ts0 + FEB_delay.at(hit.mac5)) % (1'000'000'000);
109  if(ts0 < 0) ts0 += 1000'000'000; //just in case the cable offset is negative (should be positive normally)
110 
111  uint64_t mean_poll_time = hit.last_poll_start/2 + hit.this_poll_end/2;
112  int mean_poll_time_ns = mean_poll_time % (1000'000'000);
113 
114  return mean_poll_time - mean_poll_time_ns + ts0
115  + (ts0 - mean_poll_time_ns < -500'000'000) * 1000'000'000
116  - (ts0 - mean_poll_time_ns > 500'000'000) * 1000'000'000;
117 }
