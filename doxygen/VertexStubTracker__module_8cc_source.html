5 // Author: grayputnam@uchicago.edu
6 //
7 // Art module for building "Stub" reconstruction objects. Stubs are
8 // correspond to low energy hadrons produced in neutrino interactions that
9 // produce short, highly-ionizing charge depositions near the vertex.
10 //
11 // Module operates on sbn::VertexHit objects and connects the end-point Hit
12 // to the start-point Vertex to build a stub. Merging is then done between
13 // and across planes to consoldiate and refine output. Produces the sbn::Stub
14 // object.

114 void sbn::VertexStubTracker::produce(art::Event& e)
115 {
116  // collect services
117  const geo::GeometryCore *geo = lar::providerFrom<geo::Geometry>();
118  auto const clock_data = art::ServiceHandle<detinfo::DetectorClocksService const>()->DataFor(e);
119  auto const dprop =
120  art::ServiceHandle<detinfo::DetectorPropertiesService const>()->DataFor(e, clock_data);
121  const spacecharge::SpaceCharge *sce = lar::providerFrom<spacecharge::SpaceChargeService>();
122  // If we are not correcting for SCE, then blank out the service
123  if (!fCorrectSCE) sce = nullptr;
124 
125  // output data products
126  std::unique_ptr<std::vector<sbn::Stub>> outStubs(new std::vector<sbn::Stub>);
127  std::unique_ptr<art::Assns<sbn::VertexHit, sbn::Stub>> assn(new art::Assns<sbn::VertexHit, sbn::Stub>);
128  std::unique_ptr<art::Assns<sbn::Stub, recob::Hit>> hitAssn(new art::Assns<sbn::Stub, recob::Hit>);
129  std::unique_ptr<art::Assns<sbn::Stub, recob::Slice>> slcAssn(new art::Assns<sbn::Stub, recob::Slice>);
130  std::unique_ptr<art::Assns<sbn::Stub, recob::PFParticle>> pfpAssn(new art::Assns<sbn::Stub, recob::PFParticle>);
131 
132  art::PtrMaker<sbn::Stub> StubPtrMaker {e};
133 
134  // input data
135  art::Handle<std::vector<recob::Slice>> slice_handle;
136  e.getByLabel(fPFPLabel, slice_handle);
137 
138  std::vector<art::Ptr<recob::Slice>> slices;
139  art::fill_ptr_vector(slices, slice_handle);
140 
141  art::FindManyP<sbn::VertexHit> slcVHits(slices, e, fVertexChargeLabel);
142 
143  // Setup the stub builder
144  fStubBuilder.Setup(e, fPFPLabel, fTrackLabel);
145 
146  for (unsigned i_slc = 0; i_slc < slices.size(); i_slc++) {
147  const std::vector<art::Ptr<sbn::VertexHit>> &vhits = slcVHits.at(i_slc);
148 
149  // look up info per vertex hit
150  art::FindManyP<recob::Vertex> vhitVtxs(vhits, e, fVertexChargeLabel);
151  art::FindManyP<recob::Hit> vhitHits(vhits, e, fVertexChargeLabel);
152 
153  // stuff common to each vertex hit in a slice
154  art::Ptr<recob::Slice> thisSlice = slices[i_slc];
155 
156  std::vector<sbn::StubInfo> stubs;
157  for (unsigned i_vhit = 0; i_vhit < vhits.size(); i_vhit++) {
158  // Collect data on this Vertex-Hit
159  const sbn::VertexHit &thisVHit = *vhits[i_vhit];
160  const recob::Hit &thisVHitHit = *vhitHits.at(i_vhit).at(0);
161 
162  bool passcut = (thisVHit.dqdx >= fdQdxCut) || ((abs(thisVHit.wire.Wire - thisVHit.vtxw) < 1.) && (thisVHit.dqdx >= fOneWiredQdxCut));
163 
164  if (!passcut) continue;
165 
166  sbn::StubInfo sinfo;
167  sinfo.stub = fStubBuilder.FromVertexHit(thisSlice, thisVHit, thisVHitHit, geo, sce, clock_data, dprop, sinfo.hits, sinfo.pfp);
168  sinfo.vhit = vhits[i_vhit];
169  sinfo.vhit_hit = vhitHits.at(i_vhit).at(0);
170 
171  stubs.push_back(sinfo);
172 
173  } // end iterate over vertex hits
174 
175  // Run all of the merging tools
176  for (unsigned i_mrg = 0; i_mrg < fStubMergeTools.size(); i_mrg++) {
177  stubs = fStubMergeTools[i_mrg]->Merge(stubs, geo, sce, clock_data, dprop);
178  }
179 
180  // Save!
181  for (unsigned i_stub = 0; i_stub < stubs.size(); i_stub++) {
182  const sbn::StubInfo &sinfo = stubs[i_stub];
183 
184  outStubs->push_back(sinfo.stub);
185  art::Ptr<sbn::Stub> outStub = StubPtrMaker(outStubs->size() - 1);
186  assn->addSingle(sinfo.vhit, outStub);
187  for (unsigned i_hit = 0; i_hit < sinfo.hits.size(); i_hit++) {
188  hitAssn->addSingle(outStub, sinfo.hits[i_hit]);
189  }
190  slcAssn->addSingle(outStub, thisSlice);
191  if (sinfo.pfp) {
192  pfpAssn->addSingle(outStub, sinfo.pfp);
193  }
194  } // end loop over stubs
195 
196  } // end loop over slices
197  // Save into event
198  e.put(std::move(outStubs));
199  e.put(std::move(assn));
200  e.put(std::move(hitAssn));
201  e.put(std::move(slcAssn));
202  e.put(std::move(pfpAssn));
203 }
