129  }
130  }
131  }

136  double tphoton = (CLHEP::RandExponential::shoot(fEngine, fParams.DecayTXArapucaVIS)) + simphotons[i].Time - t_min + fTimeTPB->fire();
137  if(tphoton < 0.) continue; // discard if it didn't made it to the acquisition

323  void DigiArapucaSBNDAlg::AddDarkNoise(std::vector<double>& wave,std::vector<double>& WaveformSP)
324  {
325  int nCT;
326  // Multiply by 10^9 since fDarkNoiseRate is in Hz (conversion from s to ns)
327  double mean = 1000000000.0 / fParams.DarkNoiseRate;
328  double darkNoiseTime = CLHEP::RandExponential::shoot(fEngine, mean);
329  while(darkNoiseTime < wave.size()) {

334  // Find next time to add dark noise
