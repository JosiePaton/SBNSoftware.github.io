54 /**
55  * @brief Combines discriminated PMT outputs into V1730 LVDS gates.
56  *
57  * This module simulates the combination of discriminated PMT outputs
58  * by the V1730 board LVDS output.
59  *
60  *
61  * Input data products
62  * ====================
63  *
64  * * `std::vector<icarus::trigger::OpticalTriggerGateData_t>` (labels out of
65  * `TriggerGatesTag` and `Thresholds`): full sets of discriminated waveforms,
66  * each waveform possibly covering multiple optical channels,
67  * and their associations to optical waveforms. One set per threshold.
68  *
69  *
70  * Requirements
71  * -------------
72  *
73  * The algorithms require that the list of discriminated waveforms has in each
74  * position the waveform from the optical detector channel number matching the
75  * position (e.g. entry `0` will be the discriminated waveform from channel 0,
76  * and so forth). No gaps are allowed: if no signal is present above threshold
77  * for a certain channel, a discriminated waveform should appear for that
78  * channel, with a gate always closed.
79  *
80  *
81  * Output data products
82  * =====================
83  *
84  * * `std::vector<icarus::trigger::OpticalTriggerGateData_t>` (instance name:
85  * concatenation of input module label and instance name of the input gates;
86  * the former is omitted if it is `TriggerGatesTag`): sets of gates combined
87  * according to the configuration; one set per input threshold.
88  * * `art::Assns<icarus::trigger::OpticalTriggerGateData_t, raw::OpDetWaveform>`
89  * (instance name: same as above): associations between each
90  * produced gate and the optical waveforms providing the original data.
91  *
92  *
93  * Configuration parameters
94  * =========================
95  *
96  * A terse description of the parameters is printed by running
97  * `lar --print-description LVDSgates`.
98  *
99  * * `TriggerGatesTag` (string, default: `discrimopdaq`): name of the module
100  * instance which produced the discriminated waveforms; it must not include
101  * any instance name, as the instance names will be automatically added from
102  * `Thresholds` parameter. This value is used as module name for all
103  * thresholds which do not specify one.
104  * * `Thresholds` (list of tags, mandatory): list of the discrimination
105  * thresholds to consider. A data product containing a digital signal is
106  * read for each one of the thresholds; each threshold entry is expected
107  * to be a input tag in the form `"[ModuleLabel:]InstanceName"`, and if the
108  * module label part is not specified, the one from `TriggerGatesTag` is
109  * used as default. In this way, thresholds can be specified simply by a
110  * number, e.g. `[ 200, 400, 600 ]`, which will be translated into e.g.
111  * `"discrimopdaq:200"`, `"discrimopdaq:400"` and `"discrimopdaq:600"`,
112  * while a full specification can be also used: `[ 200, "discrim:config" ]`
113  * will turn into `"discrimopdaq:200"` and `"discrim:config"` (to specify
114  * a label with no instance name, set it ending with a colon, like in
115  * `"discrim:"`).
116  * * `ChannelPairing` (list of integral number lists): channels to combine;
117  * each element of this list is itself a list of optical detector channel
118  * numbers. The channels within each group are combined according to
119  * `CombinationMode`, resulting in a single combined gate associated to
120  * all the channels in the group.
121  * All optical detector channels *must* appear in the configuration, and
122  * each channel can be combined only in one group. Groups must not be empty.
123  * * `IgnoreChannels` (list of integral numbers, optional): ID of the optical
124  * detector channels to skip.
125  * * `CombinationMode` (either: `"disable"`, `"input1"`, `"input2"`, `"AND"`
126  * or `"OR"`): for each group of channels defined in `ChannelPairing`,
127  * the selected operation is used to combine all the channels in the group:
128  * * `disable`: all channels are discarded, and the resulting gates are
129  * all closed all the way;
130  * * `input1`: the first channel in the pairing is kept, and the others
131  * are discarded;
132  * * `input2`: the second channel in the pairing is kept, and the others
133  * are discarded;
134  * * `AND`: all channels are combined in a AND, resulting in a gate open
135  * only at the time when all the channel gates in the group are open;
136  * if there is a single channel in the group, the combined gate is a
137  * copy of it;
138  * * `OR`: all channels are combined in a OR, resulting in a gate open
139  * at every time when any of the channel gates in the group is open;
140  * if there is a single channel in the group, the combined gate is a
141  * copy of it.
142  * All pairings undergo the same operation (it is not possible, for example,
143  * disabling only one group of channels and combining the other groups with
144  * a `AND`).
145  * * `LogCategory` (string): name of the output stream category for console
146  * messages (managed by MessageFacility library).
147  *
148  *
149  *
150  */

631  std::set<raw::Channel_t> configuredChannels, duplicateChannels;
632  for (auto const& pairing: fChannelPairing) {
633  if (pairing.empty()) {
634  ++nEmptyGroups;
635  continue;
636  }
637  for (raw::Channel_t channel: pairing) {
638  if (!configuredChannels.insert(channel).second)
639  duplicateChannels.insert(channel);
640  } // for channel
641  } // for pairing
642 
643  // collect invalid channel numbers
644  std::vector<raw::Channel_t> invalidChannels;
645  for (raw::Channel_t channel: configuredChannels) {
646  if (!geom.IsValidOpChannel(channel)) invalidChannels.push_back(channel);
647  } // for configured channels
648 
649  // collect missing channels
650  auto const endChannel = geom.MaxOpChannel() + 1U;
651  std::vector<raw::Channel_t> missingChannels;
652  for (auto channel: util::counter<raw::Channel_t>(endChannel)) {
653  if (!geom.IsValidOpChannel(channel)) continue;
654  if (configuredChannels.count(channel) > 0U) continue;
655  missingChannels.push_back(channel);
656  } // for channel
657 
658  // no error? good we are
659  if (invalidChannels.empty()
660  && missingChannels.empty()
661  && duplicateChannels.empty()
662  && (nEmptyGroups == 0U)
663  ) {
664  return configuredChannels.size();
665  }
666 
667  // error? good we are not
668  art::Exception e(art::errors::Configuration);
669  e << "The pairing configuration of " << configuredChannels.size()
670  << " optical channels in `ChannelPairing` presents the following errors.\n";
671  if (!invalidChannels.empty()) {
672  auto iChannel = invalidChannels.cbegin();
673  auto const cend = invalidChannels.cend();
674  e
675  << invalidChannels.size()
676  << " channel numbers do not represent valid channels: " << *iChannel;
677  while (++iChannel != cend) e << ", " << *iChannel;
678  e << ".\n";
679  } // if invalid channels
680 
681  if (!missingChannels.empty()) {
682  auto iChannel = missingChannels.cbegin();
683  auto const cend = missingChannels.cend();
684  e
685  << missingChannels.size()
686  << " channels do not appear in the configuration: " << *iChannel;
687  while (++iChannel != cend) e << ", " << *iChannel;
688  e << ".\n";
689  } // if missing channels
690 
691  if (!duplicateChannels.empty()) {
692  auto iChannel = duplicateChannels.cbegin();
693  auto const cend = duplicateChannels.cend();
694  e
695  << duplicateChannels.size()
696  << " channels are specified more than once: " << *iChannel;
697  while (++iChannel != cend) e << ", " << *iChannel;
698  e << ".\n";
699  } // if duplicate channels
700 
701  if (nEmptyGroups > 0U) {
702  e << nEmptyGroups << " channel groups are empty.";
703  }
704 
705  throw e;
706 } // icarus::trigger::LVDSgates::checkPairings()

709 //------------------------------------------------------------------------------
710 void icarus::trigger::LVDSgates::checkInput
711  (std::vector<icarus::trigger::SingleChannelOpticalTriggerGate> const& gates)
712  const
713 {
714  /*
715  * check that
716  * * each gate has a single channel
717  * * each gate is in the position matching its channel number
718  */
719  for (auto const& [ iGate, gate ]: util::enumerate(gates)) {
720 
721  auto const expectedChannel = static_cast<raw::Channel_t>(iGate);
722 
723  if (gate.nChannels() != 1U) {
724  cet::exception e("LVDSgates");
725  e << "Requirement failure: gate #" << iGate << " includes ";
726  if (!gate.hasChannels())
727  e << "no channels";
728  else {
729  // we are cheating here, because `SingleChannelOpticalTriggerGate`
730  // actually does not want more than one channel;
731  // having them is sort of a logic error
732  auto const& channels = gate.OpticalTriggerGate::channels();
733  auto iChannel = channels.begin();
734  auto const cend = channels.end();
735  e << gate.nChannels() << " channels (" << *iChannel;
736  while (++iChannel != cend) e << ", " << *iChannel;
737  e << ")";
738  }
739  e << " should have exactly one channel (" << expectedChannel << ").\n";
740  throw e;
741  }
742 
743  if (gate.channel() != expectedChannel) {
744  throw cet::exception("LVDSgates")
745  << "Requirement failure: gate #" << iGate << " gate covers channel "
746  << gate.channel() << ", should cover " << expectedChannel
747  << " instead.\n";
748  }
749 
750  } // for
751 
752  MF_LOG_TRACE(fLogCategory)
753  << "LVDSgates[" << moduleDescription().moduleLabel() << "]: input checked.";
754
