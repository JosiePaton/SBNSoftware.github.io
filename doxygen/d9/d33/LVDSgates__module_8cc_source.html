56 /**
57  * @brief Combines discriminated PMT outputs into V1730 LVDS gates.
58  *
59  * This module simulates the combination of discriminated PMT outputs
60  * by the V1730 board LVDS output.
61  *
62  *
63  * Input data products
64  * ====================
65  *
66  * * `std::vector<icarus::trigger::OpticalTriggerGateData_t>` (labels out of
67  * `TriggerGatesTag` and `Thresholds`): full sets of discriminated waveforms,
68  * each waveform possibly covering multiple optical channels,
69  * and their associations to optical waveform metadata. One set per threshold.
70  *
71  *
72  * Requirements
73  * -------------
74  *
75  * The algorithms require that the list of discriminated waveforms has in each
76  * position the waveform from the optical detector channel number matching the
77  * position (e.g. entry `0` will be the discriminated waveform from channel 0,
78  * and so forth). No gaps are allowed: if no signal is present above threshold
79  * for a certain channel, a discriminated waveform should appear for that
80  * channel, with a gate always closed.
81  *
82  *
83  * Output data products
84  * =====================
85  *
86  * * `std::vector<icarus::trigger::OpticalTriggerGateData_t>` (instance name:
87  * concatenation of input module label and instance name of the input gates;
88  * the former is omitted if it is `TriggerGatesTag`): sets of gates combined
89  * according to the configuration; one set per input threshold.
90  * * `art::Assns<icarus::trigger::OpticalTriggerGateData_t, sbn::OpDetWaveformMeta>`
91  * (instance name: same as above): associations between each
92  * produced gate and the metadata of the optical waveforms providing the
93  * original data.
94  * * `art::Assns<icarus::trigger::OpticalTriggerGateData_t, raw::OpDetWaveform>`
95  * (instance name: same as above; optional): associations between each
96  * produced gate and the optical waveforms providing the original data.
97  * It is produced only if `ProduceWaveformAssns` configuration parameter is
98  * `true`, and it relies on the assumption that there is an association
99  * available between each `sbn::OpDetWaveformMeta` and its
100  * `raw::OpDetWaveform`, produced by the same module (i.e. with the same input
101  * tag) as the one of the original `sbn::OpDetWaveformMeta` data product
102  * itself.
103  *
104  *
105  * Configuration parameters
106  * =========================
107  *
108  * A terse description of the parameters is printed by running
109  * `lar --print-description LVDSgates`.
110  *
111  * * `TriggerGatesTag` (string, default: `discrimopdaq`): name of the module
112  * instance which produced the discriminated waveforms; it must not include
113  * any instance name, as the instance names will be automatically added from
114  * `Thresholds` parameter. This value is used as module name for all
115  * thresholds which do not specify one.
116  * * `Thresholds` (list of tags, mandatory): list of the discrimination
117  * thresholds to consider. A data product containing a digital signal is
118  * read for each one of the thresholds; each threshold entry is expected
119  * to be a input tag in the form `"[ModuleLabel:]InstanceName"`, and if the
120  * module label part is not specified, the one from `TriggerGatesTag` is
121  * used as default. In this way, thresholds can be specified simply by a
122  * number, e.g. `[ 200, 400, 600 ]`, which will be translated into e.g.
123  * `"discrimopdaq:200"`, `"discrimopdaq:400"` and `"discrimopdaq:600"`,
124  * while a full specification can be also used: `[ 200, "discrim:config" ]`
125  * will turn into `"discrimopdaq:200"` and `"discrim:config"` (to specify
126  * a label with no instance name, set it ending with a colon, like in
127  * `"discrim:"`).
128  * * `ChannelPairing` (list of integral number lists): channels to combine;
129  * each element of this list is itself a list of optical detector channel
130  * numbers. The channels within each group are combined according to
131  * `CombinationMode`, resulting in a single combined gate associated to
132  * all the channels in the group.
133  * All optical detector channels *must* appear in the configuration, and
134  * each channel can be combined only in one group. Groups must not be empty.
135  * * `IgnoreChannels` (list of integral numbers, optional): ID of the optical
136  * detector channels to skip.
137  * * `CombinationMode` (either: `"disable"`, `"input1"`, `"input2"`, `"AND"`
138  * or `"OR"`): for each group of channels defined in `ChannelPairing`,
139  * the selected operation is used to combine all the channels in the group:
140  * * `disable`: all channels are discarded, and the resulting gates are
141  * all closed all the way;
142  * * `input1`: the first channel in the pairing is kept, and the others
143  * are discarded;
144  * * `input2`: the second channel in the pairing is kept, and the others
145  * are discarded;
146  * * `AND`: all channels are combined in a AND, resulting in a gate open
147  * only at the time when all the channel gates in the group are open;
148  * if there is a single channel in the group, the combined gate is a
149  * copy of it;
150  * * `OR`: all channels are combined in a OR, resulting in a gate open
151  * at every time when any of the channel gates in the group is open;
152  * if there is a single channel in the group, the combined gate is a
153  * copy of it.
154  * All pairings undergo the same operation (it is not possible, for example,
155  * disabling only one group of channels and combining the other groups with
156  * a `AND`).
157  * * `ProduceWaveformAssns` (flag, default: `true`): produce also associations
158  * between each gate and the `raw::OpDetWaveform` which contributed to it.
159  * * `LogCategory` (string): name of the output stream category for console
160  * messages (managed by MessageFacility library).
161  *
162  *
163  *
164  */

530  ;

532

548  } // if fProduceWaveformAssns
549

553  outputInstanceName
554  );
555  event.put(
556  std::make_unique<art::Assns<OpticalTriggerGateData_t, sbn::OpDetWaveformMeta>>

564 //------------------------------------------------------------------------------
565 auto icarus::trigger::LVDSgates::combineChannels(
566  std::vector<TrackedTriggerGate_t> const& gates,
567  std::vector<raw::Channel_t> const& pairing,
568  ComboMode comboMode
569 ) const -> TrackedTriggerGate_t {
570 
571  namespace GateOps = icarus::trigger::GateOps;
572 
573  switch (comboMode) {
574  case ComboMode::disable:
575  return discardChannels(gates, pairing);
576  case ComboMode::AND:
577  return binaryCombineChannel(gates, pairing, GateOps::Min);
578  case ComboMode::OR:
579  return binaryCombineChannel(gates, pairing, GateOps::Max);
580  case ComboMode::Input1:
581  return selectChannel(gates, pairing, 0U);
582  case ComboMode::Input2:
583  return selectChannel(gates, pairing, 1U);
584  default:
585  throw art::Exception(art::errors::LogicError)
586  << "Unexpected combination mode (#" << static_cast<int>(comboMode)
587  << ").\n";
588  } // switch(comboMode)

690 //------------------------------------------------------------------------------
691 unsigned int icarus::trigger::LVDSgates::checkPairings() const {
692 
693  //
694  // collect all the errors first
695  //
696  auto const& geom = *(lar::providerFrom<geo::Geometry>());
697 
698 
699  // collect configured and duplicate channel numbers
700  unsigned int nEmptyGroups = 0U;
701  std::set<raw::Channel_t> configuredChannels, duplicateChannels;
702  for (auto const& pairing: fChannelPairing) {
703  if (pairing.empty()) {
704  ++nEmptyGroups;
705  continue;
706  }
707  for (raw::Channel_t channel: pairing) {
708  if (!configuredChannels.insert(channel).second)
709  duplicateChannels.insert(channel);
710  } // for channel
711  } // for pairing
712 
713  // collect invalid channel numbers
714  std::vector<raw::Channel_t> invalidChannels;
715  for (raw::Channel_t channel: configuredChannels) {
716  if (!geom.IsValidOpChannel(channel)) invalidChannels.push_back(channel);
717  } // for configured channels
718 
719  // collect missing channels
720  auto const endChannel = geom.MaxOpChannel() + 1U;
721  std::vector<raw::Channel_t> missingChannels;
722  for (auto channel: util::counter<raw::Channel_t>(endChannel)) {
723  if (!geom.IsValidOpChannel(channel)) continue;
724  if (configuredChannels.count(channel) > 0U) continue;
725  missingChannels.push_back(channel);
726  } // for channel
727 
728  // no error? good we are
729  if (invalidChannels.empty()
730  && missingChannels.empty()
731  && duplicateChannels.empty()
732  && (nEmptyGroups == 0U)
733  ) {
734  return configuredChannels.size();
735  }
736 
737  // error? good we are not
738  art::Exception e(art::errors::Configuration);
739  e << "The pairing configuration of " << configuredChannels.size()
740  << " optical channels in `ChannelPairing` presents the following errors.\n";
741  if (!invalidChannels.empty()) {
742  auto iChannel = invalidChannels.cbegin();
743  auto const cend = invalidChannels.cend();
744  e
745  << invalidChannels.size()
746  << " channel numbers do not represent valid channels: " << *iChannel;
747  while (++iChannel != cend) e << ", " << *iChannel;
748  e << ".\n";
749  } // if invalid channels
750 
751  if (!missingChannels.empty()) {
752  auto iChannel = missingChannels.cbegin();
753  auto const cend = missingChannels.cend();
754  e
755  << missingChannels.size()
756  << " channels do not appear in the configuration: " << *iChannel;
757  while (++iChannel != cend) e << ", " << *iChannel;
758  e << ".\n";
759  } // if missing channels
760 
761  if (!duplicateChannels.empty()) {
762  auto iChannel = duplicateChannels.cbegin();
763  auto const cend = duplicateChannels.cend();
764  e
765  << duplicateChannels.size()
766  << " channels are specified more than once: " << *iChannel;
767  while (++iChannel != cend) e << ", " << *iChannel;
768  e << ".\n";
769  } // if duplicate channels
770 
771  if (nEmptyGroups > 0U) {
772  e << nEmptyGroups << " channel groups are empty.";
773  }
774 
775  throw e;
776 } // icarus::trigger::LVDSgates::checkPairings()
