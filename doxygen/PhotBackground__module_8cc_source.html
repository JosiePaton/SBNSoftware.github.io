131  // Define a "handle" to point to a vector of MCParticle objects.
132  art::Handle< vector<simb::MCParticle> > particleHandle;
133  e.getByLabel("largeant", particleHandle);

135  vector<const simb::MCParticle*> photList;
136 
137  //true CRT hits
138  art::Handle< vector<sbn::crt::CRTHit> > trueHitHandle;
139  vector< art::Ptr<sbn::crt::CRTHit> > trueHitList;
140  map<int,vector<art::Ptr<sbn::crt::CRTHit>>> muToTrueHits;
141 
142  if( e.getByLabel("crttruehit",trueHitHandle) )
143  art::fill_ptr_vector(trueHitList,trueHitHandle);
144 
145  for(auto const& particle : *particleHandle) {
146 
147  if(abs(particle.PdgCode())==13){
148  idToMu[particle.TrackId()] = &particle;
149  }
150 
151  else if(particle.PdgCode()==22 && particle.Process()=="muBrems")
152  photList.push_back(&particle);
153 
154  else
155  continue;
156  }//for MCParticles
157 
158  for(auto const& hit : trueHitList){
159  for(const int id: bt.AllTrueIds(e,*hit)) {
160  if(idToMu.find(id)==idToMu.end())
161  continue;
162  muToTrueHits[id].push_back(hit);
163  }//for trackIDs
164  }//for CRTTrueHits
165 
166  for(auto const& phot : photList) {
167 
168  const TLorentzVector& positionStart = phot->Position();
169  const TLorentzVector& positionEnd = phot->EndPosition();
170 
171  fPhotIV = false;
172  fPhotAV = false;
173  fEndReg = EnteredLar(positionEnd, fPhotIV, fPhotAV);
174  if(fEndReg==-1)
175  continue;
176  bool iv = false, av = false;
177  fStartReg = EnteredLar(positionStart, iv, av);
178 
179  for(size_t i=0; i<4; i++) {
180  fStartPos[i] = positionStart[i];
181  fEndPos[i] = positionEnd[i];
182  }
183  fStartE = phot->E();
184  fEndE = phot->E(phot->NumberTrajectoryPoints()-2);
185 
186  fIcode = -1;
187  string endprocess = phot->EndProcess();
188  if(endprocess=="conv") fIcode = 0; //pair prod
189  else if(endprocess=="compt") fIcode = 1; //compton scatter
190  else if(endprocess=="phot") fIcode = 2; //photoelectric
191  else std::cout << "uknown end process: " << endprocess << std::endl;
192 
193 
194 
195  if(idToMu.find(phot->Mother())==idToMu.end())
196  throw cet::exception("PhotBackground") <<
197  "no match muon found for mubrems photons" << std::endl;
198

201  fMuTag = false;
202  fMuNHit = 0;

204  fMuHitPos.clear();
205  fMuHitDx.clear();
206  fMuHitDt.clear();
207
