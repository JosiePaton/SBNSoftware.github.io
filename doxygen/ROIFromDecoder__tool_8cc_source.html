80 void ROIFromDecoder::FindROIs(const art::Event& event, const ArrayFloat& inputImage, const geo::PlaneID& planeID, ArrayFloat& output, ArrayBool& outputROIs)
81 {
82  // First thing is find the correct data product to recover ROIs
83  TPCIDToLabelMap::const_iterator tpcItr = fTPCIDToLabelMap.find(planeID.asTPCID());
84 
85  if (tpcItr == fTPCIDToLabelMap.end())
86  {
87  std::cout << "Can't happen? for planeID: " << planeID << std::endl;
88  return;
89  }
90 
91  art::InputTag roiLabel("");
92 
93  for(const auto& tag : fROILabelVec)
94  {
95  if (tag.instance().find(tpcItr->second) != std::string::npos)
96  {
97  roiLabel = tag;
98  break;
99  }
100  }
101 
102  if (roiLabel != "")
103  {
104  // do stuff here
105 
106  // Read in the collection of full length deconvolved waveforms
107  // Note we assume this list is sorted in increasing channel number!
108  art::Handle< std::vector<recob::Wire>> wireVecHandle;
109 
110  event.getByLabel(roiLabel, wireVecHandle);
111 
112  // The input data product will contain information for every plane in a physical TPC... unfortunately, we only want a single plane within
113  // a logical TPC... so we need to loop through to find what we want
114  for(const auto& wireData : *wireVecHandle)
115  {
116  std::vector<geo::WireID> wireIDVec = fGeometry->ChannelToWire(wireData.Channel());
117 
118  for(const auto& wireID : wireIDVec)
119  {
120  if (wireID.asPlaneID() != planeID) continue;
121 
122  if (wireID.Wire > outputROIs.size())
123  {
124  std::cout << "#################################### Wire out of bounds! Wire: " << wireID.Wire << ", max: " << outputROIs.size() << " #################" << std::endl;
125  continue;
126  }
127 
128  // Recover this wire's output vector
129  VectorBool& channelData = outputROIs[wireID.Wire];
130 
131  // What we need to do is find the ROIs in the input wire data and then translate to the output
132  const recob::Wire::RegionsOfInterest_t signalROIs = wireData.SignalROI();
133 
134  for(const auto& range : signalROIs.get_ranges())
135  {
136  size_t startTick = range.begin_index();
137  size_t roiLen = range.data().size();
138  size_t stopTick = startTick + roiLen;
139 
140  if (startTick > channelData.size())
141  {
142  std::cout << "*** ROI decoder has start tick larger than output array, start: " << startTick << ", array size: " << channelData.size() << std::endl;
143  continue;
144  }
145 
146  if (stopTick > channelData.size())
147  {
148  std::cout << "*** ROI decoder has ROI length larger than output array, start: " << startTick << ", end: " << stopTick << ", array size: " << channelData.size() << std::endl;
149  continue;
150  }
151 
152  std::fill(channelData.begin() + startTick, channelData.begin() + stopTick, true);
153  }
154  }
155  }
156 
157  }
158 
159  return;
160 }
