88 /**
89  * @brief Simulates a sliding window trigger.
90  *
91  * This module produces `raw::Trigger` objects each representing the outcome of
92  * some trigger logic applied to a discriminated input ("trigger primitives").
93  *
94  * A trigger primitive is a two-level function of time which describes when
95  * that primitive is on and when it is off. Trigger primitives are given as
96  * input to this module and their origin may vary, but the standard source in
97  * ICARUS is @ref ICARUSPMTTriggerGlossary "single trigger request (LVDS)".
98  *
99  * This module applies a sliding window pattern to the input: the pattern
100  * consists of a requirement on the main window and optional additional
101  * requirements on the neighbouring windows. This module rebases the configured
102  * pattern on each of the available windows, evaluates the requirement of the
103  * pattern in that configuration, and decides whether those requirements are
104  * met. The general trigger is considered passed if _any_ of the rebased
105  * patterns satisfies the requirement at any time, and no special treatment is
106  * performed in case multiple windows fulfil them, except that the trigger time
107  * is driven by the earliest of the satisfied patterns.
108  *
109  * A single trigger pattern is configured for each instance of the module,
110  * while multiple input sets (e.g. with different discrimination thresholds)
111  * can be processed on the same pattern by the same module instance.
112  * Conversely, testing a different pattern requires the instantiation of a new
113  * module.
114  *
115  *
116  * Configuration
117  * ==============
118  *
119  * * `TriggerGatesTag` (string, mandatory): name of the module instance which
120  * produced the trigger primitives to be used as input; it must not include
121  * any instance name, as the instance names will be automatically added from
122  * `Thresholds` parameter.
123  * The typical trigger primitives used as input are LVDS discriminated
124  * output combined into trigger windows (e.g. from
125  * `icarus::trigger::SlidingWindowTrigger` module).
126  * * `Thresholds` (list of names, mandatory): list of the discrimination
127  * thresholds to consider. A data product containing a digital signal is
128  * read for each one of the thresholds, and the tag of the data product is
129  * expected to be the instance name in this configuration parameter for the
130  * module label set in `TriggerGatesTag` (e.g. for a threshold of
131  * `"60"`, supposedly 60 ADC counts, and with `TriggerGatesTag` set to
132  * `"TrigSlidingWindows"`, the data product tag would be
133  * `TrigSlidingWindows:60`).
134  * * `Pattern` (configuration table, mandatory): describes the sliding window
135  * pattern; the configuration format for a pattern is described under
136  * `icarus::trigger::ns::fhicl::WindowPatternConfig`.
137  * * `BeamGateDuration` (time, _mandatory_): the duration of the beam
138  * gate; _the time requires the unit to be explicitly specified_: use
139  * `"1.6 us"` for BNB, `9.5 us` for NuMI (also available as
140  * `BNB_settings.spill_duration` and `NuMI_settings.spill_duration` in
141  * `trigger_icarus.fcl`);
142  * * `BeamGateStart` (time, default: `0_us`): how long after the
143  * * @ref DetectorClocksBeamGateOpening "nominal beam gate opening time"
144  * the actual beam gate opens at;
145  * * `BeamBits` (TODO): bits to be set in the produced `raw::Trigger` objects.
146  * * `LogCategory` (string, default `SlidingWindowTriggerSimulation`): name of
147  * category used to stream messages from this module into message facility.
148  *
149  * An example job configuration is provided as
150  * `simulate_sliding_window_trigger_icarus.fcl`.
151  *
152  *
153  * Output data products
154  * =====================
155  *
156  * * `std::vector<raw::Trigger>` (one instance per ADC threshold):
157  * list of triggers fired according to the configured trigger definition;
158  * there is one collection (and data product) per ADC threshold, and the
159  * data product has the same instance name as the input data one
160  * (see `TriggerGatesTag` and `Thresholds` configuration parameters);
161  * currently only at most one trigger is emitted, with time stamp matching
162  * the first time the trigger criteria are satisfied.
163  *
164  *
165  *
166  * Trigger logic algorithm
167  * ========================
168  *
169  * @anchor SlidingWindowTriggerSimulation_Algorithm
170  *
171  * This section describes the trigger logic algorithm used in
172  * `icarus::trigger::SlidingWindowTriggerSimulation` and its assumptions.
173  * Nevertheless, more up-to-date information can be found in
174  * `SlidingWindowTrigger` module (for the combination of the LVDS signals into
175  * window-wide gates) and in `icarus::trigger::SlidingWindowPatternAlg`,
176  * which applies the configured pattern logic to the input.
177  *
178  * The module receives as input a multi-level trigger gate for each of the
179  * windows to be considered.
180  * On the first input (i.e. the first event), that input is parsed to learn
181  * the windows and their relative position from the input trigger gates.
182  * This topology will be used to apply the configured patterns. On the following
183  * events, their input is checked to confirm the compatibility of the
184  * composition of its windows with the one from that first event (both aspects
185  * are handled by an `icarus::trigger::WindowTopologyManager` object).
186  *
187  * All multi-level gates are set in coincidence with the beam gate by
188  * multiplying the multi-level and the beam gates. Because of this, trigger
189  * gates are suppressed everywhere except than during the beam gate.
190  * The beam gate opens at a time configured in `DetectorClocks` service provider
191  * (`detinfo::DetectorClocks::BeamGateTime()`), optionally offset
192  * (`BeamGateStart`), and has a duration configured in this module
193  * (`BeamGateDuration`).
194  *
195  * The algorithm handles independently multiple trigger patterns.
196  * On each input, each configured pattern is applied based on the window
197  * topology. Each pattern describes a minimum level of the trigger
198  * gate in the window, that usually means the number of LVDS signals in
199  * coincidence at any given time ("majority"). A pattern may have requirements
200  * on the neighbouring windows in addition to the main one. The pattern is
201  * satisfied if all involved windows pass their specific requirements at the
202  * same time (coincidence between windows).
203  * Each pattern is applied in turn to each of the windows (which is the "main"
204  * window). The neighborhood described in the pattern is applied with respect to
205  * that main window. The trigger fires if one or more of the windows satisfy
206  * the pattern, and the trigger time is the one of the earliest satisfied
207  * pattern (more precisely, the earliest tick when the coincidence required
208  * by that pattern is satisfied).*
209  * All windows in the detector are considered independently, but the supported
210  * patterns may only include components in the same cryostat. Therefore,
211  * triggers are effectively on a single cryostat.
212  * An object of class `icarus::trigger::SlidingWindowPatternAlg` applies this
213  * logic: see its documentation for the most up-to-date details.
214  *
215  * Eventually, for each event there are as many different trigger responses as
216  * how many different patterns are configured (`Patterns` configuration
217  * parameter), _times_ how many ADC thresholds are provided in input,
218  * configured in `Thresholds`.
219  *
220  *
221  * Technical aspects of the module
222  * --------------------------------
223  *
224  * @anchor SlidingWindowTriggerSimulation_Tech
225  *
226  * This module does not build the trigger gates of the sliding windows, but
227  * rather it takes them as input (see e.g. `SlidingWindowTrigger` module).
228  * Window topology (size of the windows and their relations) is stored in
229  * `icarus::trigger::WindowChannelMap` objects, and its construction is
230  * delegated to `icarus::trigger::WindowTopologyAlg` (under the hood of the
231  * `WindowTopologyManager` class) which learns it from the actual trigger gate
232  * input rather than on explicit configuration. Pattern definitions and
233  * configuration are defined in `icarus::trigger::WindowPattern` and
234  * `icarus::trigger::ns::fhicl::WindowPatternConfig` respectively. Trigger
235  * simulation is delegated to `icarus::trigger::SlidingWindowPatternAlg`.
236  *
237  */

589 //------------------------------------------------------------------------------
590 void icarus::trigger::SlidingWindowTriggerSimulation::produce(art::Event& event)
591 {
592 
593  auto const clockData
594  = art::ServiceHandle<detinfo::DetectorClocksService const>()->DataFor(event);
595  detinfo::DetectorTimings const detTimings{clockData};
596  auto const beamGate = makeMyBeamGate(event);
597 
598  if (auto oldGate = fGateChangeCheck(beamGate); oldGate) {
599  mf::LogWarning(fLogCategory)
600  << "Beam gate has changed from " << *oldGate << " to " << beamGate << "!";
601  }
602 
603 
604  mf::LogDebug log(fLogCategory); // this will print at the end of produce()
605  log << "Event " << event.id() << ":";

785 //------------------------------------------------------------------------------
786 auto icarus::trigger::SlidingWindowTriggerSimulation::produceForThreshold(
787  art::Event& event,
788  detinfo::DetectorTimings const& detTimings,
789  ApplyBeamGateClass const& beamGate,
790  std::size_t const iThr, std::string const& thrTag
791 ) -> WindowTriggerInfo_t {
792 
793  auto& plotInfo = fThresholdPlots[iThr];
794 
795  //
796  // get the input
797  //
798  art::InputTag const& dataTag = fADCthresholds.at(thrTag);
799  auto const& gates = readTriggerGates(event, dataTag);
800 
801 
802  // extract or verify the topology of the trigger windows
803  if (fWindowMapMan(gates))
804  fPatternAlg.emplace(*fWindowMapMan, fPattern, fLogCategory);
805  assert(fPatternAlg);
806 
807  //
808  // simulate the trigger response
809  //
810  WindowTriggerInfo_t const triggerInfo
811  = fPatternAlg->simulateResponse(beamGate.applyToAll(gates));
812  if (triggerInfo) {
813  ++fTriggerCount[iThr]; // keep the unique count
814  plotInfo.eventTimes.add(eventTimestampInSeconds(event));
815  }
816 
817  //
818  // fill the plots
819  //
820  plotTriggerResponse(iThr, triggerInfo);
821 
822  //
823  // create and store the data product
824  //
825  auto triggers = std::make_unique<std::vector<raw::Trigger>>();
826  if (triggerInfo.info.fired()) {
827  triggers->push_back
828  (triggerInfoToTriggerData(detTimings, fTriggerCount[iThr], triggerInfo));
829  } // if
830  event.put(std::move(triggers), dataTag.instance());

832  return triggerInfo;
