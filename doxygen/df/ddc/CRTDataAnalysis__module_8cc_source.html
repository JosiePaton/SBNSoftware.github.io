291  void CRTDataAnalysis::FillFebMap() {
292  if(!this->fFebMap.empty())
293  return;
294  std::string fullFileName;
295  cet::search_path searchPath("FW_SEARCH_PATH");
296  searchPath.find_file("feb_map.txt",fullFileName);
297  std::ifstream fin;
298  fin.open(fullFileName,std::ios::in);
299  if(fin.good()) std::cout << "opened file 'feb_map.txt' for reading..." << std::endl;
300  else //std::cout << "could not open file 'feb_map.txt' for reading!" << std::endl;
301  throw cet::exception("CRTDataAnalysis::FillFebMap") << "Unable to find/open file 'feb_map.txt'" << std::endl;
302  std::vector<std::string> row;
303  std::string line, word;
304  while(getline(fin,line)) {
305  row.clear();
306  std::stringstream s(line);
307  int mod;
308  while (std::getline(s, word, ',')) {
309  row.push_back(word);
310  }
311  mod = std::stoi(row[0]);
312  (this->fFebMap)[mod].push_back(std::make_pair(std::stoi(row[1]),std::stoi(row[2])));
313  if(row.size()>3)
314  (this->fFebMap)[mod].push_back(std::make_pair(std::stoi(row[3]),std::stoi(row[4])));
315  }
316  std::cout << "filled febMap with " << (this->fFebMap).size() << " entries" << std::endl;
317  fin.close();
318  }
