379 int opana::ICARUSFlashAssAna::getSideByChannel( const int channel ) {
380 
381  /*
382  Channels are numbered from east to west, from North (cryo side) to South (beam side)
383  We look in the opposide direction wrt to the beam direction South->North:
384 
385  - Left is the east wall of each cryostat;
386 
387  - Right is the west side of each cryostat;
388 
389  - [ 0:89 ] and [180:269] are on the left,
390  the return value of the function is 0;
391 
392  - [ 90-179 ] and [ 270:359 ] are on the right,
393  the return value of the function is 1;
394  */
395 
396 
397  int side = channel / 90; // always round down
398 
399  return side % 2;
