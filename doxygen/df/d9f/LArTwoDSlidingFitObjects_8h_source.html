340 inline void LayerFitContribution::AddPoint(const float l, const float t)
341 {
342  const double T = static_cast<double>(t);
343  const double L = static_cast<double>(l);
344 
345  m_sumT += T;
346  m_sumL += L;
347  m_sumTT += T * T;
348  m_sumLT += L * T;
349  m_sumLL += L * L;
350  ++m_nPoints;
351 }
