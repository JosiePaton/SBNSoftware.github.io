100 /**
101  * @brief Class managing the serialization of trigger gates in a simple ROOT
102  * tree.
103  *
104  * The tree is supplied by the caller.
105  * This object will create the proper branches into the tree and assign
106  * addresses to them. Then it will assume they will stay assigned.
107  * Information may be loaded into the tree via `assignTriggerGatesInfo()`
108  * method, which sets each branch according to a value of the data structure
109  * `TriggerGatesInfo`, as described below.
110  *
111  * The branch structure is:
112  * `NChannels/i:OpDetPos_[]/D:NOpenings[]/i:OpeningTime[]/D`
113  *
114  * * `NChannels` (from the size of `TriggerGatesInfo::info`): number of
115  * channels in this event;
116  * * `OpDetPos` (one 3D point per channel, from `TriggerGateInfo::center`):
117  * the location of the centroid of the optical detectors contributing to the
118  * channel, in world coordinates [cm]; it's a vector of GenVector 3D points
119  * (can access coordinates as `OpDetPos.X()` or `OpDetPos.fCoordinates.fX`);
120  * * `NOpenings` (one count per channel, from `TriggerGateInfo::nOpenings`):
121  * the number of times the gate opened;
122  * * `OpeningTime` (one time per channel, based on
123  * `TriggerGateInfo::firstOpenTime`): first time that channel opened, or
124  * very large number (`std::numeric_limits<double>::max()`)
125  * if no opening happened at all (note that in this case the value in
126  * `TriggerGateInfo::firstOpenTime` is ignored).
127  *
128  * All branches come from a variable size vector with size the number of
129  * available trigger gates.
130  */

158 /**
159  * @brief Make ROOT tree with information about event and trigger input.
160  *
161  * This module produces a ROOT tree with information about the event and about
162  * the trigger input.
163  *
164  *
165  * Input
166  * ======
167  *
168  * The following data products are required in input:
169  *
170  * * `std::vector<simb::MCTruth>` (tag: configured in `GeneratorTags`):
171  * any number of truth information records;
172  * * `std::vector<sim::SimEnergyDeposits>` (tag: configured in
173  * `EnergyDepositTags`): energy depositions to be counted in the event
174  * information;
175  * * `std::vector<icarus::trigger::OpticalTriggerGateData_t>` (tag: configured
176  * in `TriggerGatesTag`): list of trigger gate outputs to be put into the
177  * tree.
178  *
179  *
180  * Output
181  * =======
182  *
183  * The output tree is written via `TFileService` in the ROOT directory assigned
184  * to this module.
185  *
186  * The tree contains three "sections":
187  *
188  * * event identification: see `icarus::trigger::details::EventIDTree`
189  * for the details;
190  * * event information: see `icarus::trigger::details::EventInfoTree`
191  * for the details;
192  * * information on the trigger gates (data product configured via
193  * `TriggerGatesTag` parameter): information on the state of each trigger
194  * gate _during the beam gate_ is extracted, including the number of times
195  * the gate is opened and when; see `TriggerGateTree` for details on the
196  * structure of the tree, and `extractTriggerInfo()` for the details of how
197  * each value is computed.
198  *
199  *
200  * Configuration parameters
201  * =========================
202  *
203  * * `GeneratorTags` (list of input tags, default: `[ generator ]`): a list of
204  * data products containing the particles generated by event generators;
205  * * `EnergyDepositTags`
206  * (list of input tags, default: `[ "largeant:TPCActive" ]`): a list of
207  * data products with energy depositions;
208  * * `EnergyDepositSummaryTag` (input tag): alternative to `EnergyDepositTags`,
209  * uses energy deposition information from a summary data product instead
210  * of extracting the information;
211  * * `TriggerGatesTag` (string, mandatory): name of the module
212  * instance which produced the trigger primitives to be used as input.
213  * The typical trigger primitives used as input may be LVDS discriminated
214  * output (e.g. from `icarus::trigger::LVDSgates` module) or combinations
215  * of them (e.g. from `icarus::trigger::SlidingWindowTrigger` module);
216  * * `BeamGateDuration` (time, _mandatory_): the duration of the beam
217  * gate; _the time requires the unit to be explicitly specified_: use
218  * `"1.6 us"` for BNB, `9.5 us` for NuMI (also available as
219  * `BNB_settings.spill_duration` and `NuMI_settings.spill_duration` in
220  * `trigger_icarus.fcl`);
221  * * `BeamGateStart` (time, default: 0 &micro;s): open the beam gate this long
222  * after the nominal beam gate time;
223  * * `PreSpillWindow` (time, default: 10 &micro;s): duration of the pre-spill
224  * window;
225  * * `PreSpillWindowGap` (time, default: 0 &micro;s): gap from the end of
226  * pre-spill window to the start of beam gate;
227  * * `EventTreeName` (string, default: "Treegger"): the name of the ROOT tree
228  * being written on disk;
229  * * `LogCategory` (string, default `MakeTriggerSimulationTree`): name of
230  * category used to stream messages from this module into message facility.
231  *
232  *
233  * Technical description of the module
234  * ====================================
235  *
236  *
237  */

386  /**
387  * @brief Returns a `TriggerGatesInfo` with trigger information for the tree.
388  * @param event the _art_ event where to find the trigger gates
389  * @return a `TriggerGatesInfo` with trigger information for the tree
390  *
391  * Information from the trigger gates configured in the module is extracted
392  * and stored into a `TriggerGatesInfo` object, which is returned.
393  * The data fields are defined as follows:
394  *
395  * * a `TriggerGateInfo` element is stored in `TriggerGatesInfo::info` for
396  * each and every trigger gate in the input, in the same order as the
397  * input data product;
398  * * for each gate:
399  * * `TriggerGateInfo::center`: for each channel associated to the
400  * trigger gate, the geometric center of the optical detector connected
401  * to that channel is taken; `center` is set to the middle point
402  * (unweighted 3D centroid) of all those points; the center is in world
403  * coordinates and in centimeters;it is assumed that the trigger gate
404  * is associated to at least one channel, otherwise behavior is
405  * undefined;
406  * * `TriggerGateInfo::nOpenings`: the number of times in coincidence
407  * with the beam gate, when the gate changes state from closed (opening
408  * level `0`) to open (opening level `1` or larger);
409  * * `TriggerGateInfo::firstOpenTime`: the time the first of the openings
410  * (as defined in `nOpenings`) happens; the time is in
411  * @ref DetectorClocksSimulationTime "simulation time scale"; if there
412  * was no opening during the beam gate (`nOpenings` zero), the value
413  * is undefined;
414  * * 'TriggerGateInfo::Amplitude': the amplitude of the PMT
415  *
416  */
417  TriggerGatesInfo extractTriggerInfo(art::Event const& event) const;

609 //------------------------------------------------------------------------------
610 TriggerGatesInfo icarus::trigger::MakeTriggerSimulationTree::extractTriggerInfo
611  (art::Event const& event) const
612 {
613  /*
614  * 1. get the data product from the event
615  * 2. fill one "channel" of `TriggerGateInfo` per entry in the data product
616  * 1. get centroid of associated detectors
617  * 2. find and convert to `simulation_time` the first opening
618  * 3. find the number of openings
619  * 4. find the amplitude
620  * 4. fill the information
621  *
622  */
623 
624  using detinfo::timescales::optical_tick;

637  //

642  // 2. fill one "channel" of `TriggerGateInfo` per entry in the data product
643  //

645

647  for (OpticalTriggerGateData_t const& gate: gates) {

651

654  //

663  while (openTick != OpticalTriggerGateData_t::MaxTick) {
664  ++nOpenings;
665  // move out of this open region, then find the next one
666  openTick = beamAndGate.findClose(1U, openTick);
667  openTick = beamAndGate.findOpen(1U, openTick);
668  } // while
669  if (nOpenings > 0) ++nOpenChannels;
670

672  // 2.4. fill the information
673  //
674  info.TriggerGates.push_back({
675  gateChannelCentroid(gate), // 2.1. get centroid of associated detectors
676  nOpenings,
677  // users should ignore this if `nOpenings == 0`:
678  (nOpenings == 0U)
679  ? std::numeric_limits<simulation_time>::max() //std::numeric_limits<T>::max Returns the maximum finite value representable by the
680  //numeric type T. Meaningful for all bounded types.
681  : detTimings.toSimulationTime
682  (detinfo::timescales::optical_tick{ firstOpenTick })
683  //,amplitude
684  });
685 
686  } // for all gates
687 
688  mf::LogTrace(fLogCategory)
689  << "Information from '" << fTriggerGatesTag.encode() << "' ("
690  << gates.size() << " trigger gates, " << nOpenChannels
691  << " channels) written to tree."
692  ;
