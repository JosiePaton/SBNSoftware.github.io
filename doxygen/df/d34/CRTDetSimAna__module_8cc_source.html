221  void CRTDetSimAna::analyze(const art::Event& event)
222  {
223 
224  // Fetch basic event info
225  if(fVerbose){
226  std::cout<<"============================================"<<std::endl
227  <<"Run = "<<event.run()<<", SubRun = "<<event.subRun()<<", Event = "<<event.id().event()<<std::endl
228  <<"============================================"<<std::endl;
229  }
230 
231 
232  //----------------------------------------------------------------------------------------------------------
233  // GETTING PRODUCTS
234  //----------------------------------------------------------------------------------------------------------
235 
236  // Get g4 particles
237  art::ServiceHandle<cheat::ParticleInventoryService> pi_serv;
238  auto particleHandle = event.getValidHandle<std::vector<simb::MCParticle>>(fSimModuleLabel);
239 
240  // Get CRT data from the event
241  art::Handle< std::vector<sbnd::crt::CRTData>> crtDataHandle;
242  std::vector<art::Ptr<sbnd::crt::CRTData> > crtDataList;
243  if (event.getByLabel(fCRTSimLabel, crtDataHandle))
244  art::fill_ptr_vector(crtDataList, crtDataHandle);
245 
246  art::FindManyP<sim::AuxDetIDE> findManyIdes(crtDataHandle, event, fCRTSimLabel);
247 
248  // Get all the auxdet IDEs from the event
249  art::Handle<std::vector<sim::AuxDetSimChannel> > channels;
250  event.getByLabel(fSimChannelModuleLabel, channels);
251 
252  //----------------------------------------------------------------------------------------------------------
253  // TRUTH MATCHING
254  //----------------------------------------------------------------------------------------------------------
255 
256  std::map<int, simb::MCParticle> particles;
257  // Loop over the true particles
258  for (auto const& particle: (*particleHandle)){
259 
260  // Make map with ID
261  int partID = particle.TrackId();
262  particles[partID] = particle;
263 
264  }
265 
266  //----------------------------------------------------------------------------------------------------------
267  // CRT DATA ANALYSIS
268  //----------------------------------------------------------------------------------------------------------
269 
270  std::map<int, std::vector<sbnd::crt::CRTData>> crtData;
271  for(size_t i = 0; i < crtDataList.size(); i++){
272 
273  // Truth matching
274  std::vector<int> trueIDs = fCrtBackTrack.AllTrueIds(event, *crtDataList[i]);
275  for(auto const& id : trueIDs){
276  crtData[id].push_back(*crtDataList[i]);
277  }
278 
279  int trueID = fCrtBackTrack.TrueIdFromTotalEnergy(event, *crtDataList[i]);
280  // Only consider primary muons
281  if(particles.find(trueID) == particles.end()) continue;
282  if(!(std::abs(particles[trueID].PdgCode()) == 13 && particles[trueID].Mother() == 0)) continue;
283 
284  // Get the IDEs associated with the crtData
285  std::vector<art::Ptr<sim::AuxDetIDE>> ides = findManyIdes.at(i);
286  // Calculate the average position
287  double x = 0, y = 0, z = 0;
288  for(auto const& ide : ides){
289  x += (ide->entryX + ide->exitX)/2.;
290  y += (ide->entryY + ide->exitY)/2.;
291  z += (ide->entryZ + ide->exitZ)/2.;
292  }
293  x = x/ides.size();
294  y = y/ides.size();
295  z = z/ides.size();
296  geo::Point_t cross {x, y, z};
297 
298  int channel = crtDataList[i]->Channel();
299  std::string stripName = fCrtGeo.ChannelToStripName(channel);
300  std::string tagger = fCrtGeo.GetTaggerName(stripName);
301 
302  double npe = ((double)crtDataList[i]->ADC() - fQPed)/fQSlope;
303 
304  // Calculate distance to the sipm
305  double sipmDist = fCrtGeo.DistanceBetweenSipms(cross, channel);
306  hSipmDist[tagger]->Fill(sipmDist);
307  hSipmDistADC[tagger]->Fill(sipmDist, crtDataList[i]->ADC());
308  hSipmDistNpe[tagger]->Fill(sipmDist, npe);
309 
310  if(channel % 2 == 0){
311  double npe1 = ((double)crtDataList[i+1]->ADC() - fQPed)/fQSlope;
312  hNpeRatio[tagger]->Fill(npe/npe1);
313  hSipmDistNpeRatio[tagger]->Fill(sipmDist, npe/npe1);
314  }
315 
316  // Calculate the distance to the end
317  double stripDist = std::abs(fCrtGeo.DistanceDownStrip(cross, stripName));
318  hStripDist[tagger]->Fill(stripDist);
319  hStripDistADC[tagger]->Fill(stripDist, crtDataList[i]->ADC());
320  hStripDistNpe[tagger]->Fill(stripDist, npe);
321 
322  hADC[tagger]->Fill(crtDataList[i]->ADC());
323  hNpe[tagger]->Fill(npe);
324 
325  //fTrigClock.SetTime(crtDataList[i]->T0());
326  //double time = fTrigClock.Time(); // [us]
327  double time = (double)(int)crtDataList[i]->T0()/fClockSpeedCRT; // [tick -> us]
328  hTime[tagger]->Fill(time);
329  }
330 
331  //----------------------------------------------------------------------------------------------------------
332  // EFFICIENCIES
333  //----------------------------------------------------------------------------------------------------------
334 
335  for (auto& adsc : *channels) {
336  // Get the IDEs from the Aux Det channels
337  if(adsc.AuxDetSensitiveID() == UINT_MAX) {
338  mf::LogWarning("CRTDetSimAna") << "AuxDetSimChannel with ID: UINT_MAX\n"
339  << "skipping channel...";
340  continue;
341  }
342 
343  std::string stripName = fCrtGeo.GetStrip(adsc.AuxDetSensitiveID()).name;
344  std::string tagger = fCrtGeo.GetTaggerName(stripName);
345  std::vector<sim::AuxDetIDE> ides = adsc.AuxDetIDEs();
346 
347  for(auto const& ide : ides){
348  geo::Point_t pos {(ide.entryX + ide.exitX)/2,
349  (ide.entryY + ide.exitY)/2,
350  (ide.entryZ + ide.exitZ)/2};
351  int trueID = ide.trackID;
352  // Calculate length in strip
353  double length = std::sqrt(std::pow(ide.entryX - ide.exitX, 2)
354  + std::pow(ide.entryY - ide.exitY, 2)
355  + std::pow(ide.entryZ - ide.exitZ, 2));
356  // Calculate distances of average point to sipms
357  double sipmDistIde = fCrtGeo.DistanceBetweenSipms(pos, stripName);
358  double stripDistIde = fCrtGeo.DistanceDownStrip(pos, stripName);
359 
360  // Only consider primary muons for efficiency
361  if(particles.find(trueID) == particles.end()) continue;
362  if(!(std::abs(particles[trueID].PdgCode()) == 13 && particles[trueID].Mother() == 0)) continue;
363 
364  // Fill total histograms
365  hEffSipmDistTotal[tagger]->Fill(sipmDistIde);
366  hEffStripDistTotal[tagger]->Fill(stripDistIde);
367  hEffEDepTotal[tagger]->Fill(ide.energyDeposited);
368  hEffLengthTotal[tagger]->Fill(length);
369 
370  // Get all the CRT data matched to the same true ID
371  if(crtData.find(trueID) == crtData.end()) continue;
372  bool match = false;
373  // If there is CRT data produced on the same channel then it is matched
374  for(auto const& data : crtData[trueID]){
375  std::string stripNameData = fCrtGeo.ChannelToStripName(data.Channel());
376  if(stripName == stripNameData) match = true;
377  }
378  if(!match) continue;
379 
380  // Fill reconstructed histograms
381  hEffSipmDistReco[tagger]->Fill(sipmDistIde);
382  hEffStripDistReco[tagger]->Fill(stripDistIde);
383  hEffEDepReco[tagger]->Fill(ide.energyDeposited);
384  hEffLengthReco[tagger]->Fill(length);
385 
386  }
387  }
