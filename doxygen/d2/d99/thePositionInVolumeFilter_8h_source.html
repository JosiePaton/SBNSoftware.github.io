77  /**
78  * @brief Returns whether a track along the specified point must be kept
79  * @param pos point on the track, a [x,y,z] array in "Geant4 coordinates"
80  * @brief whether a track along the specified point must be kept
81  *
82  * If the return value is true, the track must be kept.
83  * If the return value is false, no decision can be made yet.
84  * If no decision is made after the track is over, the track should be
85  * dropped.
86  */
87  bool mustKeep(Point_t const& pos) const
88  {
89  // if no volume is specified, it means we don't filter
90  if (volumeInfo.empty()) return true;
91  double local[3];
92  for(auto const& info: volumeInfo) {
93  // transform the point to relative to the volume
94  info.trans->MasterToLocal(pos.data(), local);
95  // containment check
96  if (info.vol->Contains(local)) return true;
97  } // for volumes
98  return false;
99  } // mustKeep()
