276 
277  // Loop through the list of input daq fragment collections one by one
278  // We are not trying to multi thread at this stage because we are trying to control
279  // overall memory usage at this level. We'll multi thread internally...
280  for(const auto& fragmentLabel : fFragmentsLabelVec)

292  clockData,

294  concurrentRawDigits,

303 
304  // Want the RawDigits to be sorted in channel order... has to be done somewhere so why not now?
305  std::sort(rawDigitCollection->begin(),rawDigitCollection->end(),[](const auto& left,const auto&right){return left.Channel() < right.Channel();});
306 
307  // Now transfer ownership to the event store
308  event.put(std::move(rawDigitCollection), fragmentLabel.instance());
309 
310  if (fOutputRawWaveform)

316  // Want the RawDigits to be sorted in channel order... has to be done somewhere so why not now?

318

322

386  icarus_signal_processing::VectorFloat locPedsVec(decoderTool->getWaveLessCoherent().size(),0.);

425 void DaqDecoderICARUSTPC::saveRawDigits(const icarus_signal_processing::ArrayFloat& dataArray,
426  const icarus_signal_processing::VectorFloat& pedestalVec,
427  const icarus_signal_processing::VectorFloat& rmsVec,
428  const icarus_signal_processing::VectorInt& channelVec,
429  ConcurrentRawDigitCol& rawDigitCol) const
430 {
431  if (!dataArray.empty())
432  {
433  cet::cpu_timer theClockSave;
434 
435  theClockSave.start();
436 
437  raw::RawDigit::ADCvector_t wvfm(dataArray[0].size());
438 
439  mf::LogDebug("DaqDecoderICARUSTPC") << " --> saving rawdigits for " << dataArray.size() << " channels" << std::endl;
440 
441  // Loop over the channels to recover the RawDigits after filtering
442  for(size_t chanIdx = 0; chanIdx != dataArray.size(); chanIdx++)
443  {
444  // Protect against case where there was no readout
445  if (channelVec[chanIdx] < 0) continue;
446 
447  const icarus_signal_processing::VectorFloat& dataVec = dataArray[chanIdx];
448 
449  // Need to convert from float to short int
450  std::transform(dataVec.begin(),dataVec.end(),wvfm.begin(),[](const auto& val){return short(std::round(val));});
451 
452  ConcurrentRawDigitCol::iterator newObjItr = rawDigitCol.emplace_back(channelVec[chanIdx],wvfm.size(),wvfm);
453  newObjItr->SetPedestal(pedestalVec[chanIdx],rmsVec[chanIdx]);
454  }//loop over channel indices
455 
456  theClockSave.stop();
457 
458  double totalTime = theClockSave.accumulated_real_time();
459 
460  mf::LogDebug("DaqDecoderICARUSTPC") << " --> done with save, time: " << totalTime << std::endl;
461  }
462 
463  return;
464 }
