197 template <typename... TLISTS>
198 pandora::StatusCode LArMvaHelper::ProduceTrainingExample(const std::string &trainingOutputFile, const bool result, TLISTS &&... featureLists)
199 {
200  std::ofstream outfile;
201  outfile.open(trainingOutputFile, std::ios_base::app); // always append to the output file
202 
203  if (!outfile.is_open())
204  {
205  std::cout << "LArMvaHelper: could not open file for training examples at " << trainingOutputFile << std::endl;
206  return pandora::STATUS_CODE_FAILURE;
207  }
208 
209  std::string delimiter(",");
210  outfile << GetTimestampString() << delimiter;
211 
212  PANDORA_RETURN_RESULT_IF(pandora::STATUS_CODE_SUCCESS, !=, WriteFeaturesToFile(outfile, delimiter, featureLists...));
213  outfile << static_cast<int>(result) << '\n';
214 
215  return pandora::STATUS_CODE_SUCCESS;
216 }

291

298  std::string timeString(buffer);
