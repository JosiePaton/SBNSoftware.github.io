20 /**
21  * @file larreco/TrackFinder/TrackProducerFromTrack_module.cc
22  * @class TrackProducerFromTrack
23  *
24  * @brief Produce a reco::Track collection, as a result of the fit of an existing recob::Track collection.
25  *
26  * This producer takes an input an existing recob::Track collection (and the associated hits) and re-fits them.
27  * The mandatory output are: the resulting recob::Track collection and the associated hits.
28  * Optional outputs are recob::TrackFitHitInfo and recob::SpacePoint collections, plus the Assns of SpacePoints to Hits.
29  * An option is provided to create SpacePoints from the TrajectoryPoints in the Track.
30  * The fit is performed by an user-defined tool, which must inherit from larreco/TrackFinder/TrackMaker.
31  *
32  * Parameters: trackMaker (fhicl::ParameterSet for the trkmkr::TrackMaker tool used to do the fit), inputCollection (art::InputTag of the input recob::Track collection),
33  * doTrackFitHitInfo (bool to decide whether to produce recob::TrackFitHitInfo's), doSpacePoints (bool to decide whether to produce recob::SpacePoint's), and
34  * spacePointsFromTrajP (bool to decide whether the produced recob::SpacePoint's are taken from the recob::tracking::TrajectoryPoint_t's of the fitted recob::Track).
35  *
36  * @author G. Cerati (FNAL, MicroBooNE)
37  * @date 2017
38  * @version 1.0
39  */

157  }

159  }
160

164  if (doTrackFitHitInfo_) { e.put(std::move(outputHitInfo)); }
