531  nhits.push_back(k);
532  totalDist += dist;
533  }
534  }
535  //If the distance down the 1D hit means more hits are included and they are closer to the track record it
536  if(nhits.size()>=nhitsMax.size() && totalDist/nhits.size() < minDist){
537  nhitsMax = nhits;
538  facMax = fac;
539  minDist = totalDist/nhits.size();
540  }
541  nhits.clear();
542  }
543  //Record the track candidate
544  vector<size_t> trackCand;
545  trackCand.push_back(hit_i);
546  trackCand.push_back(hit_j);
547  trackCand.insert(trackCand.end(), nhitsMax.begin(), nhitsMax.end());
548  tracks.push_back(std::make_pair(trackCand, facMax));
549  }
550  //If there is no 1D hit
551  else{
552  TVector3 start(ihit.x_pos, ihit.y_pos, ihit.z_pos);
553  TVector3 end(jhit.x_pos, jhit.y_pos, jhit.z_pos);
554  TVector3 diff = start - end;
555  vector<size_t> trackCand;
556  trackCand.push_back(hit_i);
557  trackCand.push_back(hit_j);
558 
559  //Loop over all the other hits
560  for(size_t k = 0; k < hits.size(); k++){
561 
562  if(k == hit_i || k == hit_j || hits[k].tagger == ihit.tagger || hits[k].tagger == jhit.tagger)
563  continue;
564 
565  //Calculate distance to other hits not on the planes of the track hits
566  sbn::crt::CRTHit khit = hits[k];
567  TVector3 mid(khit.x_pos, khit.y_pos, khit.z_pos);
568  TVector3 cross = CrossPoint(khit, start, diff);
569  double dist = (cross-mid).Mag();
570 
571  //Record any within a certain distance
572  if(dist < fDistanceLimit){
573  trackCand.push_back(k);
574  }
575  }
576  tracks.push_back(std::make_pair(trackCand, 1));
577  }
578  }
579 
580  //Sort track candidates by number of hits
581  std::sort(tracks.begin(), tracks.end(), [](auto& left, auto& right){
582  return left.first.size() > right.first.size();});
583 
584  //Record used hits
585  vector<size_t> usedHits;
586 
587  //Loop over candidates
588  for(auto& track : tracks){
589 
590  size_t hit_i = track.first[0];
591  size_t hit_j = track.first[1];
592 
593  // Make sure the first hit is the top high tagger if there are only two hits
594  if(hits[hit_j].tagger=="volTaggerTopHigh_0")
595  std::swap(hit_i, hit_j);
596 
597  sbn::crt::CRTHit ihit = hits[hit_i];
598  sbn::crt::CRTHit jhit = hits[hit_j];
599 
600  //Check no hits in track have been used
601  bool used = false;
602  //Loop over hits in track candidate
603  for(size_t i = 0; i < track.first.size(); i++){
604  //Check if any of the hits have been used
605  if(std::find(usedHits.begin(), usedHits.end(), track.first[i]) != usedHits.end())
606  used=true;
607  }
608  //If any of the hits have already been used skip this track
609  if(used)
610  continue;
611  ihit.x_pos -= (1.-track.second)*ihit.x_err;
612  ihit.z_pos -= (1.-track.second)*ihit.z_err;
613 
614  //Create track
615  sbn::crt::CRTTrack crtTrack = FillCrtTrack(ihit, jhit, true);
616 
617  //If only the top two planes are hit create an incomplete/stopping track
618  if(track.first.size()==2 && ihit.tagger == "volTaggerTopHigh_0" && jhit.tagger == "volTaggerTopLow_0"){
619  crtTrack.complete = false;
620  }
621 
622  returnTracks.push_back(crtTrack);
623 
624  //Record which hits were used only if the track has more than two hits
625  //If there are multiple 2 hit tracks there is no way to distinguish between them
626  //TODO: Add charge matching for ambiguous cases
627  for(size_t i = 0; i < track.first.size(); i++){
628  if(track.first.size()>2) usedHits.push_back(track.first[i]);
629  }
630  }
631 
632  return returnTracks;
