581 //------------------------------------------------------------------------------

583  (art::Event const& event)
584 {
585 
586  /*
587  * 1. find out the features of the event and the categories it belongs to
588  * 2. for each threshold:
589  * 1. read the trigger primitives
590  * 2. pick the plots to be filled
591  * 3. combine the trigger primitives, apply the beam gate,
592  * generate the trigger response, add the response to all the plots
593  * (delegated)
594  *
595  */
596 
597  ++nEvents;
598 
599  //
600  // 1. find out the features of the event and the categories it belongs to
601  //
602 
603  auto const [ detTimings, beamGate, preSpillWindow ] = makeGatePack(&event);
604 
605  if (auto oldGate = fBeamGateChangeCheck(beamGate); oldGate) {
606  mf::LogWarning(fLogCategory)
607  << "Beam gate has changed from " << oldGate->asOptTickRange()
608  << " to " << beamGate.asOptTickRange() << " (optical tick)!";
609  }
610 
611  EventInfo_t const eventInfo = fEventInfoExtractorMaker
612  (beamGate.asSimulationRange(), preSpillWindow.asSimulationRange())(event);
613 
614 
615  bool const bPlot = shouldPlotEvent(eventInfo);
616  if (bPlot) ++nPlottedEvents;
617 
618  if (fIDTree) fIDTree->assignID(event.id());
619  if (fPlotTree) fPlotTree->assign(bPlot);
620  if (fEventTree) fEventTree->assignEvent(eventInfo);
621 
622  std::vector<std::string> selectedPlotCategories
623  = selectPlotCategories(eventInfo, fPlotCategories);
624  {
625  mf::LogTrace log(fLogCategory);
626  log
627  << "Event " << event.id() << " falls in " << selectedPlotCategories.size()
628  << " categories:"
629  ;
630  for (std::string const& name: selectedPlotCategories)
631  log << " \"" << name << "\"";
632  // print the information on the event
633  } // local block
634  if (!fLogEventDetails.empty()) {
635  mf::LogTrace(fLogEventDetails)
636  << "Event " << event.id() << ": " << eventInfo;
637  }
638 
639  //
640  // 2. for each PMT threshold:
641  //
642  auto const clockData
