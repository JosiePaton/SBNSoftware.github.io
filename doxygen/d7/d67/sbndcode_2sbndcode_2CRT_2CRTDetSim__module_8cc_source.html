92 uint32_t CRTDetSim::getChannelTriggerTicks(CLHEP::HepRandomEngine* engine,
93  /*detinfo::ElecClock& clock,*/
94  float t0, float npeMean, float r) {
95  // Hit timing, with smearing and NPE dependence
96  double tDelayMean =
97  fTDelayNorm *
98  exp(-0.5 * pow((npeMean - fTDelayShift) / fTDelaySigma, 2)) +
99  fTDelayOffset;
100 
101  double tDelayRMS =
102  fTDelayRMSGausNorm *
103  exp(-pow(npeMean - fTDelayRMSGausShift, 2) / fTDelayRMSGausSigma) +
104  fTDelayRMSExpNorm *
105  exp(-(npeMean - fTDelayRMSExpShift) / fTDelayRMSExpScale);
106 
107  double tDelay = CLHEP::RandGauss::shoot(engine, tDelayMean, tDelayRMS);
108 
109  // Time resolution of the interpolator
110  tDelay += CLHEP::RandGauss::shoot(engine, 0, fTResInterpolator);
111 
112  // Propagation time
113  double tProp = CLHEP::RandGauss::shoot(fPropDelay, fPropDelayError) * r;
114 
115  double t = t0 + tProp + tDelay;
116 
117  // Get clock ticks
118  // FIXME no clock available for CRTs, have to do it by hand
119  //clock.SetTime(t / 1e3); // SetTime takes microseconds
120  int time = (t / 1e3) * fClockSpeedCRT;
121 
122  mf::LogInfo("CRT")
123  << "CRT TIMING: t0=" << t0
124  << ", tDelayMean=" << tDelayMean << ", tDelayRMS=" << tDelayRMS
125  << ", tDelay=" << tDelay << ", tDelay(interp)="
126  << tDelay << ", tProp=" << tProp << ", t=" << t << /*", ticks=" << clock.Ticks() <<*/ "\n";
127 
128  return time;//clock.Ticks();
129 }

211

213  unsigned planeID = (modulePosMother[2] > 0);
214 
215  // Determine module orientation: which way is the top (readout end)?
216  bool top = (planeID == 1) ? (modulePosMother[1] > 0) : (modulePosMother[0] < 0);
217

227  double z = (ide.entryZ + ide.exitZ) / 2;

231  double tTrueLast = (ide.entryT + ide.exitT) / 2 + fGlobalT0Offset;

240  ide_i++;
241  x += (ides[ide_i].entryX + ides[ide_i].exitX) / 2;
242  y += (ides[ide_i].entryY + ides[ide_i].exitY) / 2;
243  z += (ides[ide_i].entryZ + ides[ide_i].exitZ) / 2;
244  eDep += ides[ide_i].energyDeposited;
245  tTrue += (ides[ide_i].entryT + ides[ide_i].exitT) / 2;

249

255  y = y/nides;

264  if (top) {
265  distToReadout = abs( adsGeo.HalfWidth1() - svHitPosLocal[0]);

271  // The expected number of PE, using a quadratic model for the distance
272  // dependence, and scaling linearly with deposited energy.
273  double qr = fUseEdep ? 1.0 * eDep / fQ0 : 1.0;
274 
275  double npeExpected =
276  fNpeScaleNorm / pow(distToReadout - fNpeScaleShift, 2) * qr;
277 
278  // Put PE on channels weighted by transverse distance across the strip,
279  // using an exponential model
280  double d0 = abs(-adsGeo.HalfHeight() - svHitPosLocal[1]); // L
281  double d1 = abs( adsGeo.HalfHeight() - svHitPosLocal[1]); // R
282  double abs0 = exp(-d0 / fAbsLenEff);
283  double abs1 = exp(-d1 / fAbsLenEff);
284  double npeExp0 = npeExpected * abs0 / (abs0 + abs1);
285  double npeExp1 = npeExpected * abs1 / (abs0 + abs1);
286 
287  // Observed PE (Poisson-fluctuated)
288  long npe0 = CLHEP::RandPoisson::shoot(&fEngine, npeExp0);
289  long npe1 = CLHEP::RandPoisson::shoot(&fEngine, npeExp1);
290 
291  // Time relative to trigger, accounting for propagation delay and 'walk'
292  // for the fixed-threshold discriminator
293  uint32_t t0 =
294  getChannelTriggerTicks(&fEngine, /*trigClock,*/ tTrue, npe0, distToReadout);
295  uint32_t t1 =
296  getChannelTriggerTicks(&fEngine, /*trigClock,*/ tTrue, npe1, distToReadout);
297 
298  // Time relative to PPS: Random for now! (FIXME)
299  uint32_t ppsTicks =
300  CLHEP::RandFlat::shootInt(&fEngine, /*trigClock.Frequency()*/ fClockSpeedCRT * 1e6);
301 
302  // SiPM and ADC response: Npe to ADC counts
303  uint32_t q0 =
304  CLHEP::RandGauss::shoot(&fEngine, fQPed + fQSlope * npe0, fQRMS * sqrt(npe0));
305  if(q0 > fAdcSaturation) q0 = fAdcSaturation;
306  uint32_t q1 =
307  CLHEP::RandGauss::shoot(&fEngine, fQPed + fQSlope * npe1, fQRMS * sqrt(npe1));
308  if(q1 > fAdcSaturation) q1 = fAdcSaturation;
309 
310  // Adjacent channels on a strip are numbered sequentially.
311  //
312  // In the AuxDetChannelMapAlg methods, channels are identified by an
313  // AuxDet name (retrievable given the hit AuxDet ID) which specifies a
314  // module, and a channel number from 0 to 32.
315  uint32_t moduleID = adsc.AuxDetID();
316  uint32_t stripID = adsc.AuxDetSensitiveID();
317  uint32_t channel0ID = 32 * moduleID + 2 * stripID + 0;
318  uint32_t channel1ID = 32 * moduleID + 2 * stripID + 1;
319 
320  if (moduleID >= 127) {continue;} //Ignoring MINOS modules for now.
321 
322  // Apply ADC threshold and strip-level coincidence (both fibers fire)
323  if (q0 > fQThreshold &&
324  q1 > fQThreshold &&

327  tagger.planesHit.push_back({planeID, t0});
328  tagger.data.push_back(sbnd::crt::CRTData(channel0ID, t0, ppsTicks, q0));
329  tagger.ides.push_back(trueIdes);
330  tagger.data.push_back(sbnd::crt::CRTData(channel1ID, t1, ppsTicks, q1));
331  tagger.ides.push_back(trueIdes);
332  }
333

335  adsGeo.LocalToWorld(origin, poss);
336  mf::LogInfo("CRT")
337  << "CRT HIT in " << adsc.AuxDetID() << "/" << adsc.AuxDetSensitiveID() << "\n"
338  << "CRT HIT POS " << x << " " << y << " " << z << "\n"
339  << "CRT STRIP POS " << poss[0] << " " << poss[1] << " " << poss[2] << "\n"
340  << "CRT MODULE POS " << modulePosMother[0] << " "
341  << modulePosMother[1] << " "
342  << modulePosMother[2] << " "
343  << "\n"
344  << "CRT PATH: " << path << "\n"

346  << "CRT level 1 (array): " << nodeArray->GetName() << "\n"
347  << "CRT level 2 (module): " << nodeModule->GetName() << "\n"
348  << "CRT level 3 (tagger): " << nodeTagger->GetName() << "\n"
349  << "CRT PLANE ID: " << planeID << "\n"

352  }
353  }
