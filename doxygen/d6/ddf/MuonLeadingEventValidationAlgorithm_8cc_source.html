220

225  const CaloHitList &pfoHitList(pfoToHitsMap.at(pfoToSharedHits.first));
226 
227  const bool isGoodMatch(this->IsGoodMatch(mcHitList, pfoHitList, sharedHitList));
228 
229  if (isGoodMatch)

231  }
232 
233  if (nAboveThresholdMatches != 1)
