51  /**
52  * @brief Tool decoding the trigger information from DAQ.
53  *
54  * Produces:
55  * * `std::vector<raw::ExternalTrigger>` containing the absolute trigger time
56  * stamp from the White Rabbit system, and a trigger count;
57  * it always includes a single entry (zero _might_ be supported).
58  * * `std::vector<raw::Trigger>` containing:
59  * * `TriggerTime()`: the relative time of the trigger as reported in the
60  * White Rabbit timestamp, measured in the
61  * @ref DetectorClocksElectronicsTime "electronics time scale" (for
62  * ICARUS it will always be
63  * `detinfo::DetectorClocksData::TriggerTime()`).
64  * * `BeamGateTime()`: relative time of the announced arrival of the beam
65  * (currently not available) also in
66  * @ref DetectorClocksElectronicsTime "electronics time scale".
67  * * `TriggerCount()`: the trigger count from the beginning of the run.
68  * * `TriggerBits()`: includes the beam(s) with an open gate when the
69  * trigger happened (currently only one beam gate per trigger);
70  * definitions are in `sbn::beamType` namespace.
71  *
72  * It always includes a single entry (zero _might_ be supported).
73  * * `std::vector<sim::BeamGateInfo>` containing information on the "main"
74  * beam gate associated to each trigger (a way to say that if by any
75  * chance there are more than one bits set for the trigger, this gate
76  * will pick only one of them):
77  * * `Start()`: relative time of the announced arrival of the beam
78  * (currently not available), in
79  * @ref DetectorClocksSimulationTime "simulation time scale".
80  * * `Width()`: duration of the gate, in nanoseconds; currently set to a
81  * nominal value.
82  * * `BeamType()`: the type of the beam gate being described (BNB, NuMI).
83  * * `sbn::ExtraTriggerInfo`: the most complete collection of information,
84  * duplicating also some from the other data products. Some of the
85  * information is not available yet: if a count is not available, its
86  * value is set to `0` (which is an invalid value because their valid
87  * range starts with `1` since they include the current event), and if a
88  * timestamp is not available it is set to
89  * `sbn::ExtraTriggerInfo::NoTimestamp`; these two conditions can be
90  * checked with static methods
91  * `sbn::ExtraTriggerInfo::isValidTimestamp()` and
92  * `sbn::ExtraTriggerInfo::isValidCount()` respectively.
93  * Note that differently from the usual, this is a _single object_, not
94  * a collection; also, this data product has no instance name.
95  * The information already available includes:
96  * * `sourceType`: the type of beam or trigger source, a value from
97  * `sbn::triggerSource` (equivalent to `raw::Trigger::TriggerBits()`,
98  * but in the form of an enumerator rather than a bit mask).
99  * * `triggerTimestamp`: same as `raw::ExternalTrigger::GetTrigTime()`
100  * (nanoseconds from the Epoch, Coordinated Universal Time).
101  * * `beamGateTimestamp`: absolute time of the beam gate opening as
102  * reported by the trigger hardware, directly comparable with
103  * `triggerTimestamp` (same scale and unit).
104  * * `triggerID`: same as `raw::ExternalTrigger::GetTrigID()`. Should
105  * match the event number.
106  * * `gateID`: the count of gates since the beginning of the run, as
107  * reported by the trigger hardware.
108  *
109  * Besides the main data product (empty instance name) an additional
110  * `std::vector<raw::ExternalTrigger>` data product with instance name
111  * `"previous"` is also produced, which relays the same kind of information
112  * but for the _previous_ trigger. This information also comes from the
113  * trigger DAQ. If no previous trigger is available, this collection will be
114  * empty.
115  *
116  *
117  * Timestamps and corrections
118  * ---------------------------
119  *
120  * The reference trigger time is driven by the trigger fragment time, which
121  * is expected to have been derived from the actual trigger time from the
122  * White Rabbit system properly corrected to UTC by the board reader.
123  *
124  * All absolute timestamps are corrected to be on that same scale.
125  * The absolute timestamps related to the White Rabbit time are added an
126  * offset to achieve this correction; this offset is stored in the data
127  * product (`sbn::ExtraTriggerInfo::WRtimeToTriggerTime`).
128  *
129  */

290  void TriggerDecoder::process_fragment(const artdaq::Fragment &fragment)
291  {
292  // artdaq_ts is reworked by the trigger board reader to match the corrected
293  // trigger time; to avoid multiple (potentially inconsistent) corrections,
294  // the decoder trusts it and references all the times with respect to it.
295  uint64_t const artdaq_ts = fragment.timestamp();
296  icarus::ICARUSTriggerUDPFragment frag { makeTriggerFragment(fragment) };
297  std::string data = frag.GetDataString();
298  char *buffer = const_cast<char*>(data.c_str());
299 
300  icarus::ICARUSTriggerInfo datastream_info = parseTriggerString(buffer);
301  uint64_t const raw_wr_ts // this is raw, unadultered, uncorrected
302  = makeTimestamp(frag.getWRSeconds(), frag.getWRNanoSeconds());
303 
304  // correction (explicitly converted to signed)
305  int64_t const WRtimeToTriggerTime
306  = static_cast<int64_t>(artdaq_ts) - raw_wr_ts;
307  auto const correctWRtime = [WRtimeToTriggerTime](uint64_t time)
308  { return time + WRtimeToTriggerTime; };
309  assert(correctWRtime(raw_wr_ts) == artdaq_ts);
310 
311  // --- END ---- TEMPORARY --------------------------------------------------
312  int gate_type = datastream_info.gate_type;
313  long delta_gates_bnb [[maybe_unused]] = frag.getDeltaGatesBNB();
314  long delta_gates_numi [[maybe_unused]] = frag.getDeltaGatesOther(); //this is actually NuMI due to abrupt changes in trigger board logic
315  long delta_gates_other [[maybe_unused]] = frag.getDeltaGatesNuMI();
316  uint64_t lastTrigger = 0;
317 
318  // --- BEGIN -- TEMPORARY --------------------------------------------------
319  // remove this part when the beam gate timestamp is available via fragment
320  // or via the parser
321  auto const parsedData = icarus::details::KeyedCSVparser{}(firstLine(data));
322  unsigned int beamgate_count { std::numeric_limits<unsigned int>::max() };
323  std::uint64_t beamgate_ts { artdaq_ts }; // we cheat
324  /* [20210717, petrillo@slac.stanford.edu] `(pBeamGateInfo->nValues() == 3)`:
325  * this is an attempt to "support" a few Run0 runs (6017 to roughly 6043)
326  * which have the beam gate information truncated; this workaround should
327  * be removed when there is enough ICARUS data that these runs become
328  * uninteresting.
329  */
330  if (auto pBeamGateInfo = parsedData.findItem("Beam_TS");
331  pBeamGateInfo && (pBeamGateInfo->nValues() == 3)
332  ) {
333  // if gate information is found, it must be complete
334  beamgate_count = pBeamGateInfo->getNumber<unsigned int>(0U);
335 
336  uint64_t const raw_bg_ts = makeTimestamp( // raw and uncorrected too
337  pBeamGateInfo->getNumber<unsigned int>(1U),
338  pBeamGateInfo->getNumber<unsigned int>(2U)
339  );
340 
341  // assuming the raw times from the fragment are on the same time scale
342  // (same offset corrections)
343  beamgate_ts += raw_bg_ts - raw_wr_ts;
344 
345  } // if has gate information
346  // --- END ---- TEMPORARY --------------------------------------------------
347 
348  if(fDiagnosticOutput || fDebug)
349  {
350  std::cout << "Full Timestamp = " << artdaq_ts
351  << "\nBeam gate " << beamgate_count << " at "
352  << (beamgate_ts/1'000'000'000) << "." << std::setfill('0')
353  << std::setw(9) << (beamgate_ts%1'000'000'000) << std::setfill(' ')
354  << " s (" << timestampDiff(beamgate_ts, artdaq_ts)
355  << " ns relative to trigger)" << std::endl;
356 
357  // note that this parsing is independent from the one used above
358  std::string_view const dataLine = firstLine(data);
359  try {
360  auto const parsedData = icarus::details::KeyedCSVparser{}(dataLine);
361  std::cout << "Parsed data (from " << dataLine.size() << " characters): "
362  << parsedData << std::endl;
363  }
364  catch(icarus::details::KeyedCSVparser::Error const& e) {
365  mf::LogError("TriggerDecoder")
366  << "Error parsing " << dataLine.length()
367  << "-char long trigger string:\n==>|" << dataLine
368  << "|<==\nError message: " << e.what() << std::endl;
369  throw;
370  }
371 
372  if (fDebug) { // this grows tiresome quickly when processing many events
373  std::cout << "Trigger packet content:\n" << dataLine
374  << "\nFull trigger fragment dump:"
375  << sbndaq::dumpFragment(fragment) << std::endl;
376  }
377  }
378 
379  //
380  // extra trigger info
381  //
382  sbn::triggerSource beamGateBit;
383  switch (gate_type) {
384  case TriggerGateTypes::BNB: beamGateBit = sbn::triggerSource::BNB; break;
385  case TriggerGateTypes::NuMI: beamGateBit = sbn::triggerSource::NuMI; break;
386  case TriggerGateTypes::OffbeamBNB: beamGateBit = sbn::triggerSource::OffbeamBNB; break;
387  case TriggerGateTypes::OffbeamNuMI: beamGateBit = sbn::triggerSource::OffbeamNuMI; break;
388  default: beamGateBit = sbn::triggerSource::Unknown;
389  } // switch gate_type
390 
391  fTriggerExtra->sourceType = beamGateBit;
392  fTriggerExtra->triggerTimestamp = artdaq_ts;
393  fTriggerExtra->beamGateTimestamp = beamgate_ts;
394  fTriggerExtra->triggerID = datastream_info.wr_event_no;
395  fTriggerExtra->gateID = datastream_info.gate_id;
396  /* TODO (may need to add WRtimeToTriggerTime to some timestamps):
397  fTriggerExtra->triggerCount
398  fTriggerExtra->gateCount
399  fTriggerExtra->gateCountFromPreviousTrigger
400  fTriggerExtra->anyTriggerCountFromPreviousTrigger
401  fTriggerExtra->anyGateCountFromPreviousTrigger
402  fTriggerExtra->anyPreviousTriggerSourceType
403  fTriggerExtra->anyGateCountFromAnyPreviousTrigger
404  fTriggerExtra->previousTriggerTimestamp
405  fTriggerExtra->anyPreviousTriggerTimestamp
406  */
407  fTriggerExtra->WRtimeToTriggerTime = WRtimeToTriggerTime;
408 
409  //
410  // absolute time trigger (raw::ExternalTrigger)
411  //
412  fTrigger->emplace_back
413  (fTriggerExtra->triggerID, fTriggerExtra->triggerTimestamp);
414 
415  //
416  // previous absolute time trigger (raw::ExternalTrigger)
417  //
418  if(fTriggerExtra->triggerID == 1)
419  {
420  fLastEvent = 0;
421  }
422  else
423  {
424  fLastEvent = fTriggerExtra->triggerID - 1;
425  lastTrigger = frag.getLastTimestampBNB();
426  fPrevTrigger->emplace_back(fLastEvent, lastTrigger);
427  }
428 
429  //
430  // beam gate
431  //
432  // beam gate - trigger: hope it's negative...
433  nanoseconds const gateStartFromTrigger{
434  static_cast<double>(timestampDiff
435  (fTriggerExtra->beamGateTimestamp, fTriggerExtra->triggerTimestamp)
436  )
437  }; // narrowing!!
438  auto const elecGateStart = fDetTimings.TriggerTime() + gateStartFromTrigger;
439  auto const simGateStart = fDetTimings.toSimulationTime(elecGateStart);
440  switch (gate_type) {
441  case TriggerGateTypes::BNB:
442  fBeamGateInfo->emplace_back
443  (simGateStart.value(), BNBgateDuration.value(), sim::kBNB);
444  break;
445  case TriggerGateTypes::NuMI:
446  fBeamGateInfo->emplace_back
447  (simGateStart.value(), NuMIgateDuration.value(), sim::kNuMI);
448  break;
449  case TriggerGateTypes::OffbeamBNB:
450  fBeamGateInfo->emplace_back
451  (simGateStart.value(), BNBgateDuration.value(), sim::kBNB);
452  break;
453  case TriggerGateTypes::OffbeamNuMI:
454  fBeamGateInfo->emplace_back
455  (simGateStart.value(), NuMIgateDuration.value(), sim::kNuMI);
456  break;
457  default:
458  mf::LogWarning("TriggerDecoder") << "Unsupported gate type #" << gate_type;
459  } // switch gate_type
460 
461  //
462  // relative time trigger (raw::Trigger)
463  //
464  fRelTrigger->emplace_back(
465  static_cast<unsigned int>(datastream_info.wr_event_no), // counter
466  fDetTimings.TriggerTime().value(), // trigger_time
467  elecGateStart.value(), // beamgate_time
468  mask(beamGateBit) // bits
469  );
470 
471  //Once we have full trigger data object, set up and place information into there
472  return;
473  }
