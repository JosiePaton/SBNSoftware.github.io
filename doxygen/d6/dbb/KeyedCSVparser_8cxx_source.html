89 void icarus::details::KeyedCSVparser::parse
90  (std::string_view const& s, ParsedData_t& data) const
91 {
92 
93  auto stream = s;
94 
95  ParsedData_t::Item* currentItem = nullptr;
96 
97  // this many tokens will be assigned to the current key:
98  int forcedValues = -1; // 0 would force the first entry to be a key
99 
100  while (!stream.empty()) {
101 
102  auto const token = extractToken(stream);
103 
104  std::string tokenStr { cbegin(token), cend(token) };
105 
106  bool bKey = false;
107  do {
108 
109  // if there are values pending, this is not a key, period;
110  // if all required values have been assigned, the next token is a key.
111  if (forcedValues >= 0) {
112  bKey = (forcedValues == 0); // if no more forced values, next is key
113  --forcedValues;
114  // if we know the token is a key, we still need to check for matching
115  // patterns to assign required values
116  if (!bKey) break;
117  }
118 
119  // the token may still be a key (if `bKey` is true, it is for sure: we can
120  // decide that a non-key (!bKey) is actually a key, but not the opposite)
121  for (auto const& [ pattern, values ]: fPatterns) {
122  if (!std::regex_match(begin(token), end(token), pattern)) continue;
123  bKey = true; // matching a pattern implies this is a key
124  std::string const& key = tokenStr;
125  // how many values to expect:
126  switch (values) {
127  case FixedSize: // read the next token immediately as fixed size
128  {
129  if (stream.empty()) throw MissingSize(key);
130 
131  auto const sizeToken = peekToken(stream);
132  if (empty(sizeToken)) throw MissingSize(key);
133 
134  // the value is loaded in `forcedValues` and already excludes
135  // the size token just read
136  char const *b = begin(sizeToken), *e = end(sizeToken);
137  if (std::from_chars(b, e, forcedValues).ptr != e)
138  throw MissingSize(key, std::string{ sizeToken });
139 
140  ++forcedValues; // the size will be forced in the values anyway
141 
142  } // FixedSize
143  break;
144  case DynamicSize:
145  // nothing to do, the normal algorithm rules will follow
146  break;
147  default:
148  forcedValues = values;
149  break;
150  } // switch
151  break;
152  } // for pattern
153  if (bKey) break;
154 
155  // let the "standard" pattern decide
156  bKey = isKey(token);
157 
158  } while (false);
159 
160  if (bKey) currentItem = &(data.makeItem(std::move(tokenStr)));
161  else {
162  if (!currentItem) {
163  throw InvalidFormat(
164  "values started without a key ('" + tokenStr + "' is not a valid key)."
165  );
166  }
167  currentItem->addValue(std::move(tokenStr));
168  }
169 
170  } // while
171 
172  if (forcedValues > 0) {
173  assert(currentItem);
174  throw MissingValues(currentItem->key(), forcedValues);
175  }
176 
177 } // icarus::KeyedCSVparser::parse()
